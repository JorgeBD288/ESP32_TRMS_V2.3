812 for(uint8_t j = 0; j < 3; j++) {
813 lv_pinyin_cand_str[i][j] = pinyin_ime->cand_str[i * 3 + j];

820static void pinyin_page_proc(lv_obj_t * obj, uint16_t dir)
821{
822 lv_ime_pinyin_t * pinyin_ime = (lv_ime_pinyin_t *)obj;
823 uint16_t page_num = pinyin_ime->cand_num / LV_IME_PINYIN_CAND_TEXT_NUM;
824 uint16_t sur = pinyin_ime->cand_num % LV_IME_PINYIN_CAND_TEXT_NUM;
825
826 if(dir == 0) {
827 if(pinyin_ime->py_page) {
828 pinyin_ime->py_page--;
829 }
830 }
831 else {
832 if(sur == 0) {
833 page_num -= 1;
834 }
835 if(pinyin_ime->py_page < page_num) {
836 pinyin_ime->py_page++;
837 }
838 else return;
839 }
840
841 for(uint8_t i = 0; i < LV_IME_PINYIN_CAND_TEXT_NUM; i++) {
842 memset(lv_pinyin_cand_str[i], 0x00, sizeof(lv_pinyin_cand_str[i]));
843 lv_pinyin_cand_str[i][0] = ' ';
844 }
845
846 // fill buf
847 uint16_t offset = pinyin_ime->py_page * (3 * LV_IME_PINYIN_CAND_TEXT_NUM);
848 for(uint8_t i = 0; (i < pinyin_ime->cand_num && i < LV_IME_PINYIN_CAND_TEXT_NUM); i++) {
849 if((sur > 0) && (pinyin_ime->py_page == page_num)) {
850 if(i > sur)
851 break;
852 }
853 for(uint8_t j = 0; j < 3; j++) {
854 lv_pinyin_cand_str[i][j] = pinyin_ime->cand_str[offset + (i * 3) + j];
855 }
856 }
857}

874static void init_pinyin_dict(lv_obj_t * obj, lv_pinyin_dict_t * dict)
875{
876 lv_ime_pinyin_t * pinyin_ime = (lv_ime_pinyin_t *)obj;
877
878 char headletter = 'a';
879 uint16_t offset_sum = 0;
880 uint16_t offset_count = 0;
881 uint16_t letter_calc = 0;
882
883 pinyin_ime->dict = dict;
884
885 for(uint16_t i = 0; ; i++) {
886 if((NULL == (dict[i].py)) || (NULL == (dict[i].py_mb))) {
887 headletter = dict[i - 1].py[0];
888 letter_calc = headletter - 'a';
889 pinyin_ime->py_num[letter_calc] = offset_count;
890 break;
891 }
892
893 if(headletter == (dict[i].py[0])) {
894 offset_count++;
895 }
896 else {
897 headletter = dict[i].py[0];
898 letter_calc = headletter - 'a';
899 pinyin_ime->py_num[letter_calc - 1] = offset_count;
900 offset_sum += offset_count;
901 pinyin_ime->py_pos[letter_calc] = offset_sum;
902
903 offset_count = 1;
904 }
905 }

909static char * pinyin_search_matching(lv_obj_t * obj, char * py_str, uint16_t * cand_num)
910{
911 lv_ime_pinyin_t * pinyin_ime = (lv_ime_pinyin_t *)obj;
912
913 lv_pinyin_dict_t * cpHZ;
914 uint8_t index, len = 0, offset;
915 volatile uint8_t count = 0;
916
917 if(*py_str == '\0') return NULL;
918 if(*py_str == 'i') return NULL;
919 if(*py_str == 'u') return NULL;
920 if(*py_str == 'v') return NULL;
921
922 offset = py_str[0] - 'a';
923 len = strlen(py_str);
924
925 cpHZ = &pinyin_ime->dict[pinyin_ime->py_pos[offset]];
926 count = pinyin_ime->py_num[offset];
927
928 while(count--) {
929 for(index = 0; index < len; index++) {
930 if(*(py_str + index) != *((cpHZ->py) + index)) {
931 break;
932 }
933 }
934
935 // perfect match
936 if(len == 1 || index == len) {
937 // The Chinese character in UTF-8 encoding format is 3 bytes
938 * cand_num = strlen((const char *)(cpHZ->py_mb)) / 3;
939 return (char *)(cpHZ->py_mb);
940 }
941 cpHZ++;
942 }
943 return NULL;
944}
