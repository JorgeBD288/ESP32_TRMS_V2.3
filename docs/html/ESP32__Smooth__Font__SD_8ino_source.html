1/*
2 Sketch to demonstrate using the print class with smooth fonts
3 that are saved onto an SD Card accessed by the SD library.
4
5 For ESP32 only, GPIO 5 must be used for SD chip select.
6 This method of storing the fonts is NOT compatible with the ESP8266.
7
8 Sketch is written for a 240 x 320 display
9
10 Load the font file onto the root directory of the SD Card. The font files
11 used by this sketch can be found in the Data folder, press Ctrl+K to see it.
12
13 The library supports 16-bit Unicode characters:
14 https://en.wikipedia.org/wiki/Unicode_font
15
16 The characters supported are in the in the Basic Multilingual Plane:
17 https://en.wikipedia.org/wiki/Plane_(Unicode)#Basic_Multilingual_Plane
18
19 Make sure all the display driver and pin connections are correct by
20 editing the User_Setup.h file in the TFT_eSPI library folder.
21
22 #########################################################################
23 ###### DON'T FORGET TO UPDATE THE User_Setup.h FILE IN THE LIBRARY ######
24 #########################################################################
25*/
26

36// -------------------------------------------------------------------------
37// Setup
38// -------------------------------------------------------------------------
39void setup(void) {
40 Serial.begin(115200); // Used for messages
41
42 // Initialise the SD library before the TFT so the chip select is defined
43 if (!SD.begin()) {
44 Serial.println("Card Mount Failed");
45 return;
46 }
47 uint8_t cardType = SD.cardType();
48
49 if (cardType == CARD_NONE) {
50 Serial.println("No SD card attached");
51 return;
52 }
53
54 Serial.print("SD Card Type: ");
55 if (cardType == CARD_MMC) {
56 Serial.println("MMC");
57 } else if (cardType == CARD_SD) {
58 Serial.println("SDSC");
59 } else if (cardType == CARD_SDHC) {
60 Serial.println("SDHC");
61 } else {
62 Serial.println("UNKNOWN");
63 }
64
65 uint64_t cardSize = SD.cardSize() / (1024 * 1024);
66 Serial.printf("SD Card Size: %lluMB\n", cardSize);
67
68 // Initialise the TFT after the SD card!
69 tft.init();
70 tft.setRotation(1);
71 tft.fillScreen(TFT_BLACK);
72
73 listDir(SD, "/", 0);
74
75 Serial.println("SD and TFT initialisation done.");
76}

78// -------------------------------------------------------------------------
79// Main loop
80// -------------------------------------------------------------------------
81void loop() {
82 // Wrap test at right and bottom of screen
83 tft.setTextWrap(true, true);
84
85 // Name of font file (library adds leading / and .vlw)
86 String fileName = "Final-Frontier-28";
87
88 // Font and background colour, background colour is used for anti-alias blending
89 tft.setTextColor(TFT_WHITE, TFT_BLACK);
90
91 // Load the font
92 tft.loadFont(fileName, SD); // Use font stored on SD
93
94 // Display all characters of the font
95 tft.showFont(2000);
96
97 uint32_t dt = millis();
98
99 int count = 100;
100
101 while (count--)
102 {
103 // Set "cursor" at top left corner of display (0,0)
104 // (cursor will move to next line automatically during printing with 'tft.println'
105 // or stay on the line is there is room for the text with tft.print)
106 tft.setCursor(0, 0);
107
108 // Set the font colour to be white with a black background, set text size multiplier to 1
109 tft.setTextColor(TFT_WHITE, TFT_BLACK);
110
111 // We can now plot text on screen using the "print" class
112 tft.println("Hello World!");
113
114 // Set the font colour to be yellow
115 tft.setTextColor(TFT_YELLOW, TFT_BLACK);
116 tft.println(1234.56);
117
118 // Set the font colour to be red
119 tft.setTextColor(TFT_RED, TFT_BLACK);
120 tft.println((uint32_t)3735928559, HEX); // Should print DEADBEEF
121
122 // Set the font colour to be green with black background
123 tft.setTextColor(TFT_GREEN, TFT_BLACK);
124 tft.println("Anti-aliased font!");
125 tft.println("");
126
127 // Test some print formatting functions
128 float fnumber = 123.45;
129
130 // Set the font colour to be blue
131 tft.setTextColor(TFT_BLUE, TFT_BLACK);
132 tft.print("Float = "); tft.println(fnumber); // Print floating point number
133 tft.print("Binary = "); tft.println((int)fnumber, BIN); // Print as integer value in binary
134 tft.print("Hexadecimal = "); tft.println((int)fnumber, HEX); // Print as integer number in Hexadecimal
135 }
136
137 Serial.println(millis()-dt);
138
139 // Unload the font to recover used RAM
140 tft.unloadFont();
141
142 delay(10000);
143}

145void listDir(fs::FS &fs, const char * dirname, uint8_t levels){
146 Serial.printf("Listing directory: %s\n", dirname);
147
148 File root = fs.open(dirname);
149 if(!root){
150 Serial.println("Failed to open directory");
151 return;
152 }
153 if(!root.isDirectory()){
154 Serial.println("Not a directory");
155 return;
156 }
157
158 File file = root.openNextFile();
159 while(file){
160 if(file.isDirectory()){
161 Serial.print(" DIR : ");
162 Serial.println(file.name());
163 if(levels){
164 listDir(fs, file.name(), levels -1);
165 }
166 } else {
167 Serial.print(" FILE: ");
168 Serial.print(file.name());
169 Serial.print(" SIZE: ");
170 Serial.println(file.size());
171 }
172 file = root.openNextFile();
173 }
