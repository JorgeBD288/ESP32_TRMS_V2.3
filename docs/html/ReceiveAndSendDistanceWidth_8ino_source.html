1/*
2 * ReceiveAndSendDistanceWidth.cpp
3 *
4 * Record and play back last received distance width IR signal at button press.
5 * Using DistanceWidthProtocol covers a lot of known and unknown IR protocols,
6 * and requires less memory than raw protocol.
7 *
8 * The logic is:
9 * If the button is pressed, send the IR code.
10 * If an IR code is received, record it.
11 *
12 * An example for simultaneous receiving and sending is in the UnitTest example.
13 *
14 * An IR detector/demodulator must be connected to the input IR_RECEIVE_PIN.
15 *
16 * A button must be connected between the input SEND_BUTTON_PIN and ground.
17 * A visible LED can be connected to STATUS_PIN to provide status.
18 *
19 * See also https://dronebotworkshop.com/ir-remotes/#ReceiveAndSendDistanceWidth_Code

88void setup() {
89 pinMode(SEND_BUTTON_PIN, INPUT_PULLUP);

91 Serial.begin(115200);
92 while (!Serial)
93 ; // Wait for Serial to become available. Is optimized away for some cores.
94
95#if defined(__AVR_ATmega32U4__) || defined(SERIAL_PORT_USBVIRTUAL) || defined(SERIAL_USB) /*stm32duino*/|| defined(USBCON) /*STM32_stm32*/ \

97 delay(4000); // To be able to connect Serial monitor after reset or power up and before first print out. Do not wait for an attached Serial Monitor!

100 Serial.println(F("START " __FILE__ " from " __DATE__ "\r\nUsing library version " VERSION_IRREMOTE));

102 // Start the receiver and if not 3. parameter specified, take LED_BUILTIN pin from the internal boards definition as default feedback LED
103 IrReceiver.begin(IR_RECEIVE_PIN, ENABLE_LED_FEEDBACK);
104 Serial.println(F("Ready to receive pulse distance/width coded IR signals at pin " STR(IR_RECEIVE_PIN)));
105
106 IrSender.begin(); // Start with IR_SEND_PIN -which is defined in PinDefinitionsAndMore.h- as send pin and enable feedback LED at default feedback LED pin
107 Serial.print(F("Ready to send IR signals at pin " STR(IR_SEND_PIN) " on press of button at pin "));

111void loop() {
112
113 // If button pressed, send the code.
114 bool tSendButtonIsActive = (digitalRead(SEND_BUTTON_PIN) == LOW); // Button pin is active LOW
115
116 /*
117 * Check for current button state
118 */
119 if (tSendButtonIsActive) {
120 if (!sSendButtonWasActive) {
121 Serial.println(F("Stop receiving"));
122 IrReceiver.stop();
123 }
124 /*
125 * Button pressed -> send stored data
126 */
127 Serial.print(F("Button pressed, now sending "));
128 Serial.print(sNumberOfBits);
129 Serial.print(F(" bits 0x"));
130 Serial.print(sDecodedRawDataArray[0], HEX);
131 Serial.print(F(" with sendPulseDistanceWidthFromArray timing="));
132 IrReceiver.printDistanceWidthTimingInfo(&Serial, &sDistanceWidthTimingInfo);
133 Serial.println();
134 Serial.flush(); // To avoid disturbing the software PWM generation by serial output interrupts
135
136 IrSender.sendPulseDistanceWidthFromArray(38, &sDistanceWidthTimingInfo, &sDecodedRawDataArray[0], sNumberOfBits,
137#if defined(USE_MSB_DECODING_FOR_DISTANCE_DECODER)
138 PROTOCOL_IS_MSB_FIRST
139#else
140 PROTOCOL_IS_LSB_FIRST
141#endif
142 , 100, 0);
143
144 delay(DELAY_BETWEEN_REPEATS_MILLIS); // Wait a bit between retransmissions
145
146 } else if (sSendButtonWasActive) {
147 /*
148 * Button is just released -> activate receiving
149 */
150 // Restart receiver
151 Serial.println(F("Button released -> start receiving"));
152 IrReceiver.start();
153
154 } else if (IrReceiver.decode()) {
155 /*
156 * Button is not pressed and data available -> store received data and resume
157 * DistanceWidthTimingInfo and sNumberOfBits should be constant for all keys of the same IR remote / protocol
158 */
159 IrReceiver.printIRResultShort(&Serial);
160 if (IrReceiver.decodedIRData.protocol != UNKNOWN) {
161 IrReceiver.printIRSendUsage(&Serial);
162
163 if (memcmp(&sDistanceWidthTimingInfo, &IrReceiver.decodedIRData.DistanceWidthTimingInfo,
164 sizeof(sDistanceWidthTimingInfo)) != 0) {
165 Serial.print(F("Store new timing info data="));
166 IrReceiver.printDistanceWidthTimingInfo(&Serial, &IrReceiver.decodedIRData.DistanceWidthTimingInfo);
167 Serial.println();
168 sDistanceWidthTimingInfo = IrReceiver.decodedIRData.DistanceWidthTimingInfo; // copy content here
169 } else {
170 Serial.print(F("Timing did not change, so we can reuse already stored timing info."));
171 }
172 if (sNumberOfBits != IrReceiver.decodedIRData.numberOfBits) {
173 Serial.print(F("Store new numberOfBits="));
174 sNumberOfBits = IrReceiver.decodedIRData.numberOfBits;
175 Serial.println(IrReceiver.decodedIRData.numberOfBits);
176 }
177 if (sDecodedRawDataArray[0] != IrReceiver.decodedIRData.decodedRawDataArray[0]) {
178 *sDecodedRawDataArray = *IrReceiver.decodedIRData.decodedRawDataArray; // copy content here
179 Serial.print(F("Store new sDecodedRawDataArray[0]=0x"));
180 Serial.println(IrReceiver.decodedIRData.decodedRawDataArray[0], HEX);
181 }
182 }
183 IrReceiver.resume(); // resume receiver
184 }
185
186 sSendButtonWasActive = tSendButtonIsActive;
187 delay(100);
188}
