14static float adcToVolts(uint16_t raw)
15{
16 return (raw / 4095.0f) * 3.3f; // ADC 12 bits → 0..3.3 V
17}
18
19/**
20 * @brief El voltaje que recibe el ESP32 es un voltaje adaptado después de pasar por un circuito de adaptación.
21 * Esta función permite obtener el valor de voltaje real, generado por el tacómetro
22 */
23static float voltsAdapter(float volts)
24{
25 return (2*(1.65-volts));
26}

28/**
29 * @brief Convierte voltios → RPM según la curva del tacómetro
30 */
31static float voltsToRpm(float volts)
32{
33 if (s_cfg.voltsPer1000RPM <= 0.0f) return 0.0f;
34
35 return (volts / s_cfg.voltsPer1000RPM) * 1000.0f;
36}
