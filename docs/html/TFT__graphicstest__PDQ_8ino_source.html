219void printnice(int32_t v)
220{
221 char str[32] = { 0 };
222 sprintf(str, "%d", v);
223 for (char *p = (str+strlen(str))-3; p > str; p -= 3)
224 {
225 memmove(p+1, p, strlen(p)+1);
226 *p = ',';
227
228 }
229 while (strlen(str) < 10)
230 {
231 memmove(str+1, str, strlen(str)+1);
232 *str = ' ';
233 }
234 tft.println(str);
235}
