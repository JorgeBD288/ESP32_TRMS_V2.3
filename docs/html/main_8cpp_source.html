261void loop() {
262 // ---------------------------
263 // 1) Gestionar LVGL (display + táctil)
264 // ---------------------------
265 DisplayTouch_taskHandler();
266 delay(5);
267
268 // ---------------------------
269 // 2) Detectar actividad por TÁCTIL
270 // (cualquier toque en la pantalla cuenta como actividad)
271 // ---------------------------
272 uint16_t tx, ty;
273 if (tft.getTouch(&tx, &ty)) {
274 RegisterActivity(); // actualiza g_lastActivityMs
275 }
276
277 // ---------------------------
278 // 3) Tacómetros: actualizar RPM cada 100 ms
279 // ---------------------------
280 static uint32_t lastTachoUpdate = 0;
281 uint32_t now = millis();
282 if (now - lastTachoUpdate > 100) {
283 Tacho_update();
284 lastTachoUpdate = now;
285 }
286
287 //Elimininar mensaje de confirmación de carga de datos desde SD en caso de que esta se haya producido, pasado un tiempo
288 if (flag_Config_Message == true){
289 Show_Config_Message_Selected();
290 }
291
292 // ---------------------------
293 // 4) IR: leer mando y actuar
294 // ---------------------------
295 IRControlEvent ev = IRControl_poll();
296 if (ev.hasEvent) {
297
298 // 4.1) Primero: si estamos en modo aprendizaje de la pantalla del mando,
299 // que consuma este evento y no haga nada más.
300 if (RemoteDiagram_HandleIRLearn(ev)) {
301 // Solo se ha usado para programar un botón → no navegar, ni sliders, etc.
302 return;
303 }
304
305 // 4.2) Cambiar de pantalla si procede
306 lv_obj_t *scr_act = lv_scr_act();
307 switch (ev.screen) {
308 case IRScreenTarget::SCREEN1:
309 // Solo cambiar si NO estamos en la pantalla 9
310 if (scr_act != ui_Screen9) {
311 _ui_screen_change(
312 &ui_Screen1,
313 LV_SCR_LOAD_ANIM_FADE_ON,
314 500, 0,
315 &ui_Screen1_screen_init
316 );
317 // Habilitar control con mando a distancia en la pantalla 1
318 SetupScreen1Nav();
319 }
320 break;
321
322 case IRScreenTarget::NONE:
323 default:
324 break;
325 }
326
327 // 4.3) Navegación con flechas
328 if (ev.nav != IRNavKey::NONE) {
329 HandleIRNavigation(ev.nav);
330 }
331
332 // 4.4) Control de sliders
333 if (ev.deltaSlider != 0) {
334 HandleDeltaSlider(ev.deltaSlider);
335 }
336
337 // 4.5) Entrada numérica en labels
338 if (ev.digit >= 0) {
339 HandleNumericDigit(ev.digit);
340 }
341
342 // 4.6) Signo negativo
343 if (ev.minus) {
344 HandleNumericMinus();
345 }
346 }
347
348
349 // ---------------------------
350 // 5) Comprobar INACTIVIDAD y activar salvapantallas (Screen10)
351 // ---------------------------
352 now = millis();
353 if (now - g_lastActivityMs > INACTIVITY_TIMEOUT_MS) {
354 lv_obj_t *act = lv_scr_act();
355
356 // Si no estamos ya en Screen10, cambiamos
357 if (act != ui_Screen10) {
358
359 // 1) Guardar de qué pantalla venimos
360 if (act == ui_Screen1) g_prevScreenId = SCR_1;
361 else if (act == ui_Screen2) g_prevScreenId = SCR_2;
362 else if (act == ui_Screen3) g_prevScreenId = SCR_3;
363 else if (act == ui_Screen4) g_prevScreenId = SCR_4;
364 else if (act == ui_Screen5) g_prevScreenId = SCR_5;
365 else if (act == ui_Screen6) g_prevScreenId = SCR_6;
366 else if (act == ui_Screen7) g_prevScreenId = SCR_7;
367 else if (act == ui_Screen8) g_prevScreenId = SCR_8;
368 else if (act == ui_Screen9) g_prevScreenId = SCR_9;
369 else g_prevScreenId = SCR_NONE;
370
371 // 2) Activar flag de salvapantallas
372 g_screensaverActive = true;
373
374 // 3) Ir a Screen10 (salvapantallas)
375 _ui_screen_change(
376 &ui_Screen10,
377 LV_SCR_LOAD_ANIM_FADE_ON,
378 500, 0,
379 &ui_Screen10_screen_init
380 );
381 }
382 }
