45#if defined(APPLICATION_PIN)
46#define RELAY_PIN APPLICATION_PIN
47#else
48#define RELAY_PIN 5

51void setup() {
52 pinMode(LED_BUILTIN, OUTPUT);
53 pinMode(RELAY_PIN, OUTPUT);
54
55 Serial.begin(115200);
56 while (!Serial)
57 ; // Wait for Serial to become available. Is optimized away for some cores.
58
59#if defined(__AVR_ATmega32U4__) || defined(SERIAL_PORT_USBVIRTUAL) || defined(SERIAL_USB) /*stm32duino*/|| defined(USBCON) /*STM32_stm32*/ \
60 || defined(SERIALUSB_PID) || defined(ARDUINO_ARCH_RP2040) || defined(ARDUINO_attiny3217)
61 delay(4000); // To be able to connect Serial monitor after reset or power up and before first print out. Do not wait for an attached Serial Monitor!
62#endif
63 // Just to know which program is running on my Arduino
64 Serial.println(F("START " __FILE__ " from " __DATE__ "\r\nUsing library version " VERSION_IRREMOTE));
65
66 // Start the receiver and if not 3. parameter specified, take LED_BUILTIN pin from the internal boards definition as default feedback LED
67 IrReceiver.begin(IR_RECEIVE_PIN, ENABLE_LED_FEEDBACK);
68
69 Serial.print(F("Ready to receive IR signals of protocols: "));
70 printActiveIRProtocols(&Serial);
71 Serial.println(F("at pin " STR(IR_RECEIVE_PIN)));
72}
73
74int on = 0;
75unsigned long last = millis();
76
77void loop() {
78 if (IrReceiver.decode()) {
79 // If it's been at least 1/4 second since the last
80 // IR received, toggle the relay
81 if (millis() - last > 250) {
82 on = !on;
83 Serial.print(F("Switch relay "));
84 if (on) {
85 digitalWrite(RELAY_PIN, HIGH);
86 Serial.println(F("on"));
87 } else {
88 digitalWrite(RELAY_PIN, LOW);
89 Serial.println(F("off"));
90 }
91
92#if FLASHEND >= 0x3FFF // For 16k flash or more, like ATtiny1604
93 IrReceiver.printIRResultShort(&Serial);
94 IrReceiver.printIRSendUsage(&Serial);
95 Serial.println();
96 if (IrReceiver.decodedIRData.protocol == UNKNOWN) {
97 // We have an unknown protocol, print more info
98 Serial.println(F("Received noise or an unknown (or not yet enabled) protocol"));
99 IrReceiver.printIRResultRawFormatted(&Serial, true);
100 }
101#else
102 // Print a minimal summary of received data
103 IrReceiver.printIRResultMinimal(&Serial);
104 Serial.println();
105#endif // FLASHEND
106 }
107 last = millis();
108 IrReceiver.resume(); // Enable receiving of the next value
109 }
