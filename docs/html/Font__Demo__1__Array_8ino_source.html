1/*
2 This sketch is the same as the Font_Demo_1 example, except the fonts in this
3 example are in a FLASH (program memory) array. This means that processors
4 such as the STM32 series that are not supported by a SPIFFS library can use
5 smooth (anti-aliased) fonts.
6*/
7
8/*
9 There are four different methods of plotting anti-aliased fonts to the screen.
10
11 This sketch uses method 1, using tft.print() and tft.println() calls.
12
13 In some cases the sketch shows what can go wrong too, so read the comments!
14
15 The font is rendered WITHOUT a background, but a background colour needs to be
16 set so the anti-aliasing of the character is performed correctly. This is because
17 characters are drawn one by one.
18
19 This method is good for static text that does not change often because changing
20 values may flicker. The text appears at the tft cursor coordinates.
21
22 It is also possible to "print" text directly into a created sprite, for example using
23 spr.println("Hello"); and then push the sprite to the screen. That method is not
24 demonstrated in this sketch.
25
26*/
27

57void loop() {
58
59 tft.fillScreen(TFT_BLACK);
60
61 tft.setTextColor(TFT_WHITE, TFT_BLACK); // Set the font colour AND the background colour
62 // so the anti-aliasing works
63
64 tft.setCursor(0, 0); // Set cursor at top left of screen
65
66
67 // >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
68 // Small font
69 // >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
70 Serial.println("Loading font");
71
72 tft.loadFont(AA_FONT_SMALL); // Must load the font first
73
74 tft.println("Small 15pt font"); // println moves cursor down for a new line
75
76 tft.println(); // New line
77
78 tft.print("ABC"); // print leaves cursor at end of line
79
80 tft.setTextColor(TFT_CYAN, TFT_BLACK);
81 tft.println("1234"); // Added to line after ABC
82
83 tft.setTextColor(TFT_YELLOW, TFT_BLACK);
84 // print stream formatting can be used,see:
85 // https://www.arduino.cc/en/Serial/Print
86 int ivalue = 1234;
87 tft.println(ivalue); // print as an ASCII-encoded decimal
88 tft.println(ivalue, DEC); // print as an ASCII-encoded decimal
89 tft.println(ivalue, HEX); // print as an ASCII-encoded hexadecimal
90 tft.println(ivalue, OCT); // print as an ASCII-encoded octal
91 tft.println(ivalue, BIN); // print as an ASCII-encoded binary
92
93 tft.println(); // New line
94 tft.setTextColor(TFT_MAGENTA, TFT_BLACK);
95 float fvalue = 1.23456;
96 tft.println(fvalue, 0); // no decimal places
97 tft.println(fvalue, 1); // 1 decimal place
98 tft.println(fvalue, 2); // 2 decimal places
99 tft.println(fvalue, 5); // 5 decimal places
100
101 delay(5000);
102
103 // Get ready for the next demo while we have this font loaded
104 tft.fillScreen(TFT_BLACK);
105 tft.setCursor(0, 0); // Set cursor at top left of screen
106 tft.setTextColor(TFT_WHITE, TFT_BLACK);
107 tft.println("Wrong and right ways to");
108 tft.println("print changing values...");
109
110 tft.unloadFont(); // Remove the font to recover memory used
111
112
113 // >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
114 // Large font
115 // >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
116
117 tft.loadFont(AA_FONT_LARGE); // Load another different font
118
119 //tft.fillScreen(TFT_BLACK);
120
121 // Draw changing numbers - does not work unless a filled rectangle is drawn over the old text
122 for (int i = 0; i <= 99; i++)
123 {
124 tft.setCursor(50, 50);
125 tft.setTextColor(TFT_GREEN, TFT_BLACK); // TFT_BLACK is used for anti-aliasing only
126 // By default background fill is off
127 tft.print(" "); // Overprinting old number with spaces DOES NOT WORK!
128 tft.setCursor(50, 50);
129 tft.print(i / 10.0, 1);
130
131 // Adding a parameter "true" to the setTextColor() function fills character background
132 // This extra parameter is only for smooth fonts!
133 tft.setTextColor(TFT_GREEN, TFT_BLACK, true);
134 tft.setCursor(50, 90);
135 tft.print(i / 10.0, 1);
136
137 delay (200);
138 }
139
140 delay(5000);
141
142 // >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
143 // Large font text wrapping
144 // >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
145
146 tft.fillScreen(TFT_BLACK);
147
148 tft.setTextColor(TFT_YELLOW, TFT_BLACK); // Change the font colour and the background colour
149
150 tft.setCursor(0, 0); // Set cursor at top left of screen
151
152 tft.println("Large font!");
153
154 tft.setTextWrap(true); // Wrap on width
155 tft.setTextColor(TFT_CYAN, TFT_BLACK);
156 tft.println("Long lines wrap to the next line");
157
158 tft.setTextWrap(false, false); // Wrap on width and height switched off
159 tft.setTextColor(TFT_MAGENTA, TFT_BLACK);
160 tft.println("Unless text wrap is switched off");
161
162 tft.unloadFont(); // Remove the font to recover memory used
163
164 delay(8000);
165}
