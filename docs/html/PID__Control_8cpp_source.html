40/**
41 * Implementa:
42 * out = Kp*u + I + Kd*du/dt
43 *
44 * donde:
45 * I[k] = saturación( I[k-1] + Ki * u[k] * dt, ±I_sat )
46 */
47float PID_Update(const PIDParams &p,
48 PIDState &s,
49 float u,
50 float dt)
51{
52 // Protección básica
53 if (dt <= 0.0f) dt = 1e-3f;
54
55 // ---- Derivada du/dt ----
56 float du_dt = 0.0f;
57 if (!s.firstRun) {
58 du_dt = (u - s.prevInput) / dt;
59 } else {
60 du_dt = 0.0f;
61 s.firstRun = false;
62 }
63 s.prevInput = u;
64
65 // ---- Integrador con saturación ----
66 // En Simulink: Ki -> integrador -> saturador Ihhsat
67 s.integrator += p.Ki * u * dt;
68
69 // Saturación simétrica ±I_sat
70 if (p.I_sat > 0.0f) {
71 if (s.integrator > p.I_sat) s.integrator = p.I_sat;
72 if (s.integrator < -p.I_sat) s.integrator = -p.I_sat;
73 }
74
75 // ---- Salida total: Kp*u + I + Kd*du/dt ----
76 float outP = p.Kp * u;
77 float outI = s.integrator;
78 float outD = p.Kd * du_dt;
79
80 float out = outP + outI + outD;
81
82 return out;
83}

123static int U_to_Register(float U, float Umax)
124{
125 if (Umax <= 0.0f) return 0;

143void PID4_Update(const PID4Params &p,
144 PID4State &s,
145 float in1,
146 float in2,
147 float dt,
148 float &out1,
149 float &out2)
150{
151 // in1 = error horizontal (eh)
152 // in2 = error vertical (ev)
153
154 // PIDhh: in1 -> out1
155 float u_hh = PID_Update(p.hh, s.hh, in1, dt);
156
157 // PIDhv: in1 -> out2
158 float u_hv = PID_Update(p.hv, s.hv, in1, dt);
159
160 // PIDvh: in2 -> out1
161 float u_vh = PID_Update(p.vh, s.vh, in2, dt);
162
163 // PIDvv: in2 -> out2
164 float u_vv = PID_Update(p.vv, s.vv, in2, dt);
165
166 // Sumas MIMO
167 float Uh = u_hh + u_vh; // out_1
168 float Uv = u_hv + u_vv; // out_2
169
170 // Saturaciones Uhmax / Uvmax (simétricas ±)
171 if (p.Uh_max > 0.0f) {
172 if (Uh > p.Uh_max) Uh = p.Uh_max;
173 if (Uh < -p.Uh_max) Uh = -p.Uh_max;
174 }
175 if (p.Uv_max > 0.0f) {
176 if (Uv > p.Uv_max) Uv = p.Uv_max;
177 if (Uv < -p.Uv_max) Uv = -p.Uv_max;
178 }
179
180 out1 = Uh;
181 out2 = Uv;
182}

184void PID4_Step(float dt)
185{
186 if (!s_pid4_enabled) return;
187 if (dt <= 0.0f) dt = 1e-3f;
188
189 // 1) Leer ángulos medidos del TRMS EN GRADOS
190 float measVertDeg, measHorDeg;
191 TRMS_GetAnglesDeg(measVertDeg, measHorDeg);
192
193 // 2) Convertir referencias y medidas a RADIANES
194 float refV_rad = s_refVertDeg * DEG_TO_RAD;
195 float refH_rad = s_refHorDeg * DEG_TO_RAD;
196 float measV_rad = measVertDeg * DEG_TO_RAD;
197 float measH_rad = measHorDeg * DEG_TO_RAD;
198
199 // 3) Errores en radianes (como en el modelo original)
200 float ev = refV_rad - measV_rad;
201 float eh = refH_rad - measH_rad;
202
203 // 4) Ejecutar PID-4
204 float Uh, Uv;
205 PID4_Update(s_pid4_params, s_pid4_state, eh, ev, dt, Uh, Uv);
206
207 // 5) Convertir Uh/Uv a registros de motor [-100..100]
208 Registro_RDC = U_to_Register(Uh, s_pid4_params.Uh_max); // rotor de cola (horizontal)
209 Registro_MP = U_to_Register(Uv, s_pid4_params.Uv_max); // motor principal (vertical)
210
211 // 6) Aplicar a los DACs / sliders
212 MotorControl_update(Registro_MP, Registro_RDC);
213}
