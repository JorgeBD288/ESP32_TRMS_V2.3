213/**
214 * Switch screen with animation
215 * @param scr pointer to the new screen to load
216 * @param anim_type type of the animation from `lv_scr_load_anim_t`, e.g. `LV_SCR_LOAD_ANIM_MOVE_LEFT`
217 * @param time time of the animation
218 * @param delay delay before the transition
219 * @param auto_del true: automatically delete the old screen
220 */
221void lv_scr_load_anim(lv_obj_t * new_scr, lv_scr_load_anim_t anim_type, uint32_t time, uint32_t delay, bool auto_del)
222{
223
224 lv_disp_t * d = lv_obj_get_disp(new_scr);
225 lv_obj_t * act_scr = lv_scr_act();
226
227 if(act_scr == new_scr || d->scr_to_load == new_scr) {
228 return;
229 }
230
231 /*If an other screen load animation is in progress
232 *make target screen loaded immediately. */
233 if(d->scr_to_load) {
234 scr_load_internal(d->scr_to_load);
235 lv_anim_del(d->scr_to_load, NULL);
236 lv_obj_set_pos(d->scr_to_load, 0, 0);
237 lv_obj_remove_local_style_prop(d->scr_to_load, LV_STYLE_OPA, 0);
238
239 if(d->del_prev) {
240 lv_obj_del(act_scr);
241 }
242 act_scr = d->scr_to_load;
243 }
244
245 d->scr_to_load = new_scr;
246
247 if(d->prev_scr && d->del_prev) {
248 lv_obj_del(d->prev_scr);
249 d->prev_scr = NULL;
250 }
251
252 d->draw_prev_over_act = is_out_anim(anim_type);
253 d->del_prev = auto_del;
254
255 /*Be sure there is no other animation on the screens*/
256 lv_anim_del(new_scr, NULL);
257 lv_anim_del(lv_scr_act(), NULL);
258
259 /*Be sure both screens are in a normal position*/
260 lv_obj_set_pos(new_scr, 0, 0);
261 lv_obj_set_pos(lv_scr_act(), 0, 0);
262 lv_obj_remove_local_style_prop(new_scr, LV_STYLE_OPA, 0);
263 lv_obj_remove_local_style_prop(lv_scr_act(), LV_STYLE_OPA, 0);
264
265 /*Shortcut for immediate load*/
266 if(time == 0 && delay == 0) {
267 scr_load_internal(new_scr);
268 if(auto_del) lv_obj_del(act_scr);
269 return;
270 }
271
272 lv_anim_t a_new;
273 lv_anim_init(&a_new);
274 lv_anim_set_var(&a_new, new_scr);
275 lv_anim_set_start_cb(&a_new, scr_load_anim_start);
276 lv_anim_set_ready_cb(&a_new, scr_anim_ready);
277 lv_anim_set_time(&a_new, time);
278 lv_anim_set_delay(&a_new, delay);
279
280 lv_anim_t a_old;
281 lv_anim_init(&a_old);
282 lv_anim_set_var(&a_old, d->act_scr);
283 lv_anim_set_time(&a_old, time);
284 lv_anim_set_delay(&a_old, delay);
285
286 switch(anim_type) {
287 case LV_SCR_LOAD_ANIM_NONE:
288 /*Create a dummy animation to apply the delay*/
289 lv_anim_set_exec_cb(&a_new, set_x_anim);
290 lv_anim_set_values(&a_new, 0, 0);
291 break;
292 case LV_SCR_LOAD_ANIM_OVER_LEFT:
293 lv_anim_set_exec_cb(&a_new, set_x_anim);
294 lv_anim_set_values(&a_new, lv_disp_get_hor_res(d), 0);
295 break;
296 case LV_SCR_LOAD_ANIM_OVER_RIGHT:
297 lv_anim_set_exec_cb(&a_new, set_x_anim);
298 lv_anim_set_values(&a_new, -lv_disp_get_hor_res(d), 0);
299 break;
300 case LV_SCR_LOAD_ANIM_OVER_TOP:
301 lv_anim_set_exec_cb(&a_new, set_y_anim);
302 lv_anim_set_values(&a_new, lv_disp_get_ver_res(d), 0);
303 break;
304 case LV_SCR_LOAD_ANIM_OVER_BOTTOM:
305 lv_anim_set_exec_cb(&a_new, set_y_anim);
306 lv_anim_set_values(&a_new, -lv_disp_get_ver_res(d), 0);
307 break;
308 case LV_SCR_LOAD_ANIM_MOVE_LEFT:
309 lv_anim_set_exec_cb(&a_new, set_x_anim);
310 lv_anim_set_values(&a_new, lv_disp_get_hor_res(d), 0);
311
312 lv_anim_set_exec_cb(&a_old, set_x_anim);
313 lv_anim_set_values(&a_old, 0, -lv_disp_get_hor_res(d));
314 break;
315 case LV_SCR_LOAD_ANIM_MOVE_RIGHT:
316 lv_anim_set_exec_cb(&a_new, set_x_anim);
317 lv_anim_set_values(&a_new, -lv_disp_get_hor_res(d), 0);
318
319 lv_anim_set_exec_cb(&a_old, set_x_anim);
320 lv_anim_set_values(&a_old, 0, lv_disp_get_hor_res(d));
321 break;
322 case LV_SCR_LOAD_ANIM_MOVE_TOP:
323 lv_anim_set_exec_cb(&a_new, set_y_anim);
324 lv_anim_set_values(&a_new, lv_disp_get_ver_res(d), 0);
325
326 lv_anim_set_exec_cb(&a_old, set_y_anim);
327 lv_anim_set_values(&a_old, 0, -lv_disp_get_ver_res(d));
328 break;
329 case LV_SCR_LOAD_ANIM_MOVE_BOTTOM:
330 lv_anim_set_exec_cb(&a_new, set_y_anim);
331 lv_anim_set_values(&a_new, -lv_disp_get_ver_res(d), 0);
332
333 lv_anim_set_exec_cb(&a_old, set_y_anim);
334 lv_anim_set_values(&a_old, 0, lv_disp_get_ver_res(d));
335 break;
336 case LV_SCR_LOAD_ANIM_FADE_IN:
337 lv_anim_set_exec_cb(&a_new, opa_scale_anim);
338 lv_anim_set_values(&a_new, LV_OPA_TRANSP, LV_OPA_COVER);
339 break;
340 case LV_SCR_LOAD_ANIM_FADE_OUT:
341 lv_anim_set_exec_cb(&a_old, opa_scale_anim);
342 lv_anim_set_values(&a_old, LV_OPA_COVER, LV_OPA_TRANSP);
343 break;
344 case LV_SCR_LOAD_ANIM_OUT_LEFT:
345 lv_anim_set_exec_cb(&a_old, set_x_anim);
346 lv_anim_set_values(&a_old, 0, -lv_disp_get_hor_res(d));
347 break;
348 case LV_SCR_LOAD_ANIM_OUT_RIGHT:
349 lv_anim_set_exec_cb(&a_old, set_x_anim);
350 lv_anim_set_values(&a_old, 0, lv_disp_get_hor_res(d));
351 break;
352 case LV_SCR_LOAD_ANIM_OUT_TOP:
353 lv_anim_set_exec_cb(&a_old, set_y_anim);
354 lv_anim_set_values(&a_old, 0, -lv_disp_get_ver_res(d));
355 break;
356 case LV_SCR_LOAD_ANIM_OUT_BOTTOM:
357 lv_anim_set_exec_cb(&a_old, set_y_anim);
358 lv_anim_set_values(&a_old, 0, lv_disp_get_ver_res(d));
359 break;
360 }
361
362 lv_event_send(act_scr, LV_EVENT_SCREEN_UNLOAD_START, NULL);
363
364 lv_anim_start(&a_new);
365 lv_anim_start(&a_old);
366}

368/**
369 * Get elapsed time since last user activity on a display (e.g. click)
370 * @param disp pointer to a display (NULL to get the overall smallest inactivity)
371 * @return elapsed ticks (milliseconds) since the last activity
372 */
373uint32_t lv_disp_get_inactive_time(const lv_disp_t * disp)
374{
375 if(disp) return lv_tick_elaps(disp->last_activity_time);
376
377 lv_disp_t * d;
378 uint32_t t = UINT32_MAX;
379 d = lv_disp_get_next(NULL);
380 while(d) {
381 uint32_t elaps = lv_tick_elaps(d->last_activity_time);
382 t = LV_MIN(t, elaps);
383 d = lv_disp_get_next(d);
384 }
385
386 return t;
387}

473static void scr_load_internal(lv_obj_t * scr)
474{
475 lv_disp_t * d = lv_obj_get_disp(scr);
476 if(!d) return; /*Shouldn't happen, just to be sure*/
477
478 lv_obj_t * old_scr = d->act_scr;
479
480 if(d->act_scr) lv_event_send(old_scr, LV_EVENT_SCREEN_UNLOAD_START, NULL);
481 if(d->act_scr) lv_event_send(scr, LV_EVENT_SCREEN_LOAD_START, NULL);
482
483 d->act_scr = scr;
484 d->scr_to_load = NULL;
485
486 if(d->act_scr) lv_event_send(scr, LV_EVENT_SCREEN_LOADED, NULL);
487 if(d->act_scr) lv_event_send(old_scr, LV_EVENT_SCREEN_UNLOADED, NULL);
488
489 lv_obj_invalidate(scr);
490}

517static void scr_anim_ready(lv_anim_t * a)
518{
519 lv_disp_t * d = lv_obj_get_disp(a->var);
520
521 lv_event_send(d->act_scr, LV_EVENT_SCREEN_LOADED, NULL);
522 lv_event_send(d->prev_scr, LV_EVENT_SCREEN_UNLOADED, NULL);
523
524 if(d->prev_scr && d->del_prev) lv_obj_del(d->prev_scr);
525 d->prev_scr = NULL;

527 d->scr_to_load = NULL;
528 lv_obj_remove_local_style_prop(a->var, LV_STYLE_OPA, 0);
529 lv_obj_invalidate(d->act_scr);
