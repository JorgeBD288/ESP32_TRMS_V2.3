78void setup() {
79 Serial.begin(115200);
80 while (!Serial)
81 ; // Wait for Serial to become available. Is optimized away for some cores.
82
83 // Just to know which program is running on my Arduino
84 Serial.println(F("START " __FILE__ " from " __DATE__ "\r\nUsing library version " VERSION_IRREMOTE));
85
86 // Start the receiver and if not 3. parameter specified, take LED_BUILTIN pin from the internal boards definition as default feedback LED
87 IrReceiver.begin(IR_RECEIVE_PIN, ENABLE_LED_FEEDBACK);
88 IrReceiver.registerReceiveCompleteCallback(ReceiveCompleteCallbackHandler);
89
90 Serial.print(F("Ready to receive IR signals of protocols: "));
91 printActiveIRProtocols(&Serial);
92 Serial.println(F("at pin " STR(IR_RECEIVE_PIN)));
93}

95void loop() {
96 /*
97 * Print in loop (interrupts are enabled here) if received data is available.
98 */
99 if (sIRDataJustReceived) {
100 // Print a summary of received data
101 if (IrReceiver.decodedIRData.protocol == UNKNOWN) {
102 Serial.println(F("Received noise or an unknown (or not yet enabled) protocol"));
103 /*
104 * We have an unknown protocol here, print more info.
105 * !!!Attention!!! This prints incorrect values, if we are late (not in this simple example :-))
106 * and the the first mark of the next (repeat) data was yet received
107 */
108 IrReceiver.printIRResultRawFormatted(&Serial, true); //
109 } else {
110 IrReceiver.printIRResultShort(&Serial);
111 IrReceiver.printIRSendUsage(&Serial);
112 }
113 Serial.println();
114 }
115}

117/*
118 * Callback function
119 * Here we know, that data is available.
120 * This function is executed in ISR (Interrupt Service Routine) context (interrupts are blocked here).
121 * Make it short and fast and keep in mind, that you can not use delay(), prints longer than print buffer size etc.,
122 * because they require interrupts enabled to return.
123 * In order to enable other interrupts you can call sei() (enable interrupt again) after evaluating/copying data.
124 * Good practice, but somewhat more complex, is to copy relevant data and signal receiving to main loop.
125 */
126#if defined(ESP32) || defined(ESP8266)
127IRAM_ATTR
128# endif
129void ReceiveCompleteCallbackHandler() {
130 IrReceiver.decode(); // fill IrReceiver.decodedIRData
131 /*
132 * Enable receiving of the next value.
133 */
134 IrReceiver.resume();
135
136 /*
137 * Check the received data and perform actions according to the received command
138 * Decoded result is in the IrReceiver.decodedIRData structure.
139 *
140 * E.g. command is in IrReceiver.decodedIRData.command,
141 * address is in command is in IrReceiver.decodedIRData.address
142 * and up to 32 bit raw data in IrReceiver.decodedIRData.decodedRawData
143 */
144 if (IrReceiver.decodedIRData.command == 0x10) {
145 // do something SHORT here
146 } else if (IrReceiver.decodedIRData.command == 0x11) {
147 // do something SHORT here too
148 }
149
150 /*
151 * Set flag to trigger printing of results in main loop,
152 * since printing should not be done in a callback function
153 * running in ISR (Interrupt Service Routine) context where interrupts are disabled.
154 */
155 sIRDataJustReceived = true;
156
157}
