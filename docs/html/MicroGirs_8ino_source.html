1/**
2 * @file MicroGirs.ino
3 *
4 * @brief This is a minimalistic <a href="http://harctoolbox.org/Girs.html">Girs server</a>.
5 * It only depends on (a subset of) IRremote. It can be used with
6 * <a href="http://www.harctoolbox.org/IrScrutinizer.html">IrScrutinizer</a>
7 * (select Sending/Capturing hw = Girs Client) as well as
8 * <a href="http://lirc.org">Lirc</a>
9 * version 0.9.4 and later, using the driver <a href="http://lirc.org/html/girs.html">Girs</a>).
10 * (Authors of similar software are encourage to implement support.)
11 *
12 * It runs on all hardware on which IRemote runs.
13 *
14 * It consists of an interactive IR server, taking one-line commands from
15 * the "user" (which is typically not a person but another program), and
16 * responds with a one-line response. In the language of the Girs
17 * specifications, the modules "base", receive, and transmit are
18 * implemented. (The two latter can be disabled by not defining two
19 * corresponding CPP symbols.)
20 *
21 * It understands the following commands:
22 *
23 * The "version" command returns the program name and version,
24 * The "modules" command returns the modules implemented, normally base, receive and transmit.
25 * The "receive" command reads an IR signal using the used, demodulating IR sensor.
26 * The "send" commands transmits a supplied raw signal the requested number of times.
27 *
28 * Only the first character of the command is evaluated in this implementation.
29 *
30 * The "receive" command returns the received IR sequence as a sequence
31 * of durations, including a (dummy) trailing silence. On-periods
32 * ("marks", "flashes") are prefixed by "+", while off-periods ("spaces",
33 * "gaps") are prefixed by "-". The present version never times out.
34 *
35 * The \c send command takes the following parameters:
36 *
37 * send noSends frequencyHz introLength repeatLength endingLength durations...
38
39 * where
40 *
41 * * frequencyHz denotes the modulation frequency in Hz
42 * (\e not khz, as is normally used in IRremote)
43 * * introLength denotes the length of the intro sequence, must be even,
44 * * repeatLength denotes the length of the repeat sequence, must be even,
45 * * endingLength denotes the length of the ending sequence (normally 0), must be even.
46 * * duration... denotes the microsecond durations to send,
47 * starting with the first on-period, ending with a (possibly dummy) trailing silence
48 *
49 * Semantics: first the intro sequence will be sent once (i.e., the first
50 * repeatLength durations) (if non-empty). Then the repeat sequence will
51 * be sent (noSends-1) times, unless the intro sequence was empty, in
52 * which case it will be send noSends times. Finally, the ending
53 * sequence will be send once (if non-empty).
54 *
55 * Weaknesses of the IRremote implementation:
56 * * Reception never times out if no IR is seen.
57 * * The IRrecv class does its own decoding which is irrelevant for us.
58 * * The timeout at the end on a signal reception is not configurable.
59 * For example, a NEC1 type signal will cut after the intro sequence,
60 * and the repeats will be considered independent signals.
61 * In IrSqrutinizer, recognition of repeating signals will therefore not work.
62 * The size of the data is platform dependent ("unsigned int", which is 16 bit on AVR boards, 32 bits on 32 bit boards).
63 *
64 */

156/**
157 * Our own tokenizer class. Breaks the command line into tokens.
158 * Usage outside of this package is discouraged.
159 */
160class Tokenizer {
161private:
162 static const int invalidIndex = -1;
163
164 int index; // signed since invalidIndex is possible
165 const String &payload;
166 void trim();
167
168public:
169 Tokenizer(const String &str);
170
171 String getToken();
172 String getRest();
173 String getLine();
174 long getInt();
175 microseconds_t getMicroseconds();
176 frequency_t getFrequency();
177
178 static const int invalid = INT_MAX;
179};
180
181Tokenizer::Tokenizer(const String &str) :
182 index(0), payload(str) {
183}
184
185String Tokenizer::getRest() {
186 String result = index == invalidIndex ? String("") : payload.substring(index);
187 index = invalidIndex;
188 return result;
189}
190
191String Tokenizer::getLine() {
192 if (index == invalidIndex) return String("");
193
194 int i = payload.indexOf('\n', index);
195 String s = (i > 0) ? payload.substring(index, i) : payload.substring(index);
196 index = (i > 0) ? i + 1 : invalidIndex;
197 return s;
198}
199
200String Tokenizer::getToken() {
201 if (index < 0) return String("");
202
203 int i = payload.indexOf(' ', index);
204 String s = (i > 0) ? payload.substring(index, i) : payload.substring(index);
205 index = (i > 0) ? i : invalidIndex;
206 if (index != invalidIndex) if (index != invalidIndex) while (payload.charAt(index) == ' ')
207 index++;
208 return s;
209}
210
211long Tokenizer::getInt() {
212 String token = getToken();
213 return token == "" ? (long) invalid : token.toInt();
214}
215
216microseconds_t Tokenizer::getMicroseconds() {
217 long t = getToken().toInt();
218 return (microseconds_t) ((t < MICROSECONDS_T_MAX) ? t : MICROSECONDS_T_MAX);
219}
220
221frequency_t Tokenizer::getFrequency() {
222 long t = getToken().toInt();
223 return (frequency_t) ((t < FREQUENCY_T_MAX) ? t : FREQUENCY_T_MAX);
224}

229 return (hz + 500) / 1000;

232/**
233 * Transmits the IR signal given as argument.
234 *
235 * The intro sequence (possibly empty) is first sent. Then the repeat signal
236 * (also possibly empty) is sent, "times" times, except for the case of
237 * the intro signal being empty, in which case it is sent "times" times.
238 * Finally the ending sequence (possibly empty) is sent.
239 *
240 * @param intro Sequence to be sent exactly once at the start.
241 * @param lengthIntro Number of durations in intro sequence, must be even.
242 * @param repeat Sequence top possibly be sent multiple times
243 * @param lengthRepeat Number of durations in repeat sequence.
244 * @param ending Sequence to be sent at the end, possibly empty
245 * @param lengthEnding Number of durations in ending sequence
246 * @param frequency Modulation frequency, in Hz (not kHz as normally in IRremote)
247 * @param times Number of times to send the signal, in the sense above.
248 */
249static void sendRaw(const microseconds_t intro[], unsigned lengthIntro, const microseconds_t repeat[], unsigned lengthRepeat,
250 const microseconds_t ending[], unsigned lengthEnding, frequency_t frequency, unsigned times) {
251 if (lengthIntro > 0U) IrSender.sendRaw(intro, lengthIntro, hz2khz(frequency));
252 if (lengthRepeat > 0U) for (unsigned i = 0U; i < times - (lengthIntro > 0U); i++)
253 IrSender.sendRaw(repeat, lengthRepeat, hz2khz(frequency));
254 if (lengthEnding > 0U) IrSender.sendRaw(ending, lengthEnding, hz2khz(frequency));
255}
256#endif // TRANSMIT

260static void dump(Stream &stream) {
261 unsigned int count = IrReceiver.decodedIRData.rawDataPtr->rawlen;
262 // If buffer gets full, count = RAW_BUFFER_LENGTH, which is odd,
263 // and IrScrutinizer does not like that.
264 count &= ~1;
265 for (unsigned int i = 1; i < count; i++) {
266 stream.write(i & 1 ? '+' : '-');
267 stream.print(IrReceiver.decodedIRData.rawDataPtr->rawbuf[i] * MICROS_PER_TICK, DEC);
268 stream.print(" ");
269 }
270 stream.print('-');
271 stream.println(DUMMYENDING);
272}

328static void processCommand(const String &line, Stream &stream) {
329 Tokenizer tokenizer(line);
330 String cmd = tokenizer.getToken();
331
332 // Decode the command in cmd
333 if (cmd.length() == 0) {
334 // empty command, do nothing
335 stream.println(F(okString));
336 return;
337 }
338
339 switch (cmd[0]) {
340 case 'm':
341 stream.println(F(modulesSupported));
342 break;
343
344#if defined(RECEIVE)
345 case 'r': // receive
346 //case 'a':
347 //case 'c':
348 receive(stream);
349 break;
350#endif // RECEIVE
351
352#if defined(TRANSMIT)
353 case 's': // send
354 {
355 // TODO: handle unparsable data gracefully
356 unsigned noSends = (unsigned) tokenizer.getInt();
357 frequency_t frequency = tokenizer.getFrequency();
358 unsigned introLength = (unsigned) tokenizer.getInt();
359 unsigned repeatLength = (unsigned) tokenizer.getInt();
360 unsigned endingLength = (unsigned) tokenizer.getInt();
361 microseconds_t intro[introLength];
362 microseconds_t repeat[repeatLength];
363 microseconds_t ending[endingLength];
364 for (unsigned i = 0; i < introLength; i++)
365 intro[i] = tokenizer.getMicroseconds();
366 for (unsigned i = 0; i < repeatLength; i++)
367 repeat[i] = tokenizer.getMicroseconds();
368 for (unsigned i = 0; i < endingLength; i++)
369 ending[i] = tokenizer.getMicroseconds();
370 sendRaw(intro, introLength, repeat, repeatLength, ending, endingLength, frequency, noSends);
371 stream.println(F(okString));
372 }
373 break;
374#endif // TRANSMIT
375
376 case 'v': // version
377 stream.println(F(PROGNAME " " VERSION));
378 break;
379 default:
380 stream.println(F(errorString));
381 }
382}
