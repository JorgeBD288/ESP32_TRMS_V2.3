54static void my_disp_flush(lv_disp_drv_t *disp_drv,
55 const lv_area_t *area,
56 lv_color_t *color_p)
57{
58 if (!s_tft || !color_p) {
59 lv_disp_flush_ready(disp_drv);
60 return;
61 }
62
63 int32_t w = (area->x2 - area->x1 + 1);
64 int32_t h = (area->y2 - area->y1 + 1);
65
66 s_tft->startWrite();
67 s_tft->setAddrWindow(area->x1, area->y1, w, h);
68 s_tft->pushColors((uint16_t *)&color_p->full, w * h, true);
69 s_tft->endWrite();
70
71 lv_disp_flush_ready(disp_drv);
72}

74/**
75 * @brief Lectura del táctil para LVGL.
76 *
77 * - NO dependemos solo de la interrupción (porque ya vimos que es delicado).
78 * - Leemos el nivel del pin T_IRQ:
79 * * HIGH -> no hay toque
80 * * LOW -> hay posible toque
81 * - Cuando detectamos un flanco HIGH -> LOW, medimos el voltaje y lo mostramos.
82 * - Solo llamamos a getTouch() cuando T_IRQ está LOW.
83 */
84static void my_touchpad_read(lv_indev_drv_t *indev_drv,
85 lv_indev_data_t *data)
86{
87 (void) indev_drv;
88
89 if (!data || !s_tft) return;
90
91 static int last_level = HIGH;
92
93 uint16_t x = 0, y = 0;
94
95 // Leer nivel del pin T_IRQ (G35)
96 int level = digitalRead(s_t_irq_pin); // 1 = reposo, 0 = toque
97
98 // Detectar flanco HIGH -> LOW (inicio de toque)
99 if (last_level == HIGH && level == LOW) {
100 // Leer voltaje en G35 una vez al principio del toque
101 int raw = analogRead(s_t_irq_pin);
102 float voltage = (raw / 4095.0f) * 3.3f;
103
104 Serial.print("NUEVO TOQUE - G35 ADC raw = ");
105 Serial.print(raw);
106 Serial.print(" -> V = ");
107 Serial.print(voltage, 3);
108 Serial.print(" V | irq_count = ");
109 Serial.println(s_irq_count);
110 }
111
112 if (level == LOW) {
113 // Solo intentamos leer el táctil cuando T_IRQ está en LOW
114 bool touched = s_tft->getTouch(&x, &y);
115
116 if (touched) {
117 data->state = LV_INDEV_STATE_PR;
118 data->point.x = x;
119 data->point.y = y;
120
121 // Si quieres ver coordenadas:
122 // Serial.print("Touch: ");
123 // Serial.print(x);
124 // Serial.print(", ");
125 // Serial.println(y);
126 } else {
127 // T_IRQ LOW pero no hay lectura válida: tratamos como RELEASE
128 data->state = LV_INDEV_STATE_REL;
129 }
130 } else {
131 // T_IRQ en HIGH -> no hay toque
132 data->state = LV_INDEV_STATE_REL;
133 }
134
135 last_level = level;
136}

142void DisplayTouch_begin(TFT_eSPI &tft, const DisplayTouchConfig &cfg)
143{
144 s_tft = &tft;
145 s_width = cfg.width;
146 s_height = cfg.height;
147 s_t_irq_pin = cfg.t_irq_pin;
148
149 Serial.println("DisplayTouch_begin(): configurando TFT + táctil (modo híbrido con IRQ)");
150
151 // Pin de IRQ del táctil: entrada con pull-up
152 pinMode(s_t_irq_pin, INPUT_PULLUP);
153
154 // Configuración del ADC para G35 (para leer el voltaje cuando haya toque)
155 analogReadResolution(12); // 0..4095
156 analogSetPinAttenuation(s_t_irq_pin, ADC_11db); // ~0..3.3 V
157
158 // Adjuntamos la interrupción solo para contar eventos (depuración)
159 attachInterrupt(
160 digitalPinToInterrupt(s_t_irq_pin),
161 touch_isr,
162 CHANGE // ya vimos en tu test que así sí dispara
163 );
164
165 // Inicializar pantalla TFT
166 s_tft->init();
167 delay(200);
168 s_tft->setRotation(1);
169 s_tft->fillScreen(TFT_BLACK);
170
171 // Calibración táctil (si se facilita)
172 if (cfg.calibrationData != nullptr) {
173 s_tft->setTouch((uint16_t *)cfg.calibrationData);
174 Serial.println("Calibración táctil aplicada.");
175 } else {
176 Serial.println("Sin datos de calibración táctil.");
177 }
178
179 // Inicializar LVGL
180 lv_init();
181 delay(50);
182
183 // Inicializar buffer de dibujo de LVGL
184 uint32_t buf_size = s_width * 5;
185 if (buf_size > DISP_MAX_WIDTH * DISP_BUF_LINES) {
186 buf_size = DISP_MAX_WIDTH * DISP_BUF_LINES;
187 }
188
189 lv_disp_draw_buf_init(&s_draw_buf, s_buf, nullptr, buf_size);
190
191 // Configurar el driver de display
192 static lv_disp_drv_t disp_drv;
193 lv_disp_drv_init(&disp_drv);
194 disp_drv.hor_res = s_width;
195 disp_drv.ver_res = s_height;
196 disp_drv.flush_cb = my_disp_flush;
197 disp_drv.draw_buf = &s_draw_buf;
198 lv_disp_drv_register(&disp_drv);
199
200 // Configurar driver de entrada (táctil)
201 static lv_indev_drv_t indev_drv;
202 lv_indev_drv_init(&indev_drv);
203 indev_drv.type = LV_INDEV_TYPE_POINTER;
204 indev_drv.read_cb = my_touchpad_read;
205 lv_indev_drv_register(&indev_drv);
206
207 Serial.println("DisplayTouch inicializado (TFT + LVGL + táctil, T_IRQ leído por nivel y IRQ para depuración).");
208}
