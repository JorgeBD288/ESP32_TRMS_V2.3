2// This example if for processors with LittleFS capability (e.g. RP2040,
3// ESP32, ESP8266). It renders a png file that is stored in LittleFS
4// using the PNGdec library (available via library manager).
5
6// Note: The PNGDEC required lots of RAM to work (~40kbytes) so
7// this sketch is will not run on smaller memory processors (e.g.
8// ESP8266, STM32F103 etc.)
9
10// The test image is in the sketch "data" folder (press Ctrl+K to see it).
11// You must upload the image to LittleFS using the Arduino IDE Tools Data
12// Upload menu option (you may need to install extra tools for that).
13
14// Don't forget to use the Arduino IDE Tools menu to allocate a LittleFS
15// memory partition before uploading the sketch and data!
16
17#include <LittleFS.h>
18#define FileSys LittleFS
19
20// Include the PNG decoder library
21#include <PNGdec.h>
22
23PNG png;
24#define MAX_IMAGE_WIDTH 240 // Adjust for your images
25
26int16_t xpos = 0;
27int16_t ypos = 0;

29// Include the TFT library https://github.com/Bodmer/TFT_eSPI
30#include "SPI.h"
31#include <TFT_eSPI.h> // Hardware-specific library
32TFT_eSPI tft = TFT_eSPI(); // Invoke custom library

34//====================================================================================
35// Setup
36//====================================================================================
37void setup()
38{
39 Serial.begin(115200);
40 Serial.println("\n\n Using the PNGdec library");
41
42 // Initialise FS
43 if (!FileSys.begin()) {
44 Serial.println("LittleFS initialisation failed!");
45 while (1) yield(); // Stay here twiddling thumbs waiting
46 }
47
48 // Initialise the TFT
49 tft.begin();
50 tft.fillScreen(TFT_BLACK);
51
52 Serial.println("\r\nInitialisation done.");
53}
54
55//====================================================================================
56// Loop
57//====================================================================================
58void loop()
59{
60 // Scan LittleFS and load any *.png files
61 File root = LittleFS.open("/", "r");
62 while (File file = root.openNextFile()) {
63 String strname = file.name();
64 strname = "/" + strname;
65 Serial.println(file.name());
66 // If it is not a directory and filename ends in .png then load it
67 if (!file.isDirectory() && strname.endsWith(".png")) {
68 // Pass support callback function names to library
69 int16_t rc = png.open(strname.c_str(), pngOpen, pngClose, pngRead, pngSeek, pngDraw);
70 if (rc == PNG_SUCCESS) {
71 tft.startWrite();
72 Serial.printf("image specs: (%d x %d), %d bpp, pixel type: %d\n", png.getWidth(), png.getHeight(), png.getBpp(), png.getPixelType());
73 uint32_t dt = millis();
74 if (png.getWidth() > MAX_IMAGE_WIDTH) {
75 Serial.println("Image too wide for allocated line buffer size!");
76 }
77 else {
78 rc = png.decode(NULL, 0);
79 png.close();
80 }
81 tft.endWrite();
82 // How long did rendering take...
83 Serial.print(millis()-dt); Serial.println("ms");
84 }
85 }
86 delay(3000);
87 tft.fillScreen(random(0x10000));
88 }
89}

92//=========================================v==========================================
93// pngDraw
94//====================================================================================
95// This next function will be called during decoding of the png file to
96// render each image line to the TFT. If you use a different TFT library
97// you will need to adapt this function to suit.
98// Callback function to draw pixels to the display
99void pngDraw(PNGDRAW *pDraw) {
100 uint16_t lineBuffer[MAX_IMAGE_WIDTH];
101 png.getLineAsRGB565(pDraw, lineBuffer, PNG_RGB565_BIG_ENDIAN, 0xffffffff);
102 tft.pushImage(xpos, ypos + pDraw->y, pDraw->iWidth, 1, lineBuffer);
103}
