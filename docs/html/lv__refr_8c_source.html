130void lv_obj_redraw(lv_draw_ctx_t * draw_ctx, lv_obj_t * obj)
131{
132 const lv_area_t * clip_area_ori = draw_ctx->clip_area;
133 lv_area_t clip_coords_for_obj;
134
135 /*Truncate the clip area to `obj size + ext size` area*/
136 lv_area_t obj_coords_ext;
137 lv_obj_get_coords(obj, &obj_coords_ext);
138 lv_coord_t ext_draw_size = _lv_obj_get_ext_draw_size(obj);
139 lv_area_increase(&obj_coords_ext, ext_draw_size, ext_draw_size);
140 bool com_clip_res = _lv_area_intersect(&clip_coords_for_obj, clip_area_ori, &obj_coords_ext);
141 /*If the object is visible on the current clip area OR has overflow visible draw it.
142 *With overflow visible drawing should happen to apply the masks which might affect children */
143 bool should_draw = com_clip_res || lv_obj_has_flag(obj, LV_OBJ_FLAG_OVERFLOW_VISIBLE);
144 if(should_draw) {
145 draw_ctx->clip_area = &clip_coords_for_obj;
146
147 lv_event_send(obj, LV_EVENT_DRAW_MAIN_BEGIN, draw_ctx);
148 lv_event_send(obj, LV_EVENT_DRAW_MAIN, draw_ctx);
149 lv_event_send(obj, LV_EVENT_DRAW_MAIN_END, draw_ctx);
150#if LV_USE_REFR_DEBUG
151 lv_color_t debug_color = lv_color_make(lv_rand(0, 0xFF), lv_rand(0, 0xFF), lv_rand(0, 0xFF));
152 lv_draw_rect_dsc_t draw_dsc;
153 lv_draw_rect_dsc_init(&draw_dsc);
154 draw_dsc.bg_color.full = debug_color.full;
155 draw_dsc.bg_opa = LV_OPA_20;
156 draw_dsc.border_width = 1;
157 draw_dsc.border_opa = LV_OPA_30;
158 draw_dsc.border_color = debug_color;
159 lv_draw_rect(draw_ctx, &draw_dsc, &obj_coords_ext);
160#endif
161 }
162
163 /*With overflow visible keep the previous clip area to let the children visible out of this object too
164 *With not overflow visible limit the clip are to the object's coordinates to clip the children*/
165 lv_area_t clip_coords_for_children;
166 bool refr_children = true;
167 if(lv_obj_has_flag(obj, LV_OBJ_FLAG_OVERFLOW_VISIBLE)) {
168 clip_coords_for_children = *clip_area_ori;
169 }
170 else {
171 if(!_lv_area_intersect(&clip_coords_for_children, clip_area_ori, &obj->coords)) {
172 refr_children = false;
173 }
174 }
175
176 if(refr_children) {
177 draw_ctx->clip_area = &clip_coords_for_children;
178 uint32_t i;
179 uint32_t child_cnt = lv_obj_get_child_cnt(obj);
180 for(i = 0; i < child_cnt; i++) {
181 lv_obj_t * child = obj->spec_attr->children[i];
182 refr_obj(draw_ctx, child);
183 }
184 }
185
186 /*If the object was visible on the clip area call the post draw events too*/
187 if(should_draw) {
188 draw_ctx->clip_area = &clip_coords_for_obj;
189
190 /*If all the children are redrawn make 'post draw' draw*/
191 lv_event_send(obj, LV_EVENT_DRAW_POST_BEGIN, draw_ctx);
192 lv_event_send(obj, LV_EVENT_DRAW_POST, draw_ctx);
193 lv_event_send(obj, LV_EVENT_DRAW_POST_END, draw_ctx);
194 }
195
196 draw_ctx->clip_area = clip_area_ori;
197}

200/**
201 * Invalidate an area on display to redraw it
202 * @param area_p pointer to area which should be invalidated (NULL: delete the invalidated areas)
203 * @param disp pointer to display where the area should be invalidated (NULL can be used if there is
204 * only one display)
205 */
206void _lv_inv_area(lv_disp_t * disp, const lv_area_t * area_p)
207{
208 if(!disp) disp = lv_disp_get_default();
209 if(!disp) return;
210 if(!lv_disp_is_invalidation_enabled(disp)) return;
211
212 if(disp->rendering_in_progress) {
213 LV_LOG_ERROR("detected modifying dirty areas in render");
214 return;
215 }
216
217 /*Clear the invalidate buffer if the parameter is NULL*/
218 if(area_p == NULL) {
219 disp->inv_p = 0;
220 return;
221 }
222
223 lv_area_t scr_area;
224 scr_area.x1 = 0;
225 scr_area.y1 = 0;
226 scr_area.x2 = lv_disp_get_hor_res(disp) - 1;
227 scr_area.y2 = lv_disp_get_ver_res(disp) - 1;
228
229 lv_area_t com_area;
230 bool suc;
231
232 suc = _lv_area_intersect(&com_area, area_p, &scr_area);
233 if(suc == false) return; /*Out of the screen*/
234
235 /*If there were at least 1 invalid area in full refresh mode, redraw the whole screen*/
236 if(disp->driver->full_refresh) {
237 disp->inv_areas[0] = scr_area;
238 disp->inv_p = 1;
239 if(disp->refr_timer) lv_timer_resume(disp->refr_timer);
240 return;
241 }
242
243 if(disp->driver->rounder_cb) disp->driver->rounder_cb(disp->driver, &com_area);
244
245 /*Save only if this area is not in one of the saved areas*/
246 uint16_t i;
247 for(i = 0; i < disp->inv_p; i++) {
248 if(_lv_area_is_in(&com_area, &disp->inv_areas[i], 0) != false) return;
249 }
250
251 /*Save the area*/
252 if(disp->inv_p < LV_INV_BUF_SIZE) {
253 lv_area_copy(&disp->inv_areas[disp->inv_p], &com_area);
254 }
255 else { /*If no place for the area add the screen*/
256 disp->inv_p = 0;
257 lv_area_copy(&disp->inv_areas[disp->inv_p], &scr_area);
258 }
259 disp->inv_p++;
260 if(disp->refr_timer) lv_timer_resume(disp->refr_timer);
261}

287void _lv_disp_refr_timer(lv_timer_t * tmr)
288{
289 REFR_TRACE("begin");
290
291 uint32_t start = lv_tick_get();
292 volatile uint32_t elaps = 0;
293
294 if(tmr) {
295 disp_refr = tmr->user_data;
296#if LV_USE_PERF_MONITOR == 0 && LV_USE_MEM_MONITOR == 0
297 /**
298 * Ensure the timer does not run again automatically.
299 * This is done before refreshing in case refreshing invalidates something else.
300 */
301 lv_timer_pause(tmr);
302#endif
303 }
304 else {
305 disp_refr = lv_disp_get_default();
306 }
307
308 /*Refresh the screen's layout if required*/
309 lv_obj_update_layout(disp_refr->act_scr);
310 if(disp_refr->prev_scr) lv_obj_update_layout(disp_refr->prev_scr);
311
312 lv_obj_update_layout(disp_refr->top_layer);
313 lv_obj_update_layout(disp_refr->sys_layer);
314
315 /*Do nothing if there is no active screen*/
316 if(disp_refr->act_scr == NULL) {
317 disp_refr->inv_p = 0;
318 LV_LOG_WARN("there is no active screen");
319 REFR_TRACE("finished");
320 return;
321 }
322
323 lv_refr_join_area();
324 refr_sync_areas();
325 refr_invalid_areas();
326
327 /*If refresh happened ...*/
328 if(disp_refr->inv_p != 0) {
329
330 /*Copy invalid areas for sync next refresh in double buffered direct mode*/
331 if(disp_refr->driver->direct_mode && disp_refr->driver->draw_buf->buf2) {
332
333 uint16_t i;
334 for(i = 0; i < disp_refr->inv_p; i++) {
335 if(disp_refr->inv_area_joined[i])
336 continue;
337
338 lv_area_t * sync_area = _lv_ll_ins_tail(&disp_refr->sync_areas);
339 *sync_area = disp_refr->inv_areas[i];
340 }
341 }
342
343 /*Clean up*/
344 lv_memset_00(disp_refr->inv_areas, sizeof(disp_refr->inv_areas));
345 lv_memset_00(disp_refr->inv_area_joined, sizeof(disp_refr->inv_area_joined));
346 disp_refr->inv_p = 0;
347
348 elaps = lv_tick_elaps(start);
349
350 /*Call monitor cb if present*/
351 if(disp_refr->driver->monitor_cb) {
352 disp_refr->driver->monitor_cb(disp_refr->driver, elaps, px_num);
353 }
354 }
355
356 lv_mem_buf_free_all();
357 _lv_font_clean_up_fmt_txt();
358
359#if LV_DRAW_COMPLEX
360 _lv_draw_mask_cleanup();
361#endif
362
363#if LV_USE_PERF_MONITOR && LV_USE_LABEL
364 lv_obj_t * perf_label = perf_monitor.perf_label;
365 if(perf_label == NULL) {
366 perf_label = lv_label_create(lv_layer_sys());

368 lv_obj_set_style_bg_color(perf_label, lv_color_black(), 0);
369 lv_obj_set_style_text_color(perf_label, lv_color_white(), 0);

371 lv_obj_set_style_pad_bottom(perf_label, 3, 0);
372 lv_obj_set_style_pad_left(perf_label, 3, 0);
373 lv_obj_set_style_pad_right(perf_label, 3, 0);
374 lv_obj_set_style_text_align(perf_label, LV_TEXT_ALIGN_RIGHT, 0);
375 lv_label_set_text(perf_label, "?");
376 lv_obj_align(perf_label, LV_USE_PERF_MONITOR_POS, 0, 0);
377 perf_monitor.perf_label = perf_label;
378 }
379
380 if(lv_tick_elaps(perf_monitor.perf_last_time) < 300) {
381 if(px_num > 5000) {
382 perf_monitor.elaps_sum += elaps;
383 perf_monitor.frame_cnt ++;
384 }
385 }
386 else {
387 perf_monitor.perf_last_time = lv_tick_get();
388 uint32_t fps_limit;
389 uint32_t fps;
390
391 if(disp_refr->refr_timer) {

394 else {
395 fps_limit = 1000 / LV_DISP_DEF_REFR_PERIOD;
396 }

399 perf_monitor.elaps_sum = 1;
400 }

402 fps = fps_limit;
403 }
404 else {
405 fps = (1000 * perf_monitor.frame_cnt) / perf_monitor.elaps_sum;
406 }
407 perf_monitor.elaps_sum = 0;
408 perf_monitor.frame_cnt = 0;
409 if(fps > fps_limit) {
410 fps = fps_limit;
411 }
412
413 perf_monitor.fps_sum_all += fps;
414 perf_monitor.fps_sum_cnt ++;
415 uint32_t cpu = 100 - lv_timer_get_idle();
416 lv_label_set_text_fmt(perf_label, "%"LV_PRIu32" FPS\n%"LV_PRIu32"%% CPU", fps, cpu);
417 }
418#endif
419
420#if LV_USE_MEM_MONITOR && LV_MEM_CUSTOM == 0 && LV_USE_LABEL

435

448 }
449#endif
450
451 REFR_TRACE("finished");
452}

475/**
476 * Join the areas which has got common parts
477 */
478static void lv_refr_join_area(void)
479{
480 uint32_t join_from;
481 uint32_t join_in;
482 lv_area_t joined_area;
483 for(join_in = 0; join_in < disp_refr->inv_p; join_in++) {
484 if(disp_refr->inv_area_joined[join_in] != 0) continue;
485
486 /*Check all areas to join them in 'join_in'*/
487 for(join_from = 0; join_from < disp_refr->inv_p; join_from++) {
488 /*Handle only unjoined areas and ignore itself*/
489 if(disp_refr->inv_area_joined[join_from] != 0 || join_in == join_from) {
490 continue;
491 }
492
493 /*Check if the areas are on each other*/
494 if(_lv_area_is_on(&disp_refr->inv_areas[join_in], &disp_refr->inv_areas[join_from]) == false) {
495 continue;
496 }
497
498 _lv_area_join(&joined_area, &disp_refr->inv_areas[join_in], &disp_refr->inv_areas[join_from]);
499
500 /*Join two area only if the joined area size is smaller*/
501 if(lv_area_get_size(&joined_area) < (lv_area_get_size(&disp_refr->inv_areas[join_in]) +
502 lv_area_get_size(&disp_refr->inv_areas[join_from]))) {
503 lv_area_copy(&disp_refr->inv_areas[join_in], &joined_area);
504
505 /*Mark 'join_form' is joined into 'join_in'*/
506 disp_refr->inv_area_joined[join_from] = 1;
507 }
508 }
509 }

512/**
513 * Refresh the sync areas
514 */
515static void refr_sync_areas(void)
516{
517 /*Do not sync if not direct mode*/
518 if(!disp_refr->driver->direct_mode) return;
519
520 /*Do not sync if not double buffered*/
521 if(disp_refr->driver->draw_buf->buf2 == NULL) return;
522
523 /*Do not sync if no sync areas*/
524 if(_lv_ll_is_empty(&disp_refr->sync_areas)) return;
525
526 /*The buffers are already swapped.
527 *So the active buffer is the off screen buffer where LVGL will render*/
528 void * buf_off_screen = disp_refr->driver->draw_buf->buf_act;
529 void * buf_on_screen = disp_refr->driver->draw_buf->buf_act == disp_refr->driver->draw_buf->buf1
530 ? disp_refr->driver->draw_buf->buf2
531 : disp_refr->driver->draw_buf->buf1;
532
533 /*Get stride for buffer copy*/
534 lv_coord_t stride = lv_disp_get_hor_res(disp_refr);
535
536 /*Iterate through invalidated areas to see if sync area should be copied*/
537 lv_area_t res[4] = {0};
538 int8_t res_c, j;
539 uint32_t i;
540 lv_area_t * sync_area, *new_area, *next_area;
541 for(i = 0; i < disp_refr->inv_p; i++) {
542 /*Skip joined areas*/
543 if(disp_refr->inv_area_joined[i]) continue;
544
545 /*Iterate over sync areas*/
546 sync_area = _lv_ll_get_head(&disp_refr->sync_areas);
547 while(sync_area != NULL) {
548 /*Get next sync area*/
549 next_area = _lv_ll_get_next(&disp_refr->sync_areas, sync_area);
550
551 /*Remove intersect of redraw area from sync area and get remaining areas*/
552 res_c = _lv_area_diff(res, sync_area, &disp_refr->inv_areas[i]);
553
554 /*New sub areas created after removing intersect*/
555 if(res_c != -1) {
556 /*Replace old sync area with new areas*/
557 for(j = 0; j < res_c; j++) {
558 new_area = _lv_ll_ins_prev(&disp_refr->sync_areas, sync_area);
559 *new_area = res[j];
560 }
561 _lv_ll_remove(&disp_refr->sync_areas, sync_area);
562 lv_mem_free(sync_area);
563 }
564
565 /*Move on to next sync area*/
566 sync_area = next_area;
567 }
568 }
569
570 /*Copy sync areas (if any remaining)*/
571 for(sync_area = _lv_ll_get_head(&disp_refr->sync_areas); sync_area != NULL;
572 sync_area = _lv_ll_get_next(&disp_refr->sync_areas, sync_area)) {
573 disp_refr->driver->draw_ctx->buffer_copy(
574 disp_refr->driver->draw_ctx,
575 buf_off_screen, stride, sync_area,
576 buf_on_screen, stride, sync_area
577 );
578 }
579
580 /*Clear sync areas*/
581 _lv_ll_clear(&disp_refr->sync_areas);
582}

584/**
585 * Refresh the joined areas
586 */
587static void refr_invalid_areas(void)
588{
589 px_num = 0;
590
591 if(disp_refr->inv_p == 0) return;
592
593 /*Find the last area which will be drawn*/
594 int32_t i;
595 int32_t last_i = 0;
596 for(i = disp_refr->inv_p - 1; i >= 0; i--) {
597 if(disp_refr->inv_area_joined[i] == 0) {
598 last_i = i;
599 break;
600 }
601 }
602
603 /*Notify the display driven rendering has started*/
604 if(disp_refr->driver->render_start_cb) {
605 disp_refr->driver->render_start_cb(disp_refr->driver);
606 }
607
608 disp_refr->driver->draw_buf->last_area = 0;
609 disp_refr->driver->draw_buf->last_part = 0;
610 disp_refr->rendering_in_progress = true;
611
612 for(i = 0; i < disp_refr->inv_p; i++) {
613 /*Refresh the unjoined areas*/
614 if(disp_refr->inv_area_joined[i] == 0) {
615
616 if(i == last_i) disp_refr->driver->draw_buf->last_area = 1;
617 disp_refr->driver->draw_buf->last_part = 0;
618 refr_area(&disp_refr->inv_areas[i]);
619
620 px_num += lv_area_get_size(&disp_refr->inv_areas[i]);
621 }
622 }

624 disp_refr->rendering_in_progress = false;
625}

627/**
628 * Refresh an area if there is Virtual Display Buffer
629 * @param area_p pointer to an area to refresh
630 */
631static void refr_area(const lv_area_t * area_p)
632{
633 lv_draw_ctx_t * draw_ctx = disp_refr->driver->draw_ctx;
634 draw_ctx->buf = disp_refr->driver->draw_buf->buf_act;
635
636 /*With full refresh just redraw directly into the buffer*/
637 /*In direct mode draw directly on the absolute coordinates of the buffer*/
638 if(disp_refr->driver->full_refresh || disp_refr->driver->direct_mode) {
639 lv_area_t disp_area;
640 lv_area_set(&disp_area, 0, 0, lv_disp_get_hor_res(disp_refr) - 1, lv_disp_get_ver_res(disp_refr) - 1);
641 draw_ctx->buf_area = &disp_area;
642
643 if(disp_refr->driver->full_refresh) {
644 disp_refr->driver->draw_buf->last_part = 1;
645 draw_ctx->clip_area = &disp_area;
646 refr_area_part(draw_ctx);
647 }
648 else {
649 disp_refr->driver->draw_buf->last_part = disp_refr->driver->draw_buf->last_area;
650 draw_ctx->clip_area = area_p;
651 refr_area_part(draw_ctx);
652 }
653 return;
654 }
655
656 /*Normal refresh: draw the area in parts*/
657 /*Calculate the max row num*/
658 lv_coord_t w = lv_area_get_width(area_p);
659 lv_coord_t h = lv_area_get_height(area_p);
660 lv_coord_t y2 = area_p->y2 >= lv_disp_get_ver_res(disp_refr) ?
661 lv_disp_get_ver_res(disp_refr) - 1 : area_p->y2;
662
663 int32_t max_row = get_max_row(disp_refr, w, h);
664
665 lv_coord_t row;
666 lv_coord_t row_last = 0;
667 lv_area_t sub_area;
668 for(row = area_p->y1; row + max_row - 1 <= y2; row += max_row) {
669 /*Calc. the next y coordinates of draw_buf*/
670 sub_area.x1 = area_p->x1;
671 sub_area.x2 = area_p->x2;
672 sub_area.y1 = row;
673 sub_area.y2 = row + max_row - 1;
674 draw_ctx->buf_area = &sub_area;
675 draw_ctx->clip_area = &sub_area;
676 draw_ctx->buf = disp_refr->driver->draw_buf->buf_act;
677 if(sub_area.y2 > y2) sub_area.y2 = y2;
678 row_last = sub_area.y2;
679 if(y2 == row_last) disp_refr->driver->draw_buf->last_part = 1;
680 refr_area_part(draw_ctx);
681 }
682
683 /*If the last y coordinates are not handled yet ...*/
684 if(y2 != row_last) {
685 /*Calc. the next y coordinates of draw_buf*/
686 sub_area.x1 = area_p->x1;
687 sub_area.x2 = area_p->x2;
688 sub_area.y1 = row;
689 sub_area.y2 = y2;
690 draw_ctx->buf_area = &sub_area;
691 draw_ctx->clip_area = &sub_area;
692 draw_ctx->buf = disp_refr->driver->draw_buf->buf_act;
693 disp_refr->driver->draw_buf->last_part = 1;
694 refr_area_part(draw_ctx);
695 }
696}

698static void refr_area_part(lv_draw_ctx_t * draw_ctx)
699{
700 lv_disp_draw_buf_t * draw_buf = lv_disp_get_draw_buf(disp_refr);
701
702 if(draw_ctx->init_buf)
703 draw_ctx->init_buf(draw_ctx);
704
705 /* Below the `area_p` area will be redrawn into the draw buffer.
706 * In single buffered mode wait here until the buffer is freed.
707 * In full double buffered mode wait here while the buffers are swapped and a buffer becomes available*/
708 bool full_sized = draw_buf->size == (uint32_t)disp_refr->driver->hor_res * disp_refr->driver->ver_res;
709 if((draw_buf->buf1 && !draw_buf->buf2) ||
710 (draw_buf->buf1 && draw_buf->buf2 && full_sized)) {
711 while(draw_buf->flushing) {
712 if(disp_refr->driver->wait_cb) disp_refr->driver->wait_cb(disp_refr->driver);
713 }
714
715 /*If the screen is transparent initialize it when the flushing is ready*/
716#if LV_COLOR_SCREEN_TRANSP
717 if(disp_refr->driver->screen_transp) {
718 if(disp_refr->driver->clear_cb) {
719 disp_refr->driver->clear_cb(disp_refr->driver, disp_refr->driver->draw_buf->buf_act, disp_refr->driver->draw_buf->size);
720 }
721 else {
722 lv_memset_00(disp_refr->driver->draw_buf->buf_act, disp_refr->driver->draw_buf->size * LV_IMG_PX_SIZE_ALPHA_BYTE);
723 }
724 }
725#endif
726 }
727
728 lv_obj_t * top_act_scr = NULL;
729 lv_obj_t * top_prev_scr = NULL;
730
731 /*Get the most top object which is not covered by others*/
732 top_act_scr = lv_refr_get_top_obj(draw_ctx->buf_area, lv_disp_get_scr_act(disp_refr));
733 if(disp_refr->prev_scr) {
734 top_prev_scr = lv_refr_get_top_obj(draw_ctx->buf_area, disp_refr->prev_scr);
735 }
736
737 /*Draw a display background if there is no top object*/
738 if(top_act_scr == NULL && top_prev_scr == NULL) {
739 lv_area_t a;
740 lv_area_set(&a, 0, 0,
741 lv_disp_get_hor_res(disp_refr) - 1, lv_disp_get_ver_res(disp_refr) - 1);
742 if(draw_ctx->draw_bg) {
743 lv_draw_rect_dsc_t dsc;
744 lv_draw_rect_dsc_init(&dsc);
745 dsc.bg_img_src = disp_refr->bg_img;
746 dsc.bg_img_opa = disp_refr->bg_opa;
747 dsc.bg_color = disp_refr->bg_color;
748 dsc.bg_opa = disp_refr->bg_opa;
749 draw_ctx->draw_bg(draw_ctx, &dsc, &a);
750 }
751 else if(disp_refr->bg_img) {
752 lv_img_header_t header;
753 lv_res_t res = lv_img_decoder_get_info(disp_refr->bg_img, &header);
754 if(res == LV_RES_OK) {
755 lv_draw_img_dsc_t dsc;
756 lv_draw_img_dsc_init(&dsc);
757 dsc.opa = disp_refr->bg_opa;
758 lv_draw_img(draw_ctx, &dsc, &a, disp_refr->bg_img);
759 }
760 else {
761 LV_LOG_WARN("Can't draw the background image");
762 }
763 }
764 else {
765 lv_draw_rect_dsc_t dsc;
766 lv_draw_rect_dsc_init(&dsc);
767 dsc.bg_color = disp_refr->bg_color;
768 dsc.bg_opa = disp_refr->bg_opa;
769 lv_draw_rect(draw_ctx, &dsc, draw_ctx->buf_area);
770 }
771 }
772
773 if(disp_refr->draw_prev_over_act) {
774 if(top_act_scr == NULL) top_act_scr = disp_refr->act_scr;
775 refr_obj_and_children(draw_ctx, top_act_scr);
776
777 /*Refresh the previous screen if any*/
778 if(disp_refr->prev_scr) {
779 if(top_prev_scr == NULL) top_prev_scr = disp_refr->prev_scr;
780 refr_obj_and_children(draw_ctx, top_prev_scr);
781 }
782 }
783 else {
784 /*Refresh the previous screen if any*/
785 if(disp_refr->prev_scr) {
786 if(top_prev_scr == NULL) top_prev_scr = disp_refr->prev_scr;
787 refr_obj_and_children(draw_ctx, top_prev_scr);
788 }
789
790 if(top_act_scr == NULL) top_act_scr = disp_refr->act_scr;
791 refr_obj_and_children(draw_ctx, top_act_scr);
792 }
793
794 /*Also refresh top and sys layer unconditionally*/
795 refr_obj_and_children(draw_ctx, lv_disp_get_layer_top(disp_refr));
796 refr_obj_and_children(draw_ctx, lv_disp_get_layer_sys(disp_refr));
797
798 draw_buf_flush(disp_refr);
799}

801/**
802 * Search the most top object which fully covers an area
803 * @param area_p pointer to an area
804 * @param obj the first object to start the searching (typically a screen)
805 * @return
806 */
807static lv_obj_t * lv_refr_get_top_obj(const lv_area_t * area_p, lv_obj_t * obj)
808{
809 lv_obj_t * found_p = NULL;
810
811 if(_lv_area_is_in(area_p, &obj->coords, 0) == false) return NULL;
812 if(lv_obj_has_flag(obj, LV_OBJ_FLAG_HIDDEN)) return NULL;
813 if(_lv_obj_get_layer_type(obj) != LV_LAYER_TYPE_NONE) return NULL;
814
815 /*If this object is fully cover the draw area then check the children too*/
816 lv_cover_check_info_t info;
817 info.res = LV_COVER_RES_COVER;
818 info.area = area_p;
819 lv_event_send(obj, LV_EVENT_COVER_CHECK, &info);
820 if(info.res == LV_COVER_RES_MASKED) return NULL;
821
822 int32_t i;
823 int32_t child_cnt = lv_obj_get_child_cnt(obj);
824 for(i = child_cnt - 1; i >= 0; i--) {
825 lv_obj_t * child = obj->spec_attr->children[i];
826 found_p = lv_refr_get_top_obj(area_p, child);
827
828 /*If a children is ok then break*/
829 if(found_p != NULL) {
830 break;
831 }
832 }
833
834 /*If no better children use this object*/
835 if(found_p == NULL && info.res == LV_COVER_RES_COVER) {
836 found_p = obj;
837 }
838
839 return found_p;
840}

842/**
843 * Make the refreshing from an object. Draw all its children and the youngers too.
844 * @param top_p pointer to an objects. Start the drawing from it.
845 * @param mask_p pointer to an area, the objects will be drawn only here
846 */
847static void refr_obj_and_children(lv_draw_ctx_t * draw_ctx, lv_obj_t * top_obj)
848{
849 /*Normally always will be a top_obj (at least the screen)
850 *but in special cases (e.g. if the screen has alpha) it won't.
851 *In this case use the screen directly*/
852 if(top_obj == NULL) top_obj = lv_disp_get_scr_act(disp_refr);
853 if(top_obj == NULL) return; /*Shouldn't happen*/
854
855 /*Refresh the top object and its children*/
856 refr_obj(draw_ctx, top_obj);
857
858 /*Draw the 'younger' sibling objects because they can be on top_obj*/
859 lv_obj_t * parent;
860 lv_obj_t * border_p = top_obj;
861
862 parent = lv_obj_get_parent(top_obj);
863
864 /*Do until not reach the screen*/
865 while(parent != NULL) {
866 bool go = false;
867 uint32_t i;
868 uint32_t child_cnt = lv_obj_get_child_cnt(parent);
869 for(i = 0; i < child_cnt; i++) {
870 lv_obj_t * child = parent->spec_attr->children[i];
871 if(!go) {
872 if(child == border_p) go = true;
873 }
874 else {
875 /*Refresh the objects*/
876 refr_obj(draw_ctx, child);
877 }
878 }
879
880 /*Call the post draw draw function of the parents of the to object*/
881 lv_event_send(parent, LV_EVENT_DRAW_POST_BEGIN, (void *)draw_ctx);
882 lv_event_send(parent, LV_EVENT_DRAW_POST, (void *)draw_ctx);
883 lv_event_send(parent, LV_EVENT_DRAW_POST_END, (void *)draw_ctx);
884
885 /*The new border will be the last parents,
886 *so the 'younger' brothers of parent will be refreshed*/
887 border_p = parent;
888 /*Go a level deeper*/
889 parent = lv_obj_get_parent(parent);
890 }
891}

894static lv_res_t layer_get_area(lv_draw_ctx_t * draw_ctx, lv_obj_t * obj, lv_layer_type_t layer_type,
895 lv_area_t * layer_area_out)
896{
897 lv_coord_t ext_draw_size = _lv_obj_get_ext_draw_size(obj);
898 lv_area_t obj_coords_ext;
899 lv_obj_get_coords(obj, &obj_coords_ext);
900 lv_area_increase(&obj_coords_ext, ext_draw_size, ext_draw_size);
901
902 if(layer_type == LV_LAYER_TYPE_TRANSFORM) {
903 /*Get the transformed area and clip it to the current clip area.
904 *This area needs to be updated on the screen.*/
905 lv_area_t clip_coords_for_obj;
906 lv_area_t tranf_coords = obj_coords_ext;
907 lv_obj_get_transformed_area(obj, &tranf_coords, false, false);
908 if(!_lv_area_intersect(&clip_coords_for_obj, draw_ctx->clip_area, &tranf_coords)) {
909 return LV_RES_INV;
910 }
911
912 /*Transform back (inverse) the transformed area.
913 *It will tell which area of the non-transformed widget needs to be redrawn
914 *in order to cover transformed area after transformation.*/
915 lv_area_t inverse_clip_coords_for_obj = clip_coords_for_obj;
916 lv_obj_get_transformed_area(obj, &inverse_clip_coords_for_obj, false, true);
917 if(!_lv_area_intersect(&inverse_clip_coords_for_obj, &inverse_clip_coords_for_obj, &obj_coords_ext)) {
918 return LV_RES_INV;
919 }
920
921 *layer_area_out = inverse_clip_coords_for_obj;
922 }
923 else if(layer_type == LV_LAYER_TYPE_SIMPLE) {
924 lv_area_t clip_coords_for_obj;
925 if(!_lv_area_intersect(&clip_coords_for_obj, draw_ctx->clip_area, &obj_coords_ext)) {
926 return LV_RES_INV;
927 }
928 *layer_area_out = clip_coords_for_obj;
929 }
930 else {
931 LV_LOG_WARN("Unhandled intermediate layer type");
932 return LV_RES_INV;
933 }
934
935 return LV_RES_OK;
936}

938static void layer_alpha_test(lv_obj_t * obj, lv_draw_ctx_t * draw_ctx, lv_draw_layer_ctx_t * layer_ctx,
939 lv_draw_layer_flags_t flags)
940{
941 bool has_alpha;
942 /*If globally the layer has alpha maybe this smaller section has not (e.g. not on a rounded corner)
943 *If turns out that this section has no alpha renderer can choose faster algorithms*/
944 if(flags & LV_DRAW_LAYER_FLAG_HAS_ALPHA) {
945 /*Test for alpha by assuming there is no alpha. If it fails, fall back to rendering with alpha*/
946 has_alpha = true;
947 if(_lv_area_is_in(&layer_ctx->area_act, &obj->coords, 0)) {
948 lv_cover_check_info_t info;
949 info.res = LV_COVER_RES_COVER;
950 info.area = &layer_ctx->area_act;
951 lv_event_send(obj, LV_EVENT_COVER_CHECK, &info);
952 if(info.res == LV_COVER_RES_COVER) has_alpha = false;
953 }
954
955 if(has_alpha) {
956 layer_ctx->area_act.y2 = layer_ctx->area_act.y1 + layer_ctx->max_row_with_alpha - 1;
957 }
958 }
959 else {
960 has_alpha = false;
961 }
962
963 if(layer_ctx->area_act.y2 > layer_ctx->area_full.y2) layer_ctx->area_act.y2 = layer_ctx->area_full.y2;
964 lv_draw_layer_adjust(draw_ctx, layer_ctx, has_alpha ? LV_DRAW_LAYER_FLAG_HAS_ALPHA : LV_DRAW_LAYER_FLAG_NONE);
965}

968void refr_obj(lv_draw_ctx_t * draw_ctx, lv_obj_t * obj)
969{
970 /*Do not refresh hidden objects*/
971 if(lv_obj_has_flag(obj, LV_OBJ_FLAG_HIDDEN)) return;
972 lv_layer_type_t layer_type = _lv_obj_get_layer_type(obj);
973 if(layer_type == LV_LAYER_TYPE_NONE) {
974 lv_obj_redraw(draw_ctx, obj);
975 }
976 else {
977 lv_opa_t opa = lv_obj_get_style_opa_layered(obj, 0);
978 if(opa < LV_OPA_MIN) return;
979
980 lv_area_t layer_area_full;
981 lv_res_t res = layer_get_area(draw_ctx, obj, layer_type, &layer_area_full);
982 if(res != LV_RES_OK) return;
983
984 lv_draw_layer_flags_t flags = LV_DRAW_LAYER_FLAG_HAS_ALPHA;
985
986 if(_lv_area_is_in(&layer_area_full, &obj->coords, 0)) {
987 lv_cover_check_info_t info;
988 info.res = LV_COVER_RES_COVER;
989 info.area = &layer_area_full;
990 lv_event_send(obj, LV_EVENT_COVER_CHECK, &info);
991 if(info.res == LV_COVER_RES_COVER) flags &= ~LV_DRAW_LAYER_FLAG_HAS_ALPHA;
992 }
993
994 if(layer_type == LV_LAYER_TYPE_SIMPLE) flags |= LV_DRAW_LAYER_FLAG_CAN_SUBDIVIDE;
995
996 lv_draw_layer_ctx_t * layer_ctx = lv_draw_layer_create(draw_ctx, &layer_area_full, flags);
997 if(layer_ctx == NULL) {
998 LV_LOG_WARN("Couldn't create a new layer context");
999 return;
1000 }
1001 lv_point_t pivot = {
1002 .x = lv_obj_get_style_transform_pivot_x(obj, 0),
1003 .y = lv_obj_get_style_transform_pivot_y(obj, 0)
1004 };
1005
1006 if(LV_COORD_IS_PCT(pivot.x)) {
1007 pivot.x = (LV_COORD_GET_PCT(pivot.x) * lv_area_get_width(&obj->coords)) / 100;
1008 }
1009 if(LV_COORD_IS_PCT(pivot.y)) {
1010 pivot.y = (LV_COORD_GET_PCT(pivot.y) * lv_area_get_height(&obj->coords)) / 100;
1011 }
1012
1013 lv_draw_img_dsc_t draw_dsc;
1014 lv_draw_img_dsc_init(&draw_dsc);
1015 draw_dsc.opa = opa;
1016 draw_dsc.angle = lv_obj_get_style_transform_angle(obj, 0);
1017 if(draw_dsc.angle > 3600) draw_dsc.angle -= 3600;
1018 else if(draw_dsc.angle < 0) draw_dsc.angle += 3600;
1019
1020 draw_dsc.zoom = lv_obj_get_style_transform_zoom(obj, 0);
1021 draw_dsc.blend_mode = lv_obj_get_style_blend_mode(obj, 0);
1022 draw_dsc.antialias = disp_refr->driver->antialiasing;
1023
1024 if(flags & LV_DRAW_LAYER_FLAG_CAN_SUBDIVIDE) {
1025 layer_ctx->area_act = layer_ctx->area_full;
1026 layer_ctx->area_act.y2 = layer_ctx->area_act.y1 + layer_ctx->max_row_with_no_alpha - 1;
1027 if(layer_ctx->area_act.y2 > layer_ctx->area_full.y2) layer_ctx->area_act.y2 = layer_ctx->area_full.y2;
1028 }
1029
1030 while(layer_ctx->area_act.y1 <= layer_area_full.y2) {
1031 if(flags & LV_DRAW_LAYER_FLAG_CAN_SUBDIVIDE) {
1032 layer_alpha_test(obj, draw_ctx, layer_ctx, flags);
1033 }
1034
1035 lv_obj_redraw(draw_ctx, obj);
1036
1037 draw_dsc.pivot.x = obj->coords.x1 + pivot.x - draw_ctx->buf_area->x1;
1038 draw_dsc.pivot.y = obj->coords.y1 + pivot.y - draw_ctx->buf_area->y1;
1039
1040 /*With LV_DRAW_LAYER_FLAG_CAN_SUBDIVIDE it should also go the next chunk*/
1041 lv_draw_layer_blend(draw_ctx, layer_ctx, &draw_dsc);
1042
1043 if((flags & LV_DRAW_LAYER_FLAG_CAN_SUBDIVIDE) == 0) break;
1044
1045 layer_ctx->area_act.y1 = layer_ctx->area_act.y2 + 1;
1046 layer_ctx->area_act.y2 = layer_ctx->area_act.y1 + layer_ctx->max_row_with_no_alpha - 1;
1047 }
1048
1049 lv_draw_layer_destroy(draw_ctx, layer_ctx);
1050 }
1051}

1054static uint32_t get_max_row(lv_disp_t * disp, lv_coord_t area_w, lv_coord_t area_h)
1055{
1056 int32_t max_row = (uint32_t)disp->driver->draw_buf->size / area_w;
1057
1058 if(max_row > area_h) max_row = area_h;
1059
1060 /*Round down the lines of draw_buf if rounding is added*/
1061 if(disp_refr->driver->rounder_cb) {
1062 lv_area_t tmp;
1063 tmp.x1 = 0;
1064 tmp.x2 = 0;
1065 tmp.y1 = 0;
1066
1067 lv_coord_t h_tmp = max_row;
1068 do {
1069 tmp.y2 = h_tmp - 1;
1070 disp_refr->driver->rounder_cb(disp_refr->driver, &tmp);
1071
1072 /*If this height fits into `max_row` then fine*/
1073 if(lv_area_get_height(&tmp) <= max_row) break;
1074
1075 /*Decrement the height of the area until it fits into `max_row` after rounding*/
1076 h_tmp--;
1077 } while(h_tmp > 0);
1078
1079 if(h_tmp <= 0) {
1080 LV_LOG_WARN("Can't set draw_buf height using the round function. (Wrong round_cb or to "
1081 "small draw_buf)");
1082 return 0;
1083 }
1084 else {
1085 max_row = tmp.y2 + 1;
1086 }
1087 }
1088
1089 return max_row;
1090}

1092static void draw_buf_rotate_180(lv_disp_drv_t * drv, lv_area_t * area, lv_color_t * color_p)
1093{
1094 lv_coord_t area_w = lv_area_get_width(area);
1095 lv_coord_t area_h = lv_area_get_height(area);
1096 uint32_t total = area_w * area_h;
1097 /*Swap the beginning and end values*/
1098 lv_color_t tmp;
1099 uint32_t i = total - 1, j = 0;
1100 while(i > j) {
1101 tmp = color_p[i];
1102 color_p[i] = color_p[j];
1103 color_p[j] = tmp;
1104 i--;
1105 j++;
1106 }
1107 lv_coord_t tmp_coord;
1108 tmp_coord = area->y2;
1109 area->y2 = drv->ver_res - area->y1 - 1;
1110 area->y1 = drv->ver_res - tmp_coord - 1;
1111 tmp_coord = area->x2;
1112 area->x2 = drv->hor_res - area->x1 - 1;
1113 area->x1 = drv->hor_res - tmp_coord - 1;
1114}

1116static void LV_ATTRIBUTE_FAST_MEM draw_buf_rotate_90(bool invert_i, lv_coord_t area_w, lv_coord_t area_h,
1117 lv_color_t * orig_color_p, lv_color_t * rot_buf)
1118{
1119
1120 uint32_t invert = (area_w * area_h) - 1;
1121 uint32_t initial_i = ((area_w - 1) * area_h);
1122 for(lv_coord_t y = 0; y < area_h; y++) {
1123 uint32_t i = initial_i + y;
1124 if(invert_i)
1125 i = invert - i;
1126 for(lv_coord_t x = 0; x < area_w; x++) {
1127 rot_buf[i] = *(orig_color_p++);
1128 if(invert_i)
1129 i += area_h;
1130 else
1131 i -= area_h;
1132 }
1133 }
1134}

1164 &color_p[j * w + inv_i]

1169 &color_p[i * w + j],

1176 }

1180/**
1181 * Rotate the draw_buf to the display's native orientation.
1182 */
1183static void draw_buf_rotate(lv_area_t * area, lv_color_t * color_p)
1184{
1185 lv_disp_drv_t * drv = disp_refr->driver;
1186 if(disp_refr->driver->full_refresh && drv->sw_rotate) {
1187 LV_LOG_ERROR("cannot rotate a full refreshed display!");
1188 return;
1189 }
1190 if(drv->rotated == LV_DISP_ROT_180) {
1191 draw_buf_rotate_180(drv, area, color_p);
1192 call_flush_cb(drv, area, color_p);
1193 }
1194 else if(drv->rotated == LV_DISP_ROT_90 || drv->rotated == LV_DISP_ROT_270) {
1195 /*Allocate a temporary buffer to store rotated image*/
1196 lv_color_t * rot_buf = NULL;
1197 lv_disp_draw_buf_t * draw_buf = lv_disp_get_draw_buf(disp_refr);
1198 lv_coord_t area_w = lv_area_get_width(area);
1199 lv_coord_t area_h = lv_area_get_height(area);
1200 /*Determine the maximum number of rows that can be rotated at a time*/
1201 lv_coord_t max_row = LV_MIN((lv_coord_t)((LV_DISP_ROT_MAX_BUF / sizeof(lv_color_t)) / area_w), area_h);
1202 lv_coord_t init_y_off;
1203 init_y_off = area->y1;
1204 if(drv->rotated == LV_DISP_ROT_90) {
1205 area->y2 = drv->ver_res - area->x1 - 1;
1206 area->y1 = area->y2 - area_w + 1;
1207 }
1208 else {
1209 area->y1 = area->x1;
1210 area->y2 = area->y1 + area_w - 1;
1211 }
1212
1213 /*Rotate the screen in chunks, flushing after each one*/
1214 lv_coord_t row = 0;
1215 while(row < area_h) {
1216 lv_coord_t height = LV_MIN(max_row, area_h - row);
1217 draw_buf->flushing = 1;
1218 if((row == 0) && (area_h >= area_w)) {
1219 /*Rotate the initial area as a square*/
1220 height = area_w;
1221 draw_buf_rotate_90_sqr(drv->rotated == LV_DISP_ROT_270, area_w, color_p);
1222 if(drv->rotated == LV_DISP_ROT_90) {
1223 area->x1 = init_y_off;
1224 area->x2 = init_y_off + area_w - 1;
1225 }
1226 else {
1227 area->x2 = drv->hor_res - 1 - init_y_off;
1228 area->x1 = area->x2 - area_w + 1;
1229 }
1230 }
1231 else {
1232 /*Rotate other areas using a maximum buffer size*/
1233 if(rot_buf == NULL) rot_buf = lv_mem_buf_get(LV_DISP_ROT_MAX_BUF);
1234 draw_buf_rotate_90(drv->rotated == LV_DISP_ROT_270, area_w, height, color_p, rot_buf);
1235
1236 if(drv->rotated == LV_DISP_ROT_90) {
1237 area->x1 = init_y_off + row;
1238 area->x2 = init_y_off + row + height - 1;
1239 }
1240 else {
1241 area->x2 = drv->hor_res - 1 - init_y_off - row;
1242 area->x1 = area->x2 - height + 1;
1243 }
1244 }
1245
1246 /* The original part (chunk of the current area) were split into more parts here.
1247 * Set the original last_part flag on the last part of rotation. */
1248 if(row + height >= area_h && draw_buf->last_area && draw_buf->last_part) {
1249 draw_buf->flushing_last = 1;
1250 }
1251 else {
1252 draw_buf->flushing_last = 0;
1253 }
1254
1255 /*Flush the completed area to the display*/
1256 call_flush_cb(drv, area, rot_buf == NULL ? color_p : rot_buf);
1257 /*FIXME: Rotation forces legacy behavior where rendering and flushing are done serially*/
1258 while(draw_buf->flushing) {
1259 if(drv->wait_cb) drv->wait_cb(drv);
1260 }
1261 color_p += area_w * height;
1262 row += height;
1263 }
1264 /*Free the allocated buffer at the end if necessary*/
1265 if(rot_buf != NULL) lv_mem_buf_release(rot_buf);
1266 }
1267}

1269/**
1270 * Flush the content of the draw buffer
1271 */
1272static void draw_buf_flush(lv_disp_t * disp)
1273{
1274 lv_disp_draw_buf_t * draw_buf = lv_disp_get_draw_buf(disp_refr);
1275
1276 /*Flush the rendered content to the display*/
1277 lv_draw_ctx_t * draw_ctx = disp->driver->draw_ctx;
1278 if(draw_ctx->wait_for_finish) draw_ctx->wait_for_finish(draw_ctx);
1279
1280 /* In partial double buffered mode wait until the other buffer is freed
1281 * and driver is ready to receive the new buffer */
1282 bool full_sized = draw_buf->size == (uint32_t)disp_refr->driver->hor_res * disp_refr->driver->ver_res;
1283 if(draw_buf->buf1 && draw_buf->buf2 && !full_sized) {
1284 while(draw_buf->flushing) {
1285 if(disp_refr->driver->wait_cb) disp_refr->driver->wait_cb(disp_refr->driver);
1286 }
1287 }
1288
1289 draw_buf->flushing = 1;
1290
1291 if(disp_refr->driver->draw_buf->last_area && disp_refr->driver->draw_buf->last_part) draw_buf->flushing_last = 1;
1292 else draw_buf->flushing_last = 0;
1293
1294 bool flushing_last = draw_buf->flushing_last;
1295
1296 if(disp->driver->flush_cb) {
1297 /*Rotate the buffer to the display's native orientation if necessary*/
1298 if(disp->driver->rotated != LV_DISP_ROT_NONE && disp->driver->sw_rotate) {
1299 draw_buf_rotate(draw_ctx->buf_area, draw_ctx->buf);
1300 }
1301 else {
1302 call_flush_cb(disp->driver, draw_ctx->buf_area, draw_ctx->buf);
1303 }
1304 }
1305
1306 /*If there are 2 buffers swap them. With direct mode swap only on the last area*/
1307 if(draw_buf->buf1 && draw_buf->buf2 && (!disp->driver->direct_mode || flushing_last)) {
1308 if(draw_buf->buf_act == draw_buf->buf1)
1309 draw_buf->buf_act = draw_buf->buf2;
1310 else
1311 draw_buf->buf_act = draw_buf->buf1;
1312 }
1313}
