25void HandleIRNavigation(IRNavKey nav)
26{
27 if (g_navGroup == nullptr) return;
28
29 switch (nav) {
30 case IRNavKey::LEFT:
31 case IRNavKey::UP:
32 lv_group_focus_prev(g_navGroup);
33 break;
34
35 case IRNavKey::RIGHT:
36 case IRNavKey::DOWN:
37 lv_group_focus_next(g_navGroup);
38 break;
39
40 case IRNavKey::ENTER: {
41 lv_obj_t * focused = lv_group_get_focused(g_navGroup);
42 if (focused == nullptr) break;
43
44 // Caso especial: commit de ángulos en Label64/65
45 if (focused == ui_Label64 || focused == ui_Label65) {
46
47 // Si no hay edición activa para esta label, no hacemos nada
48 if (s_numEditTarget == focused && s_numEditBuffer[0] != '\0') {
49
50 int magnitude = atoi(s_numEditBuffer);
51 int value = s_numNegative ? -magnitude : magnitude;
52
53 // Limitar a [-180, 180]
54 if (value > 180) value = 180;
55 if (value < -180) value = -180;
56
57 float newAngle = (float)value;
58
59 // Partimos de los valores actuales
60 float currH = AngSelect_GetRefHorizontal();
61 float currV = AngSelect_GetRefVertical();
62
63 if (focused == ui_Label64) {
64 // Editamos vertical
65 currV = newAngle;
66 } else {
67 // Editamos horizontal
68 currH = newAngle;
69 }
70
71 // Esto actualiza g_refH/g_refV, labels y dibuja el punto con líneas
72 AngSelect_ApplyRefFromAngles(currH, currV);
73
74 // Salimos del modo edición
75 s_numEditBuffer[0] = '\0';
76 s_numNegative = false;
77 s_numEditTarget = nullptr;
78 }
79
80 // Para las labels de ángulo, NO simulamos click extra
81 break;
82 }
83
84 // Resto de objetos: comportamiento normal (click)
85 lv_event_send(focused, LV_EVENT_CLICKED, NULL);
86 lv_event_send(focused, LV_EVENT_PRESSED, NULL);
87 lv_event_send(focused, LV_EVENT_RELEASED, NULL);
88 break;
89 }
90
91 default:
92 break;
93 }

244// Entrada numérica con dígitos (preview)
245// ============================
246void HandleNumericDigit(int digit)
247{
248 if (digit < 0 || digit > 9) return;
249 if (g_navGroup == nullptr) return;
250
251 lv_obj_t * focused = lv_group_get_focused(g_navGroup);
252 if (focused == nullptr) return;
253
254 // Solo en estas labels
255 if (focused != ui_Label64 && focused != ui_Label65) {
256 return;
257 }
258
259 // Si cambiamos de label, empezamos edición nueva
260 if (s_numEditTarget != focused) {
261 s_numEditTarget = focused;
262 s_numEditBuffer[0] = '\0';
263 s_numNegative = false;
264 }
265
266 size_t len = strlen(s_numEditBuffer);
267
268 // Límite de 3 dígitos → si lo superamos, empezamos número nuevo (manteniendo signo)
269 if (len >= 3) {
270 s_numEditBuffer[0] = '\0';
271 len = 0;
272 }
273
274 // Añadir dígito
275 s_numEditBuffer[len] = '0' + digit;
276 s_numEditBuffer[len + 1] = '\0';
277
278 // Convertimos a valor con signo
279 int magnitude = atoi(s_numEditBuffer);
280 int value = s_numNegative ? -magnitude : magnitude;
281
282 // Limitar a [-180, 180] también en preview
283 if (value > 180) value = 180;
284 if (value < -180) value = -180;
285
286 // Ajustar buffer y signo según el valor limitado
287 if (value < 0) {
288 s_numNegative = true;
289 magnitude = -value;
290 } else {
291 s_numNegative = false;
292 magnitude = value;
293 }
294 snprintf(s_numEditBuffer, sizeof(s_numEditBuffer), "%d", magnitude);
295
296 // Actualizar texto de la label (solo preview, sin tocar g_ref)
297 char txt[48];
298 if (focused == ui_Label64) {
299 snprintf(txt, sizeof(txt), "A. vertical = %d º", value);
300 } else {
301 snprintf(txt, sizeof(txt), "A. horizontal = %d º", value);
302 }
303 lv_label_set_text(focused, txt);
304}
