93void lv_btnmatrix_set_map(lv_obj_t * obj, const char * map[])
94{
95 LV_ASSERT_OBJ(obj, MY_CLASS);
96 if(map == NULL) return;
97
98 lv_btnmatrix_t * btnm = (lv_btnmatrix_t *)obj;
99
100 /*Analyze the map and create the required number of buttons*/
101 allocate_btn_areas_and_controls(obj, map);
102 btnm->map_p = map;
103
104 lv_base_dir_t base_dir = lv_obj_get_style_base_dir(obj, LV_PART_MAIN);
105
106 /*Set size and positions of the buttons*/
107 lv_coord_t pleft = lv_obj_get_style_pad_left(obj, LV_PART_MAIN);
108 lv_coord_t ptop = lv_obj_get_style_pad_top(obj, LV_PART_MAIN);
109 lv_coord_t prow = lv_obj_get_style_pad_row(obj, LV_PART_MAIN);
110 lv_coord_t pcol = lv_obj_get_style_pad_column(obj, LV_PART_MAIN);
111
112 lv_coord_t max_w = lv_obj_get_content_width(obj);
113 lv_coord_t max_h = lv_obj_get_content_height(obj);
114
115 /*Calculate the position of each row*/
116 lv_coord_t max_h_no_gap = max_h - (prow * (btnm->row_cnt - 1));
117
118 /*Count the units and the buttons in a line
119 *(A button can be 1,2,3... unit wide)*/
120 uint32_t txt_tot_i = 0; /*Act. index in the str map*/
121 uint32_t btn_tot_i = 0; /*Act. index of button areas*/
122 const char ** map_row = map;
123
124 /*Count the units and the buttons in a line*/
125 uint32_t row;
126 for(row = 0; row < btnm->row_cnt; row++) {
127 uint16_t unit_cnt = 0; /*Number of units in a row*/
128 uint16_t btn_cnt = 0; /*Number of buttons in a row*/
129 /*Count the buttons and units in this row*/
130 while(map_row[btn_cnt] && strcmp(map_row[btn_cnt], "\n") != 0 && map_row[btn_cnt][0] != '\0') {
131 unit_cnt += get_button_width(btnm->ctrl_bits[btn_tot_i + btn_cnt]);
132 btn_cnt++;
133 }
134
135 /*Only deal with the non empty lines*/
136 if(btn_cnt == 0) {
137 map_row = &map_row[btn_cnt + 1]; /*Set the map to the next row*/
138 continue;
139 }
140
141 lv_coord_t row_y1 = ptop + (max_h_no_gap * row) / btnm->row_cnt + row * prow;
142 lv_coord_t row_y2 = ptop + (max_h_no_gap * (row + 1)) / btnm->row_cnt + row * prow - 1;
143
144 /*Set the button size and positions*/
145 lv_coord_t max_w_no_gap = max_w - (pcol * (btn_cnt - 1));
146 if(max_w_no_gap < 0) max_w_no_gap = 0;
147
148 uint32_t row_unit_cnt = 0; /*The current unit position in the row*/
149 uint32_t btn;
150 for(btn = 0; btn < btn_cnt; btn++, btn_tot_i++, txt_tot_i++) {
151 uint32_t btn_u = get_button_width(btnm->ctrl_bits[btn_tot_i]);
152
153 lv_coord_t btn_x1 = (max_w_no_gap * row_unit_cnt) / unit_cnt + btn * pcol;
154 lv_coord_t btn_x2 = (max_w_no_gap * (row_unit_cnt + btn_u)) / unit_cnt + btn * pcol - 1;
155
156 /*If RTL start from the right*/
157 if(base_dir == LV_BASE_DIR_RTL) {
158 lv_coord_t tmp = btn_x1;
159 btn_x1 = btn_x2;
160 btn_x2 = tmp;
161
162 btn_x1 = max_w - btn_x1;
163 btn_x2 = max_w - btn_x2;
164 }
165
166 btn_x1 += pleft;
167 btn_x2 += pleft;
168
169 lv_area_set(&btnm->button_areas[btn_tot_i], btn_x1, row_y1, btn_x2, row_y2);
170
171 row_unit_cnt += btn_u;
172 }
173
174 map_row = &map_row[btn_cnt + 1]; /*Set the map to the next line*/
175 }
176
177 /*Popovers in the top row will draw outside the widget and the extended draw size depends on
178 *the row height which may have changed when setting the new map*/
179 lv_obj_refresh_ext_draw_size(obj);
180
181 lv_obj_invalidate(obj);
182}

207void lv_btnmatrix_set_btn_ctrl(lv_obj_t * obj, uint16_t btn_id, lv_btnmatrix_ctrl_t ctrl)
208{
209 LV_ASSERT_OBJ(obj, MY_CLASS);
210
211 lv_btnmatrix_t * btnm = (lv_btnmatrix_t *)obj;
212
213 if(btn_id >= btnm->btn_cnt) return;
214
215 if(btnm->one_check && (ctrl & LV_BTNMATRIX_CTRL_CHECKED)) {
216 lv_btnmatrix_clear_btn_ctrl_all(obj, LV_BTNMATRIX_CTRL_CHECKED);
217 }
218
219 /* If we hide a button if all buttons are now hidden hide the whole button matrix to make focus behave correctly */
220 if(ctrl & LV_BTNMATRIX_CTRL_HIDDEN) {
221 bool all_buttons_hidden = true;
222 if(btnm->btn_cnt > 1) {
223 for(uint16_t btn_idx = 0; btn_idx < btnm->btn_cnt; btn_idx++) {
224 if(btn_idx == btn_id) continue;
225 if(!(btnm->ctrl_bits[btn_idx] & LV_BTNMATRIX_CTRL_HIDDEN)) all_buttons_hidden = false;
226 }
227
228 }
229 if(all_buttons_hidden) lv_obj_add_flag(obj, LV_OBJ_FLAG_HIDDEN);
230 }
231
232 btnm->ctrl_bits[btn_id] |= ctrl;
233 invalidate_button_area(obj, btn_id);
234
235 if(ctrl & LV_BTNMATRIX_CTRL_POPOVER) {
236 lv_obj_refresh_ext_draw_size(obj);
237 }

404static void lv_btnmatrix_event(const lv_obj_class_t * class_p, lv_event_t * e)
405{
406 LV_UNUSED(class_p);
407
408 lv_res_t res;
409
410 /*Call the ancestor's event handler*/
411 res = lv_obj_event_base(MY_CLASS, e);
412 if(res != LV_RES_OK) return;
413
414 lv_event_code_t code = lv_event_get_code(e);
415 lv_obj_t * obj = lv_event_get_target(e);
416 lv_btnmatrix_t * btnm = (lv_btnmatrix_t *)obj;
417 lv_point_t p;
418
419 if(code == LV_EVENT_REFR_EXT_DRAW_SIZE) {
420 if(has_popovers_in_top_row(obj)) {
421 /*reserve one row worth of extra space to account for popovers in the top row*/
422 lv_coord_t s = btnm->row_cnt > 0 ? lv_obj_get_content_height(obj) / btnm->row_cnt : 0;
423 lv_event_set_ext_draw_size(e, s);
424 }
425 }
426 if(code == LV_EVENT_STYLE_CHANGED) {
427 lv_btnmatrix_set_map(obj, btnm->map_p);
428 }
429 else if(code == LV_EVENT_SIZE_CHANGED) {
430 lv_btnmatrix_set_map(obj, btnm->map_p);
431 }
432 else if(code == LV_EVENT_PRESSED) {
433 void * param = lv_event_get_param(e);
434 invalidate_button_area(obj, btnm->btn_id_sel);
435
436 lv_indev_type_t indev_type = lv_indev_get_type(lv_indev_get_act());
437 if(indev_type == LV_INDEV_TYPE_POINTER || indev_type == LV_INDEV_TYPE_BUTTON) {
438 uint16_t btn_pr;
439 /*Search the pressed area*/
440 lv_indev_get_point(param, &p);
441 btn_pr = get_button_from_point(obj, &p);
442 /*Handle the case where there is no button there*/
443 btnm->btn_id_sel = LV_BTNMATRIX_BTN_NONE;
444 if(btn_pr != LV_BTNMATRIX_BTN_NONE) {
445 if(button_is_inactive(btnm->ctrl_bits[btn_pr]) == false &&
446 button_is_hidden(btnm->ctrl_bits[btn_pr]) == false) {
447 btnm->btn_id_sel = btn_pr;
448 invalidate_button_area(obj, btnm->btn_id_sel); /*Invalidate the new area*/
449 }
450 }
451 else {
452 btnm->btn_id_sel = LV_BTNMATRIX_BTN_NONE;
453 }
454 }
455
456 if(btnm->btn_id_sel != LV_BTNMATRIX_BTN_NONE) {
457 if(button_is_click_trig(btnm->ctrl_bits[btnm->btn_id_sel]) == false &&
458 button_is_popover(btnm->ctrl_bits[btnm->btn_id_sel]) == false &&
459 button_is_inactive(btnm->ctrl_bits[btnm->btn_id_sel]) == false &&
460 button_is_hidden(btnm->ctrl_bits[btnm->btn_id_sel]) == false) {
461 uint32_t b = btnm->btn_id_sel;
462 res = lv_event_send(obj, LV_EVENT_VALUE_CHANGED, &b);
463 if(res != LV_RES_OK) return;
464 }
465 }
466 }
467 else if(code == LV_EVENT_PRESSING) {
468 void * param = lv_event_get_param(e);
469 uint16_t btn_pr = LV_BTNMATRIX_BTN_NONE;
470 /*Search the pressed area*/
471 lv_indev_t * indev = lv_indev_get_act();
472 lv_indev_type_t indev_type = lv_indev_get_type(indev);
473 if(indev_type == LV_INDEV_TYPE_ENCODER || indev_type == LV_INDEV_TYPE_KEYPAD) return;
474
475 lv_indev_get_point(indev, &p);
476 btn_pr = get_button_from_point(obj, &p);
477 /*Invalidate to old and the new areas*/
478 if(btn_pr != btnm->btn_id_sel) {
479 if(btnm->btn_id_sel != LV_BTNMATRIX_BTN_NONE) {
480 invalidate_button_area(obj, btnm->btn_id_sel);
481 }
482
483 btnm->btn_id_sel = btn_pr;
484
485 lv_indev_reset_long_press(param); /*Start the log press time again on the new button*/
486 if(btn_pr != LV_BTNMATRIX_BTN_NONE &&
487 button_is_inactive(btnm->ctrl_bits[btn_pr]) == false &&
488 button_is_hidden(btnm->ctrl_bits[btn_pr]) == false) {
489 invalidate_button_area(obj, btn_pr);
490 /*Send VALUE_CHANGED for the newly pressed button*/
491 if(button_is_click_trig(btnm->ctrl_bits[btn_pr]) == false &&
492 button_is_popover(btnm->ctrl_bits[btnm->btn_id_sel]) == false) {
493 uint32_t b = btn_pr;
494 res = lv_event_send(obj, LV_EVENT_VALUE_CHANGED, &b);
495 if(res != LV_RES_OK) return;
496 }
497 }
498 }
499 }
500 else if(code == LV_EVENT_RELEASED) {
501 if(btnm->btn_id_sel != LV_BTNMATRIX_BTN_NONE) {
502 /*Toggle the button if enabled*/
503 if(button_is_checkable(btnm->ctrl_bits[btnm->btn_id_sel]) &&
504 !button_is_inactive(btnm->ctrl_bits[btnm->btn_id_sel])) {
505 if(button_get_checked(btnm->ctrl_bits[btnm->btn_id_sel]) && !btnm->one_check) {
506 btnm->ctrl_bits[btnm->btn_id_sel] &= (~LV_BTNMATRIX_CTRL_CHECKED);
507 }
508 else {
509 btnm->ctrl_bits[btnm->btn_id_sel] |= LV_BTNMATRIX_CTRL_CHECKED;
510 }
511 if(btnm->one_check) make_one_button_checked(obj, btnm->btn_id_sel);
512 }
513
514
515 if((button_is_click_trig(btnm->ctrl_bits[btnm->btn_id_sel]) == true ||
516 button_is_popover(btnm->ctrl_bits[btnm->btn_id_sel]) == true) &&
517 button_is_inactive(btnm->ctrl_bits[btnm->btn_id_sel]) == false &&
518 button_is_hidden(btnm->ctrl_bits[btnm->btn_id_sel]) == false) {
519 uint32_t b = btnm->btn_id_sel;
520 res = lv_event_send(obj, LV_EVENT_VALUE_CHANGED, &b);
521 if(res != LV_RES_OK) return;
522 }
523 }
524
525 /*Invalidate to old pressed area*/;
526 invalidate_button_area(obj, btnm->btn_id_sel);
527
528 }
529 else if(code == LV_EVENT_LONG_PRESSED_REPEAT) {
530 if(btnm->btn_id_sel != LV_BTNMATRIX_BTN_NONE) {
531 if(button_is_repeat_disabled(btnm->ctrl_bits[btnm->btn_id_sel]) == false &&
532 button_is_inactive(btnm->ctrl_bits[btnm->btn_id_sel]) == false &&
533 button_is_hidden(btnm->ctrl_bits[btnm->btn_id_sel]) == false) {
534 uint32_t b = btnm->btn_id_sel;
535 res = lv_event_send(obj, LV_EVENT_VALUE_CHANGED, &b);
536 if(res != LV_RES_OK) return;
537 }
538 }
539 }
540 else if(code == LV_EVENT_PRESS_LOST) {
541 invalidate_button_area(obj, btnm->btn_id_sel);
542 btnm->btn_id_sel = LV_BTNMATRIX_BTN_NONE;
543 }
544 else if(code == LV_EVENT_FOCUSED) {
545 if(btnm->btn_cnt == 0) return;
546
547 lv_indev_t * indev = lv_event_get_param(e);
548 lv_indev_type_t indev_type = lv_indev_get_type(indev);
549
550 /*If not focused by an input device assume the last input device*/
551 if(indev == NULL) {
552 indev = lv_indev_get_next(NULL);
553 indev_type = lv_indev_get_type(indev);
554 }
555
556 bool editing = lv_group_get_editing(lv_obj_get_group(obj));
557 /*Focus the first button if there is not selected button*/
558 if(btnm->btn_id_sel == LV_BTNMATRIX_BTN_NONE) {
559 if(indev_type == LV_INDEV_TYPE_KEYPAD || (indev_type == LV_INDEV_TYPE_ENCODER && editing)) {
560 uint32_t b = 0;
561 if(btnm->one_check) {
562 while(button_is_hidden(btnm->ctrl_bits[b]) || button_is_inactive(btnm->ctrl_bits[b]) ||
563 button_is_checked(btnm->ctrl_bits[b]) == false) b++;
564 }
565 else {
566 while(button_is_hidden(btnm->ctrl_bits[b]) || button_is_inactive(btnm->ctrl_bits[b])) b++;
567 }
568
569 btnm->btn_id_sel = b;
570 }
571 else {
572 btnm->btn_id_sel = LV_BTNMATRIX_BTN_NONE;
573 }
574 }
575 }
576 else if(code == LV_EVENT_DEFOCUSED || code == LV_EVENT_LEAVE) {
577 if(btnm->btn_id_sel != LV_BTNMATRIX_BTN_NONE) invalidate_button_area(obj, btnm->btn_id_sel);
578 btnm->btn_id_sel = LV_BTNMATRIX_BTN_NONE;
579 }
580 else if(code == LV_EVENT_KEY) {
581
582 invalidate_button_area(obj, btnm->btn_id_sel);
583
584 char c = *((char *)lv_event_get_param(e));
585 if(c == LV_KEY_RIGHT) {
586 if(btnm->btn_id_sel == LV_BTNMATRIX_BTN_NONE) btnm->btn_id_sel = 0;
587 else btnm->btn_id_sel++;
588 if(btnm->btn_id_sel >= btnm->btn_cnt) btnm->btn_id_sel = 0;
589
590 uint16_t btn_id_start = btnm->btn_id_sel;
591 while(button_is_hidden(btnm->ctrl_bits[btnm->btn_id_sel]) || button_is_inactive(btnm->ctrl_bits[btnm->btn_id_sel])) {
592 btnm->btn_id_sel++;
593 if(btnm->btn_id_sel >= btnm->btn_cnt) btnm->btn_id_sel = 0;
594
595 if(btnm->btn_id_sel == btn_id_start) {
596 btnm->btn_id_sel = LV_BTNMATRIX_BTN_NONE;
597 break;
598 }
599 }
600 }
601 else if(c == LV_KEY_LEFT) {
602 if(btnm->btn_id_sel == LV_BTNMATRIX_BTN_NONE) btnm->btn_id_sel = 0;
603
604 if(btnm->btn_id_sel == 0) btnm->btn_id_sel = btnm->btn_cnt - 1;
605 else if(btnm->btn_id_sel > 0) btnm->btn_id_sel--;
606
607 uint16_t btn_id_start = btnm->btn_id_sel;
608 while(button_is_hidden(btnm->ctrl_bits[btnm->btn_id_sel]) || button_is_inactive(btnm->ctrl_bits[btnm->btn_id_sel])) {
609 if(btnm->btn_id_sel > 0) btnm->btn_id_sel--;
610 else btnm->btn_id_sel = btnm->btn_cnt - 1;
611
612 if(btnm->btn_id_sel == btn_id_start) {
613 btnm->btn_id_sel = LV_BTNMATRIX_BTN_NONE;
614 break;
615 }
616 }
617 }
618 else if(c == LV_KEY_DOWN) {
619 lv_coord_t col_gap = lv_obj_get_style_pad_column(obj, LV_PART_MAIN);
620 /*Find the area below the current*/
621 if(btnm->btn_id_sel == LV_BTNMATRIX_BTN_NONE) {
622 btnm->btn_id_sel = 0;
623 while(button_is_hidden(btnm->ctrl_bits[btnm->btn_id_sel]) || button_is_inactive(btnm->ctrl_bits[btnm->btn_id_sel])) {
624 btnm->btn_id_sel++;
625 if(btnm->btn_id_sel >= btnm->btn_cnt) {
626 btnm->btn_id_sel = LV_BTNMATRIX_BTN_NONE;
627 break;
628 }
629 }
630 }
631 else {
632 uint16_t area_below;
633 lv_coord_t pr_center =
634 btnm->button_areas[btnm->btn_id_sel].x1 + (lv_area_get_width(&btnm->button_areas[btnm->btn_id_sel]) >> 1);
635
636 for(area_below = btnm->btn_id_sel; area_below < btnm->btn_cnt; area_below++) {
637 if(btnm->button_areas[area_below].y1 > btnm->button_areas[btnm->btn_id_sel].y1 &&
638 pr_center >= btnm->button_areas[area_below].x1 &&
639 pr_center <= btnm->button_areas[area_below].x2 + col_gap &&
640 button_is_inactive(btnm->ctrl_bits[area_below]) == false &&
641 button_is_hidden(btnm->ctrl_bits[area_below]) == false) {
642 break;
643 }
644 }
645
646 if(area_below < btnm->btn_cnt) btnm->btn_id_sel = area_below;
647 }
648 }
649 else if(c == LV_KEY_UP) {
650 lv_coord_t col_gap = lv_obj_get_style_pad_column(obj, LV_PART_MAIN);
651 /*Find the area below the current*/
652 if(btnm->btn_id_sel == LV_BTNMATRIX_BTN_NONE) {
653 btnm->btn_id_sel = 0;
654 while(button_is_hidden(btnm->ctrl_bits[btnm->btn_id_sel]) || button_is_inactive(btnm->ctrl_bits[btnm->btn_id_sel])) {
655 btnm->btn_id_sel++;
656 if(btnm->btn_id_sel >= btnm->btn_cnt) {
657 btnm->btn_id_sel = LV_BTNMATRIX_BTN_NONE;
658 break;
659 }
660 }
661 }
662 else {
663 int16_t area_above;
664 lv_coord_t pr_center =
665 btnm->button_areas[btnm->btn_id_sel].x1 + (lv_area_get_width(&btnm->button_areas[btnm->btn_id_sel]) >> 1);
666
667 for(area_above = btnm->btn_id_sel; area_above >= 0; area_above--) {
668 if(btnm->button_areas[area_above].y1 < btnm->button_areas[btnm->btn_id_sel].y1 &&
669 pr_center >= btnm->button_areas[area_above].x1 - col_gap &&
670 pr_center <= btnm->button_areas[area_above].x2 &&
671 button_is_inactive(btnm->ctrl_bits[area_above]) == false &&
672 button_is_hidden(btnm->ctrl_bits[area_above]) == false) {
673 break;
674 }
675 }
676 if(area_above >= 0) btnm->btn_id_sel = area_above;
677 }
678 }
679
680 invalidate_button_area(obj, btnm->btn_id_sel);
681 }
682 else if(code == LV_EVENT_DRAW_MAIN) {
683 draw_main(e);
684 }
685

688static void draw_main(lv_event_t * e)
689{
690 lv_obj_t * obj = lv_event_get_target(e);
691 lv_btnmatrix_t * btnm = (lv_btnmatrix_t *)obj;
692 if(btnm->btn_cnt == 0) return;
693
694 lv_draw_ctx_t * draw_ctx = lv_event_get_draw_ctx(e);
695 obj->skip_trans = 1;
696
697 lv_area_t area_obj;
698 lv_obj_get_coords(obj, &area_obj);
699
700 lv_area_t btn_area;
701
702 uint16_t btn_i = 0;
703 uint16_t txt_i = 0;
704
705 lv_draw_rect_dsc_t draw_rect_dsc_act;
706 lv_draw_label_dsc_t draw_label_dsc_act;
707
708 lv_draw_rect_dsc_t draw_rect_dsc_def;
709 lv_draw_label_dsc_t draw_label_dsc_def;
710
711 lv_state_t state_ori = obj->state;
712 obj->state = LV_STATE_DEFAULT;
713 obj->skip_trans = 1;
714 lv_draw_rect_dsc_init(&draw_rect_dsc_def);
715 lv_draw_label_dsc_init(&draw_label_dsc_def);
716 lv_obj_init_draw_rect_dsc(obj, LV_PART_ITEMS, &draw_rect_dsc_def);
717 lv_obj_init_draw_label_dsc(obj, LV_PART_ITEMS, &draw_label_dsc_def);
718 obj->skip_trans = 0;
719 obj->state = state_ori;
720
721 lv_coord_t ptop = lv_obj_get_style_pad_top(obj, LV_PART_MAIN);
722 lv_coord_t pbottom = lv_obj_get_style_pad_bottom(obj, LV_PART_MAIN);
723 lv_coord_t pleft = lv_obj_get_style_pad_left(obj, LV_PART_MAIN);
724 lv_coord_t pright = lv_obj_get_style_pad_right(obj, LV_PART_MAIN);
725
726#if LV_USE_ARABIC_PERSIAN_CHARS
727 const size_t txt_ap_size = 256 ;
728 char * txt_ap = lv_mem_buf_get(txt_ap_size);
729#endif
730
731 lv_obj_draw_part_dsc_t part_draw_dsc;
732 lv_obj_draw_dsc_init(&part_draw_dsc, draw_ctx);
733 part_draw_dsc.part = LV_PART_ITEMS;
734 part_draw_dsc.class_p = MY_CLASS;
735 part_draw_dsc.type = LV_BTNMATRIX_DRAW_PART_BTN;
736 part_draw_dsc.rect_dsc = &draw_rect_dsc_act;
737 part_draw_dsc.label_dsc = &draw_label_dsc_act;
738
739 for(btn_i = 0; btn_i < btnm->btn_cnt; btn_i++, txt_i++) {
740 /*Search the next valid text in the map*/
741 while(strcmp(btnm->map_p[txt_i], "\n") == 0) {
742 txt_i++;
743 }
744
745 /*Skip hidden buttons*/
746 if(button_is_hidden(btnm->ctrl_bits[btn_i])) continue;
747
748 /*Get the state of the button*/
749 lv_state_t btn_state = LV_STATE_DEFAULT;
750 if(button_get_checked(btnm->ctrl_bits[btn_i])) btn_state |= LV_STATE_CHECKED;
751
752 if(button_is_inactive(btnm->ctrl_bits[btn_i])) btn_state |= LV_STATE_DISABLED;
753 else if(btn_i == btnm->btn_id_sel) {
754 if(state_ori & LV_STATE_PRESSED) btn_state |= LV_STATE_PRESSED;
755 if(state_ori & LV_STATE_FOCUSED) btn_state |= LV_STATE_FOCUSED;
756 if(state_ori & LV_STATE_FOCUS_KEY) btn_state |= LV_STATE_FOCUS_KEY;
757 if(state_ori & LV_STATE_EDITED) btn_state |= LV_STATE_EDITED;
758 }
759
760 /*Get the button's area*/
761 lv_area_copy(&btn_area, &btnm->button_areas[btn_i]);
762 btn_area.x1 += area_obj.x1;
763 btn_area.y1 += area_obj.y1;
764 btn_area.x2 += area_obj.x1;
765 btn_area.y2 += area_obj.y1;
766
767 /*Set up the draw descriptors*/
768 if(btn_state == LV_STATE_DEFAULT) {
769 lv_memcpy(&draw_rect_dsc_act, &draw_rect_dsc_def, sizeof(lv_draw_rect_dsc_t));
770 lv_memcpy(&draw_label_dsc_act, &draw_label_dsc_def, sizeof(lv_draw_label_dsc_t));
771 }
772 /*In other cases get the styles directly without caching them*/
773 else {
774 obj->state = btn_state;
775 obj->skip_trans = 1;
776 lv_draw_rect_dsc_init(&draw_rect_dsc_act);
777 lv_draw_label_dsc_init(&draw_label_dsc_act);
778 lv_obj_init_draw_rect_dsc(obj, LV_PART_ITEMS, &draw_rect_dsc_act);
779 lv_obj_init_draw_label_dsc(obj, LV_PART_ITEMS, &draw_label_dsc_act);
780 obj->state = state_ori;
781 obj->skip_trans = 0;
782 }
783
784 bool recolor = button_is_recolor(btnm->ctrl_bits[btn_i]);
785 if(recolor) draw_label_dsc_act.flag |= LV_TEXT_FLAG_RECOLOR;
786 else draw_label_dsc_act.flag &= ~LV_TEXT_FLAG_RECOLOR;
787
788
789 part_draw_dsc.draw_area = &btn_area;
790 part_draw_dsc.id = btn_i;
791 lv_event_send(obj, LV_EVENT_DRAW_PART_BEGIN, &part_draw_dsc);
792
793 /*Remove borders on the edges if `LV_BORDER_SIDE_INTERNAL`*/
794 if(draw_rect_dsc_act.border_side & LV_BORDER_SIDE_INTERNAL) {
795 draw_rect_dsc_act.border_side = LV_BORDER_SIDE_FULL;
796 if(btn_area.x1 == obj->coords.x1 + pleft) draw_rect_dsc_act.border_side &= ~LV_BORDER_SIDE_LEFT;
797 if(btn_area.x2 == obj->coords.x2 - pright) draw_rect_dsc_act.border_side &= ~LV_BORDER_SIDE_RIGHT;
798 if(btn_area.y1 == obj->coords.y1 + ptop) draw_rect_dsc_act.border_side &= ~LV_BORDER_SIDE_TOP;
799 if(btn_area.y2 == obj->coords.y2 - pbottom) draw_rect_dsc_act.border_side &= ~LV_BORDER_SIDE_BOTTOM;
800 }
801
802 lv_coord_t btn_height = lv_area_get_height(&btn_area);
803
804 if((btn_state & LV_STATE_PRESSED) && (btnm->ctrl_bits[btn_i] & LV_BTNMATRIX_CTRL_POPOVER)) {
805 /*Push up the upper boundary of the btn area to create the popover*/
806 btn_area.y1 -= btn_height;
807 }
808
809 /*Draw the background*/
810 lv_draw_rect(draw_ctx, &draw_rect_dsc_act, &btn_area);
811
812 /*Calculate the size of the text*/
813 const lv_font_t * font = draw_label_dsc_act.font;
814 lv_coord_t letter_space = draw_label_dsc_act.letter_space;
815 lv_coord_t line_space = draw_label_dsc_act.line_space;
816 const char * txt = btnm->map_p[txt_i];
817
818#if LV_USE_ARABIC_PERSIAN_CHARS
819 /*Get the size of the Arabic text and process it*/
820 size_t len_ap = _lv_txt_ap_calc_bytes_cnt(txt);
821 if(len_ap < txt_ap_size) {
822 _lv_txt_ap_proc(txt, txt_ap);
823 txt = txt_ap;
824 }
825#endif
826 lv_point_t txt_size;
827 lv_txt_get_size(&txt_size, txt, font, letter_space,
828 line_space, lv_area_get_width(&area_obj), draw_label_dsc_act.flag);
829
830 btn_area.x1 += (lv_area_get_width(&btn_area) - txt_size.x) / 2;
831 btn_area.y1 += (lv_area_get_height(&btn_area) - txt_size.y) / 2;
832 btn_area.x2 = btn_area.x1 + txt_size.x;
833 btn_area.y2 = btn_area.y1 + txt_size.y;
834
835 if((btn_state & LV_STATE_PRESSED) && (btnm->ctrl_bits[btn_i] & LV_BTNMATRIX_CTRL_POPOVER)) {
836 /*Push up the button text into the popover*/
837 btn_area.y1 -= btn_height / 2;
838 btn_area.y2 -= btn_height / 2;
839 }
840
841 /*Draw the text*/
842 lv_draw_label(draw_ctx, &draw_label_dsc_act, &btn_area, txt, NULL);
843
844 lv_event_send(obj, LV_EVENT_DRAW_PART_END, &part_draw_dsc);
845 }
846
847 obj->skip_trans = 0;
848#if LV_USE_ARABIC_PERSIAN_CHARS
849 lv_mem_buf_release(txt_ap);
850#endif
851}
852/**
853 * Create the required number of buttons and control bytes according to a map
854 * @param obj pointer to button matrix object
855 * @param map_p pointer to a string array
856 */
857static void allocate_btn_areas_and_controls(const lv_obj_t * obj, const char ** map)
858{
859 lv_btnmatrix_t * btnm = (lv_btnmatrix_t *)obj;
860 btnm->row_cnt = 1;
861 /*Count the buttons in the map*/
862 uint16_t btn_cnt = 0;
863 uint16_t i = 0;
864 while(map[i] && map[i][0] != '\0') {
865 if(strcmp(map[i], "\n") != 0) { /*Do not count line breaks*/
866 btn_cnt++;
867 }
868 else {
869 btnm->row_cnt++;
870 }
871 i++;
872 }
873
874 /*Do not allocate memory for the same amount of buttons*/
875 if(btn_cnt == btnm->btn_cnt) return;
876
877 if(btnm->button_areas != NULL) {
878 lv_mem_free(btnm->button_areas);
879 btnm->button_areas = NULL;
880 }
881 if(btnm->ctrl_bits != NULL) {
882 lv_mem_free(btnm->ctrl_bits);
883 btnm->ctrl_bits = NULL;
884 }
885
886 btnm->button_areas = lv_mem_alloc(sizeof(lv_area_t) * btn_cnt);
887 LV_ASSERT_MALLOC(btnm->button_areas);
888 btnm->ctrl_bits = lv_mem_alloc(sizeof(lv_btnmatrix_ctrl_t) * btn_cnt);
889 LV_ASSERT_MALLOC(btnm->ctrl_bits);
890 if(btnm->button_areas == NULL || btnm->ctrl_bits == NULL) btn_cnt = 0;
891
892 lv_memset_00(btnm->ctrl_bits, sizeof(lv_btnmatrix_ctrl_t) * btn_cnt);
893
894 btnm->btn_cnt = btn_cnt;
895}

952/**
953 * Gives the button id of a button under a given point
954 * @param obj pointer to a button matrix object
955 * @param p a point with absolute coordinates
956 * @return the id of the button or LV_BTNMATRIX_BTN_NONE.
957 */
958static uint16_t get_button_from_point(lv_obj_t * obj, lv_point_t * p)
959{
960 lv_area_t obj_cords;
961 lv_area_t btn_area;
962 lv_btnmatrix_t * btnm = (lv_btnmatrix_t *)obj;;
963 uint16_t i;
964 lv_obj_get_coords(obj, &obj_cords);
965
966 lv_coord_t w = lv_obj_get_width(obj);
967 lv_coord_t h = lv_obj_get_height(obj);
968 lv_coord_t pleft = lv_obj_get_style_pad_left(obj, LV_PART_MAIN);
969 lv_coord_t pright = lv_obj_get_style_pad_right(obj, LV_PART_MAIN);
970 lv_coord_t ptop = lv_obj_get_style_pad_top(obj, LV_PART_MAIN);
971 lv_coord_t pbottom = lv_obj_get_style_pad_bottom(obj, LV_PART_MAIN);
972 lv_coord_t prow = lv_obj_get_style_pad_row(obj, LV_PART_MAIN);
973 lv_coord_t pcol = lv_obj_get_style_pad_column(obj, LV_PART_MAIN);
974
975 /*Get the half gap. Button look larger with this value. (+1 for rounding error)*/
976 prow = (prow / 2) + 1 + (prow & 1);
977 pcol = (pcol / 2) + 1 + (pcol & 1);
978
979 prow = LV_MIN(prow, BTN_EXTRA_CLICK_AREA_MAX);
980 pcol = LV_MIN(pcol, BTN_EXTRA_CLICK_AREA_MAX);
981 pright = LV_MIN(pright, BTN_EXTRA_CLICK_AREA_MAX);
982 ptop = LV_MIN(ptop, BTN_EXTRA_CLICK_AREA_MAX);
983 pbottom = LV_MIN(pbottom, BTN_EXTRA_CLICK_AREA_MAX);
984
985 for(i = 0; i < btnm->btn_cnt; i++) {
986 lv_area_copy(&btn_area, &btnm->button_areas[i]);
987 if(btn_area.x1 <= pleft) btn_area.x1 += obj_cords.x1 - LV_MIN(pleft, BTN_EXTRA_CLICK_AREA_MAX);
988 else btn_area.x1 += obj_cords.x1 - pcol;
989
990 if(btn_area.y1 <= ptop) btn_area.y1 += obj_cords.y1 - LV_MIN(ptop, BTN_EXTRA_CLICK_AREA_MAX);
991 else btn_area.y1 += obj_cords.y1 - prow;
992
993 if(btn_area.x2 >= w - pright - 2) btn_area.x2 += obj_cords.x1 + LV_MIN(pright,
994 BTN_EXTRA_CLICK_AREA_MAX); /*-2 for rounding error*/
995 else btn_area.x2 += obj_cords.x1 + pcol;
996
997 if(btn_area.y2 >= h - pbottom - 2) btn_area.y2 += obj_cords.y1 + LV_MIN(pbottom,
998 BTN_EXTRA_CLICK_AREA_MAX); /*-2 for rounding error*/
999 else btn_area.y2 += obj_cords.y1 + prow;
1000
1001 if(_lv_area_is_point_on(&btn_area, p, 0) != false) {
1002 break;
1003 }
1004 }
1005
1006 if(i == btnm->btn_cnt) i = LV_BTNMATRIX_BTN_NONE;
1007
1008 return i;
1009}
1010
1011static void invalidate_button_area(const lv_obj_t * obj, uint16_t btn_idx)
1012{
1013 if(btn_idx == LV_BTNMATRIX_BTN_NONE) return;
1014
1015 lv_area_t btn_area;
1016 lv_area_t obj_area;
1017
1018 lv_btnmatrix_t * btnm = (lv_btnmatrix_t *)obj;;
1019 if(btn_idx >= btnm->btn_cnt) return;
1020
1021 lv_area_copy(&btn_area, &btnm->button_areas[btn_idx]);
1022 lv_obj_get_coords(obj, &obj_area);
1023
1024 /*The buttons might have outline and shadow so make the invalidation larger with the gaps between the buttons.
1025 *It assumes that the outline or shadow is smaller than the gaps*/
1026 lv_coord_t row_gap = lv_obj_get_style_pad_row(obj, LV_PART_MAIN);
1027 lv_coord_t col_gap = lv_obj_get_style_pad_column(obj, LV_PART_MAIN);
1028
1029 /*Be sure to have a minimal extra space if row/col_gap is small*/
1030 lv_coord_t dpi = lv_disp_get_dpi(lv_obj_get_disp(obj));
1031 row_gap = LV_MAX(row_gap, dpi / 10);
1032 col_gap = LV_MAX(col_gap, dpi / 10);
1033
1034 /*Convert relative coordinates to absolute*/
1035 btn_area.x1 += obj_area.x1 - row_gap;
1036 btn_area.y1 += obj_area.y1 - col_gap;
1037 btn_area.x2 += obj_area.x1 + row_gap;
1038 btn_area.y2 += obj_area.y1 + col_gap;
