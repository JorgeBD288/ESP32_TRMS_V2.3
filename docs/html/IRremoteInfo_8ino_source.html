71void dumpTIMER() {

85#endif
86#if defined(IR_USE_TIMER4)
87 Serial.print(F("Timer defined for use: ")); Serial.println(F("Timer4")); flag = true;
88#endif
89#if defined(IR_USE_TIMER5)
90 Serial.print(F("Timer defined for use: ")); Serial.println(F("Timer5")); flag = true;
91#endif
92#if defined(IR_USE_TIMER4_HS)
93 Serial.print(F("Timer defined for use: ")); Serial.println(F("Timer4_HS")); flag = true;
94#endif
95#if defined(IR_USE_TIMER_CMT)
96 Serial.print(F("Timer defined for use: ")); Serial.println(F("Timer_CMT")); flag = true;
97#endif
98#if defined(IR_USE_TIMER_TPM1)
99 Serial.print(F("Timer defined for use: ")); Serial.println(F("Timer_TPM1")); flag = true;
100#endif
101#if defined(IR_USE_TIMER_TINY0)
102 Serial.print(F("Timer defined for use: ")); Serial.println(F("Timer_TINY0")); flag = true;
103#endif
104
105 if (!flag) {
106 Serial.print(F("Timer Error: "));
107 Serial.println(F("not defined"));
108 }
