115// This function scrolls and wraps the graphic in a 1 bit per pixel Sprite, dy and dy
116// control the scroll direction. Pixels that scroll off one side appear on the other.
117// Scrolling is achieved by plotting one Sprite inside another with an offset. This has
118// to be done by plotting 4 times into a second frame buffer Sprite.
119//==========================================================================================
120void wrappingScroll(int16_t dx, int16_t dy)
121{
122 // Position the quadrants so they overlap all areas of the buffer
123 scroll_x += dx;
124 if (scroll_x < -WIDTH) scroll_x += WIDTH;
125 if (scroll_x > 0) scroll_x -= WIDTH;
126
127 scroll_y += dy;
128 if (scroll_y < -HEIGHT) scroll_y += HEIGHT;
129 if (scroll_y > 0) scroll_y -= HEIGHT;
130
131 // push the 4 quadrants of gfx. sprite into the fb. sprite
132 // pushImage will do the cropping
133 fb.pushImage(scroll_x, scroll_y, WIDTH, HEIGHT, gfxPtr);
134 fb.pushImage(scroll_x + WIDTH, scroll_y, WIDTH, HEIGHT, gfxPtr);
135 fb.pushImage(scroll_x, scroll_y + HEIGHT, WIDTH, HEIGHT, gfxPtr);
136 fb.pushImage(scroll_x + WIDTH, scroll_y + HEIGHT, WIDTH, HEIGHT, gfxPtr);
137}
