196static bool PID_LoadConfigFromFile(const char *path)
197{
198 // Asegurar montaje SD
199 if (!SD_EnsureMounted()) {
200 Serial.println("No se puede guardar: SD no montada");
201 return false;
202 }
203
204 File f = SD.open(path, FILE_READ);
205 if (!f) {
206 Serial.print("ERROR abriendo fichero de config: ");
207 Serial.println(path);
208 return false;
209 }
210
211 Serial.print("Leyendo config PID de: ");
212 Serial.println(path);
213
214 PIDSection section = PID_SECTION_NONE;
215
216 while (f.available()) {
217 String line = f.readStringUntil('\n');
218 line.trim();
219 if (line.length() == 0) continue;
220 if (line.startsWith("#")) continue;
221
222 // Sección
223 if (line.startsWith("[")) {
224 if (line == "[PID_MIN]") section = PID_SECTION_MIN;
225 else if (line == "[PID_CURR]") section = PID_SECTION_CURR;
226 else if (line == "[PID_MAX]") section = PID_SECTION_MAX;
227 else section = PID_SECTION_NONE;
228 continue;
229 }
230
231 int eqPos = line.indexOf('=');
232 if (eqPos < 0) continue;
233
234 String key = line.substring(0, eqPos);
235 String val = line.substring(eqPos + 1);
236 key.trim();
237 val.trim();
238
239 float v = val.toFloat();
