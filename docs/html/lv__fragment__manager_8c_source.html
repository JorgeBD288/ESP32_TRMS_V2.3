65lv_fragment_manager_t * lv_fragment_manager_create(lv_fragment_t * parent)
66{
67 lv_fragment_manager_t * instance = lv_mem_alloc(sizeof(lv_fragment_manager_t));
68 lv_memset_00(instance, sizeof(lv_fragment_manager_t));
69 instance->parent = parent;
70 _lv_ll_init(&instance->attached, sizeof(lv_fragment_managed_states_t));
71 _lv_ll_init(&instance->stack, sizeof(lv_fragment_stack_item_t));
72 return instance;
73}

88void lv_fragment_manager_create_obj(lv_fragment_manager_t * manager)
89{
90 LV_ASSERT_NULL(manager);
91 lv_fragment_stack_item_t * top = _lv_ll_get_tail(&manager->stack);
92 lv_fragment_managed_states_t * states = NULL;
93 _LV_LL_READ(&manager->attached, states) {
94 if(states->in_stack && top->states != states) {
95 /*Only create obj for top item in stack*/

119void lv_fragment_manager_remove(lv_fragment_manager_t * manager, lv_fragment_t * fragment)
120{
121 LV_ASSERT_NULL(manager);
122 LV_ASSERT_NULL(fragment);
123 LV_ASSERT_NULL(fragment->managed);
124 LV_ASSERT(fragment->managed->manager == manager);
125 lv_fragment_managed_states_t * states = fragment->managed;
126 lv_fragment_managed_states_t * prev = NULL;
127 bool was_top = false;
128 if(states->in_stack) {
129 void * stack_top = _lv_ll_get_tail(&manager->stack);
130 lv_fragment_stack_item_t * item = NULL;
131 _LV_LL_READ_BACK(&manager->stack, item) {
132 if(item->states == states) {
133 was_top = stack_top == item;
134 void * stack_prev = _lv_ll_get_prev(&manager->stack, item);
135 if(!stack_prev) break;
136 prev = ((lv_fragment_stack_item_t *) stack_prev)->states;
137 break;
138 }
139 }
140 if(item) {
141 _lv_ll_remove(&manager->stack, item);
142 lv_mem_free(item);
143 }
144 }
145 item_del_obj(states);
146 item_del_fragment(states);
147 _lv_ll_remove(&manager->attached, states);
148 lv_mem_free(states);
149 if(prev && was_top) {
150 item_create_obj(prev);
151 }
152}

154void lv_fragment_manager_push(lv_fragment_manager_t * manager, lv_fragment_t * fragment, lv_obj_t * const * container)
155{
156 lv_fragment_stack_item_t * top = _lv_ll_get_tail(&manager->stack);
157 if(top != NULL) {
158 item_del_obj(top->states);
159 }
160 lv_fragment_managed_states_t * states = fragment_attach(manager, fragment, container);
161 states->in_stack = true;
162 /*Add fragment to the top of the stack*/
163 lv_fragment_stack_item_t * item = _lv_ll_ins_tail(&manager->stack);
164 lv_memset_00(item, sizeof(lv_fragment_stack_item_t));
165 item->states = states;
166 item_create_obj(states);
167}

169bool lv_fragment_manager_pop(lv_fragment_manager_t * manager)

171 lv_fragment_t * top = lv_fragment_manager_get_top(manager);
172 if(top == NULL) return false;
173 lv_fragment_manager_remove(manager, top);

187bool lv_fragment_manager_send_event(lv_fragment_manager_t * manager, int code, void * userdata)
188{
189 LV_ASSERT_NULL(manager);
190 lv_fragment_managed_states_t * p = NULL;
191 _LV_LL_READ_BACK(&manager->attached, p) {
192 if(!p->obj_created || p->destroying_obj) continue;
193 lv_fragment_t * instance = p->instance;
194 if(!instance) continue;
195 if(lv_fragment_manager_send_event(instance->child_manager, code, userdata)) return true;
196 if(p->cls->event_cb && p->cls->event_cb(instance, code, userdata)) return true;
197 }
198 return false;
199}

250static void item_del_fragment(lv_fragment_managed_states_t * item)
251{
252 lv_fragment_t * instance = item->instance;
253 if(instance->cls->detached_cb) {
254 instance->cls->detached_cb(instance);
255 }
256 instance->managed = NULL;
257 lv_fragment_del(instance);
258 item->instance = NULL;
259}

262static lv_fragment_managed_states_t * fragment_attach(lv_fragment_manager_t * manager, lv_fragment_t * fragment,
263 lv_obj_t * const * container)
264{
265 LV_ASSERT(manager);
266 LV_ASSERT(fragment);
267 LV_ASSERT(fragment->managed == NULL);
268 lv_fragment_managed_states_t * states = _lv_ll_ins_tail(&manager->attached);
269 lv_memset_00(states, sizeof(lv_fragment_managed_states_t));
270 states->cls = fragment->cls;
271 states->manager = manager;
272 states->container = container;
273 states->instance = fragment;
274 fragment->managed = states;
275 if(fragment->cls->attached_cb) {
276 fragment->cls->attached_cb(fragment);
277 }
278 return states;
279}
