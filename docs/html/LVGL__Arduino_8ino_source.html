30/* Display flushing */
31void my_disp_flush( lv_disp_drv_t *disp_drv, const lv_area_t *area, lv_color_t *color_p )
32{
33 uint32_t w = ( area->x2 - area->x1 + 1 );
34 uint32_t h = ( area->y2 - area->y1 + 1 );
35
36 tft.startWrite();
37 tft.setAddrWindow( area->x1, area->y1, w, h );
38 tft.pushColors( ( uint16_t * )&color_p->full, w * h, true );
39 tft.endWrite();
40
41 lv_disp_flush_ready( disp_drv );
42}
