1/*
2 * IRCommandDispatcher.hpp
3 *
4 * Library to process IR commands by calling functions specified in a mapping array.
5 * Commands can be tagged as blocking or non blocking.
6 *
7 * To run this example you need to install the "IRremote" or "IRMP" library.
8 * Install it under "Tools -> Manage Libraries..." or "Ctrl+Shift+I"
9 *
10 * The IR library calls a callback function, which executes a non blocking command directly in ISR (Interrupt Service Routine) context!
11 * A blocking command is stored and sets a stop flag for an already running blocking function to terminate.
12 * The blocking command can in turn be executed by main loop by calling IRDispatcher.checkAndRunSuspendedBlockingCommands().

178/*
179 * The main dispatcher function called by IR-ISR, main loop and checkAndRunSuspendedBlockingCommands()
180 * Non blocking commands are executed directly, blocking commands are executed if enabled by parameter and no other command is just running.
181 * Otherwise request to stop (requestToStopReceived) is set and command is stored for main loop to be later execute by checkAndRunSuspendedBlockingCommands().
182 * Sets flags justCalledRegularIRCommand, executingBlockingCommand, requestToStopReceived
183 * @param aCallBlockingCommandImmediately Run blocking command directly, if no other command is just running. Should be false if called by ISR in order not to block ISR.
184 */
185void IRCommandDispatcher::checkAndCallCommand(bool aCallBlockingCommandImmediately) {
186 if (IRReceivedData.command == COMMAND_EMPTY) {
187 return;
188 }
189
190 /*
191 * Search for command in Array of IRToCommandMappingStruct
192 */
193 for (uint_fast8_t i = 0; i < sizeof(IRMapping) / sizeof(struct IRToCommandMappingStruct); ++i) {
194 if (IRReceivedData.command == IRMapping[i].IRCode) {
195 /*
196 * Command found
197 */
198#if defined(LOCAL_INFO)
199 const __FlashStringHelper *tCommandName = reinterpret_cast<const __FlashStringHelper*>(IRMapping[i].CommandString);
200#endif
201 /*
202 * Check for repeat and if repeat is allowed for the current command
203 */
204 if (IRReceivedData.isRepeat && !(IRMapping[i].Flags & IR_COMMAND_FLAG_REPEATABLE)) {
205#if defined(LOCAL_DEBUG)
206 Serial.print(F("Repeats of command \""));
207 Serial.print(tCommandName);
208 Serial.println("\" not accepted");
209#endif
210 return;
211 }
212
213 /*
214 * Do not accept recursive call of the same command
215 */
216 if (currentBlockingCommandCalled == IRReceivedData.command) {
217#if defined(LOCAL_DEBUG)
218 Serial.print(F("Recursive command \""));
219 Serial.print(tCommandName);
220 Serial.println("\" not accepted");
221#endif
222 return;
223 }
224
225 /*
226 * Execute commands
227 */
228 bool tIsNonBlockingCommand = (IRMapping[i].Flags & IR_COMMAND_FLAG_NON_BLOCKING);
229 if (tIsNonBlockingCommand) {
230 // short command here, just call
231 CD_INFO_PRINT(F("Run non blocking command: "));
232 CD_INFO_PRINTLN(tCommandName);
233#if defined(BUZZER_PIN) && defined(USE_TINY_IR_RECEIVER)
234 /*
235 * Do (non blocking) buzzer feedback before command is executed
236 */
237 if(IRMapping[i].Flags & IR_COMMAND_FLAG_BEEP) {
238 tone(BUZZER_PIN, 2200, 50);
239 }
240#endif
241 IRMapping[i].CommandToCall();
242 } else {
243 /*
244 * Blocking command here
245 */
246 if (aCallBlockingCommandImmediately && currentBlockingCommandCalled == COMMAND_EMPTY) {
247 /*
248 * Here no blocking command was running and we are called from main loop
249 */
250 requestToStopReceived = false; // Do not stop the command executed now
251 justCalledBlockingCommand = true;
252 currentBlockingCommandCalled = IRReceivedData.command; // set lock for recursive calls
253 lastBlockingCommandCalled = IRReceivedData.command; // set history, can be evaluated by main loop
254 /*
255 * This call is blocking!!!
256 */
257 CD_INFO_PRINT(F("Run blocking command: "));
258 CD_INFO_PRINTLN(tCommandName);
259
260#if defined(BUZZER_PIN) && defined(USE_TINY_IR_RECEIVER)
261 /*
262 * Do (non blocking) buzzer feedback before command is executed
263 */
264 if(IRMapping[i].Flags & IR_COMMAND_FLAG_BEEP) {
265 tone(BUZZER_PIN, 2200, 50);
266 }
267#endif
268
269 IRMapping[i].CommandToCall();
270#if defined(TRACE)
271 Serial.println(F("End of blocking command"));
272#endif
273 currentBlockingCommandCalled = COMMAND_EMPTY;
274 } else {
275 /*
276 * Called by ISR or another command still running.
277 * Do not run command directly, but set request to stop to true and store command
278 * for main loop to execute by checkAndRunSuspendedBlockingCommands()
279 */
280 BlockingCommandToRunNext = IRReceivedData.command;
281 requestToStopReceived = true; // to stop running command
282 CD_INFO_PRINT(F("Requested stop and stored blocking command "));
283 CD_INFO_PRINT(tCommandName);
284 CD_INFO_PRINTLN(F(" as next command to run."));
285 }
286 }
287 break; // command found
288 } // if (IRReceivedData.command == IRMapping[i].IRCode)
289 } // for loop
290 return;
291}

293/*
294 * Intended to be called from main loop
295 * @return true, if command was called
296 */
297bool IRCommandDispatcher::checkAndRunSuspendedBlockingCommands() {
298 /*
299 * Take last rejected command and call associated function
300 */
301 if (BlockingCommandToRunNext != COMMAND_EMPTY) {
302
303 CD_INFO_PRINT(F("Run stored command=0x"));
304 CD_INFO_PRINTLN(BlockingCommandToRunNext, HEX);
305
306 IRReceivedData.command = BlockingCommandToRunNext;
307 BlockingCommandToRunNext = COMMAND_EMPTY;
308 IRReceivedData.isRepeat = false;
309 requestToStopReceived = false; // Do not stop the command executed now
310 checkAndCallCommand(true);
311 return true;
312 }
313 return false;
314}

331/*
332 * Special delay function for the IRCommandDispatcher. Returns prematurely if requestToStopReceived is set.
333 * To be used in blocking functions as delay
334 * @return true - as soon as stop received
335 */
336bool IRCommandDispatcher::delayAndCheckForStop(uint16_t aDelayMillis) {
337 uint32_t tStartMillis = millis();
338 do {
339 if (requestToStopReceived) {
340 return true;
341 }
342 } while (millis() - tStartMillis < aDelayMillis);
343 return false;
344}
