154IRControlEvent IRControl_poll() {
155 IRControlEvent ev{};
156 ev.hasEvent = false;
157 ev.command = 0;
158 ev.deltaSlider = 0;
159 ev.screen = IRScreenTarget::NONE; // Cambios entre pantallas
160 ev.nav = IRNavKey::NONE; // Navegación utilizando flechas
161 ev.digit = -1; // Sin dígito
162 ev.minus = false; // Tecla "-" especial
163 ev.power = false; // Tecla POWER
164
165 // Si no hay datos IR, salimos
166 if (!IrReceiver.decode()) {
167 return ev;
168 }
169
170 uint32_t cmd = IrReceiver.decodedIRData.command;
171 ev.hasEvent = true;
172 ev.command = cmd;
173
174 // Registrar actividad para cualquier comando NO CERO y NO 0x40 (ruido)
175 if (cmd != 0x0 && cmd != 0x40) {
176 RegisterActivity();
177
178 // Si el salvapantallas está activo, cualquier tecla lo quita
179 // excepto el botón de "inicio" (0x2) que ya gestiona su propia lógica
180 if (g_screensaverActive) {
181 if (cmd != 0x2) {
182 lv_event_send(ui_Screen10, LV_EVENT_CLICKED, NULL);
183 }
184 }
185 }
186
187 Serial.print("Código IR recibido: 0x");
188 Serial.println(cmd, HEX);
189
190 // -----------------------------
191 // 1) Botón de cambio de pantalla (home)
192 // Lo dejamos fijo en 0x2, como antes
193 // -----------------------------
194 if (cmd == 0x2) {
195 ev.screen = IRScreenTarget::SCREEN1;
196 }
197
198 // -----------------------------
199 // 2) Flechas + ENTER (tabla configurable)
200 // -----------------------------
201 if (cmd == s_navCodeLeft) ev.nav = IRNavKey::LEFT;
202 else if (cmd == s_navCodeRight) ev.nav = IRNavKey::RIGHT;
203 else if (cmd == s_navCodeUp) ev.nav = IRNavKey::UP;
204 else if (cmd == s_navCodeDown) ev.nav = IRNavKey::DOWN;
205 else if (cmd == s_navCodeEnter) ev.nav = IRNavKey::ENTER;
206
207 // -----------------------------
208 // 3) + / - (para sliders)
209 // -----------------------------
210 if (cmd == s_plusCode) {
211 ev.deltaSlider = +10;
212 }
213 if (cmd == s_minusCode) {
214 ev.deltaSlider = -10;
215 ev.minus = true; // reutilizado también para signo en labels numéricas
216 }
217
218 // -----------------------------
219 // 4) POWER
220 // -----------------------------
221 if (s_powerCode != 0 && cmd == s_powerCode) {
222 ev.power = true;
223 }
224
225 // -----------------------------
226 // 5) Dígitos 0–9 (tabla configurable)
227 // -----------------------------
228 for (int d = 0; d < 10; ++d) {
229 if (cmd == s_digitCode[d]) {
230 ev.digit = d;
231 break;
232 }
233 }
234
235 // Preparar para el próximo comando
236 IrReceiver.resume();
237
238 return ev;
239}

245void IRControl_loadConfigFromNVS()
246{
247 if (!s_prefs.begin("ir_codes", true)) { // true = solo lectura
248 Serial.println("[IRControl] ERROR: no se pudo abrir NVS (ir_codes) para lectura");
249 return;
250 }
251
252 // Dígitos 0–9
253 for (uint8_t d = 0; d < 10; d++) {
254 char key[8];
255 snprintf(key, sizeof(key), "d%u", d); // "d0", "d1", ...
256 uint32_t def = s_digitCode[d]; // valor por defecto actual
257 uint32_t val = s_prefs.getUInt(key, def); // si no existe, deja el defecto
258 s_digitCode[d] = val;
259 }
260
261 // Flechas / ENTER
262 s_navCodeLeft = s_prefs.getUInt("navL", s_navCodeLeft);
263 s_navCodeRight = s_prefs.getUInt("navR", s_navCodeRight);
264 s_navCodeUp = s_prefs.getUInt("navU", s_navCodeUp);
265 s_navCodeDown = s_prefs.getUInt("navD", s_navCodeDown);
266 s_navCodeEnter = s_prefs.getUInt("navE", s_navCodeEnter);
267
268 // + / -
269 s_plusCode = s_prefs.getUInt("plus", s_plusCode);
270 s_minusCode = s_prefs.getUInt("minus", s_minusCode);
271
272 // POWER
273 s_powerCode = s_prefs.getUInt("power", s_powerCode);
274
275 s_prefs.end();
276 Serial.println("[IRControl] Configuración IR cargada desde NVS");
277}
