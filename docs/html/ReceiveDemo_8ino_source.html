164void loop() {
165 /*
166 * Check if received data is available and if yes, try to decode it.
167 * Decoded result is in the IrReceiver.decodedIRData structure.
168 *
169 * E.g. command is in IrReceiver.decodedIRData.command
170 * address is in command is in IrReceiver.decodedIRData.address
171 * and up to 32 bit raw data in IrReceiver.decodedIRData.decodedRawData
172 */
173 if (IrReceiver.decode()) {
174 Serial.println();
175#if FLASHEND < 0x3FFF //
176 // For less than 16k flash, only print a minimal summary of received data
177 IrReceiver.printIRResultMinimal(&Serial);
178#else
179
180 /*
181 *
182 */
183 if (IrReceiver.decodedIRData.flags & IRDATA_FLAGS_WAS_OVERFLOW) {
184 handleOverflow();
185 } else {
186 /*
187 * No overflow here.
188 * Stop receiver, generate a single beep, print short info and send usage and start receiver again
189 */
190 if ((IrReceiver.decodedIRData.protocol != SONY) && (IrReceiver.decodedIRData.protocol != PULSE_WIDTH)
191 && (IrReceiver.decodedIRData.protocol != PULSE_DISTANCE) && (IrReceiver.decodedIRData.protocol != UNKNOWN)
192 && digitalRead(DEBUG_BUTTON_PIN) != LOW) {
193 /*
194 * For SONY the tone prevents the detection of a repeat after the 15 ms SONY gap.
195 * In debug mode and for unknown protocols, we need the time for extended output.
196 * Skipping tone will get exact gap time between transmissions and not running into repeat frames while wait for tone to end.
197 * This in turn enables the next CheckForRecordGapsMicros() call a chance to eventually propose a change of the current RECORD_GAP_MICROS value.
198 */
199 generateTone();
200 }
201
202 /*
203 * Print info
204 */
205 if (IrReceiver.decodedIRData.protocol == UNKNOWN || digitalRead(DEBUG_BUTTON_PIN) == LOW) {
206 // We have debug enabled or an unknown protocol, print extended info
207 if (IrReceiver.decodedIRData.protocol == UNKNOWN) {
208 Serial.println(F("Received noise or an unknown (or not yet enabled) protocol"));
209 }
210 IrReceiver.printIRResultRawFormatted(&Serial, true);
211 }
212 if (IrReceiver.decodedIRData.protocol != UNKNOWN) {
213 /*
214 * The info output for a successful receive
215 */
216 IrReceiver.printIRResultShort(&Serial);
217 IrReceiver.printIRSendUsage(&Serial);
218 }
219 }
220#endif // #if FLASHEND >= 0x3FFF
221
222 /*
223 * !!!Important!!! Enable receiving of the next value, because receiving
224 * has stopped after the end of the current received data packet.
225 * Do it here, to preserve raw data for printing with printIRResultRawFormatted()
226 */
227 IrReceiver.resume();
228
229 /*
230 * Finally check the received data and perform actions according to the received address and commands
231 */
232 if (IrReceiver.decodedIRData.address == 0) {
233 if (IrReceiver.decodedIRData.command == 0x10) {
234 // do something
235 } else if (IrReceiver.decodedIRData.command == 0x11) {
236 // do something else
237 }
238 }
239
240 // Check if repeats of the IR command was sent for more than 1000 ms
241 if (detectLongPress(1000)) {
242 Serial.print(F("Command 0x"));
243 Serial.print(IrReceiver.decodedIRData.command, HEX);
244 Serial.println(F(" was repeated for more than 2 seconds"));
245 }
246 } // if (IrReceiver.decode())
247
248 /*
249 * Your code here
250 * For all users of the FastLed library, use this code for strip.show() to improve receiving performance (which is still not 100%):
251 * if (IrReceiver.isIdle()) {
252 * strip.show();
253 * }
254 */
255
256}

261void generateTone() {
262#if !defined(ESP8266) && !defined(NRF5) // tone on esp8266 works only once, then it disables IrReceiver.restartTimer() / timerConfigForReceive().
263# if defined(ESP32) // ESP32 uses another timer for tone(), maybe other platforms (not tested yet) too.
264 tone(TONE_PIN, 2200, 8);
265# else
266 IrReceiver.stopTimer(); // Stop timer consistently before calling tone() or other functions using the timer resource.
267 tone(TONE_PIN, 2200, 8);
268 delay(8);
269 IrReceiver.restartTimer(); // Restart IR timer after timer resource is no longer blocked.
270# endif
271#endif
272}

275 Serial.println(F("Overflow detected"));
276 Serial.println(F("Try to increase the \"RAW_BUFFER_LENGTH\" value of " STR(RAW_BUFFER_LENGTH) " in " __FILE__));

279#if !defined(ESP8266) && !defined(NRF5) // tone on esp8266 works once, then it disables IrReceiver.restartTimer() / timerConfigForReceive().
280 /*

283# if defined(ESP32) // ESP32 uses another timer for tone()

294# endif
295#endif
296}

298unsigned long sMillisOfFirstReceive;
299bool sLongPressJustDetected;
300/**
301 * True once we received the consecutive repeats for more than aLongPressDurationMillis milliseconds.
302 * The first frame, which is no repeat, is NOT counted for the duration!
303 * @return true once after the repeated IR command was received for longer than aLongPressDurationMillis milliseconds, false otherwise.
304 */
305bool detectLongPress(uint16_t aLongPressDurationMillis) {
306 if (!sLongPressJustDetected && (IrReceiver.decodedIRData.flags & IRDATA_FLAGS_IS_REPEAT)) {
307 /*
308 * Here the repeat flag is set (which implies, that command is the same as the previous one)
309 */
310 if (millis() - aLongPressDurationMillis > sMillisOfFirstReceive) {
311 sLongPressJustDetected = true; // Long press here
312 }
313 } else {
314 // No repeat here
315 sMillisOfFirstReceive = millis();
316 sLongPressJustDetected = false;
317 }
318 return sLongPressJustDetected; // No long press here
319}
