1/*
2 Sketch to show creation of a 1bpp sprite with a transparent
3 background, then plot it on the TFT.
4
5 Example for library:
6 https://github.com/Bodmer/TFT_eSPI
7
8 A Sprite is notionally an invisible graphics screen that is
9 kept in the processors RAM. Graphics can be drawn into the
10 Sprite just as it can be drawn directly to the screen. Once
11 the Sprite is completed it can be plotted onto the screen in
12 any position. If there is sufficient RAM then the Sprite can
13 be the same size as the screen and used as a frame buffer.
14
15 A 1-bit Sprite occupies (width * height)/8 bytes in RAM. So,
16 for example, a 320 x 240 pixel Sprite occupies 9600 bytes.
17*/
18// A new setBitmapColor(fg_color, bg_color) function allows
19// any 2 colours to be used for the 1 bit sprite.
20// One colour can also be defined as transparent when
21// rendering to the screen.
22
23
24#include <TFT_eSPI.h> // Include the graphics library (this includes the sprite functions)
25
26TFT_eSPI tft = TFT_eSPI(); // Create object "tft"
27
28TFT_eSprite img = TFT_eSprite(&tft); // Create Sprite object "img" with pointer to "tft" object
29 // the pointer is used by pushSprite() to push it onto the TFT
30
31#define BITS_PER_PIXEL 1 // How many bits per pixel in Sprite
32
33// =========================================================================
34void setup(void) {
35 Serial.begin(250000);
36
37 tft.init();
38
39 tft.setRotation(0);
40}
41// =========================================================================
42void loop() {
43
44 tft.fillScreen(TFT_NAVY);
45
46 // Draw 10 sprites containing a "transparent" colour
47 for (int i = 0; i < 10; i++)
48 {
49 int x = random(240 - 70);
50 int y = random(320 - 80);
51 int c = random(0x10000); // Random colour 0 - 0xFFFF
52 drawStar(x, y, c);
53 }
54
55 delay(2000);
56
57 // Now go bananas and draw 500 more
58 for (int i = 0; i < 500; i++)
59 {
60 int x = random(240 - 70);
61 int y = random(320 - 80);
62 int c = random(0x10000); // Random colour
63 drawStar(x, y, c);
64 yield(); // Stop watchdog reset
65 }
66
67 delay(2000);
68}
69// =========================================================================
70// Create sprite, plot graphics in it, plot to screen, then delete sprite
71// =========================================================================
72void drawStar(int x, int y, int star_color)
73{
74 // 1 bpp colour values can only be 1 or 0 (one or zero)
75 uint16_t transparent = 0; // The transparent colour, can only be 1 or 0
76
77 // Create an 1 bit (2 colour) sprite 70x80 pixels (uses 70*80/8 = 700 bytes of RAM)
78 // Colour depths of 8 bits per pixel and 16 bits are also supported.
79 img.setColorDepth(BITS_PER_PIXEL); // Set colour depth first
80 img.createSprite(70, 80); // then create the sprite
81
82 // Fill Sprite with the colour that will be defined later as "transparent"
83 // We could also fill with any colour as transparent, and later specify that
84 // same colour when we push the Sprite onto the display screen.
85 img.fillSprite(transparent);
86
87 // Draw 2 triangles to create a filled in star
88 img.fillTriangle(35, 0, 0, 59, 69, 59, star_color);
89 img.fillTriangle(35, 79, 0, 20, 69, 20, star_color);
90
91 // Punch a star shaped hole in the middle with a smaller "transparent" star
92 img.fillTriangle(35, 7, 6, 56, 63, 56, transparent);
93 img.fillTriangle(35, 73, 6, 24, 63, 24, transparent);
94
95 // Set the 2 pixel colours that 1 and 0 represent on the display screen
96 img.setBitmapColor(star_color, transparent);
97
98 // Push sprite to TFT screen CGRAM at coordinate x,y (top left corner)
99 // Specify what colour is to be treated as transparent (black in this example)
100 img.pushSprite(x, y, transparent);
101
102 // Delete Sprite to free memory, creating and deleting takes very little time.
103 img.deleteSprite();
104}
