71#if LV_DEMO_MUSIC_LARGE

73#else
74 static const lv_coord_t grid_rows[] = {22, 17, LV_GRID_TEMPLATE_LAST};
75#endif
76 lv_style_init(&style_btn);
77 lv_style_set_bg_opa(&style_btn, LV_OPA_TRANSP);
78 lv_style_set_grid_column_dsc_array(&style_btn, grid_cols);

80 lv_style_set_grid_row_align(&style_btn, LV_GRID_ALIGN_CENTER);

82#if LV_DEMO_MUSIC_LARGE

119 uint32_t track_id;

128 _lv_demo_music_list_btn_check(0, true);

146void _lv_demo_music_list_btn_check(uint32_t track_id, bool state)
147{
148 lv_obj_t * btn = lv_obj_get_child(list, track_id);
149 lv_obj_t * icon = lv_obj_get_child(btn, 0);
150
151 if(state) {
152 lv_obj_add_state(btn, LV_STATE_CHECKED);
153 lv_img_set_src(icon, &img_lv_demo_music_btn_list_pause);
154 lv_obj_scroll_to_view(btn, LV_ANIM_ON);
155 }
156 else {
157 lv_obj_clear_state(btn, LV_STATE_CHECKED);
158 lv_img_set_src(icon, &img_lv_demo_music_btn_list_play);
159 }
160}

166static lv_obj_t * add_list_btn(lv_obj_t * parent, uint32_t track_id)
167{
168 uint32_t t = _lv_demo_music_get_track_length(track_id);
169 char time[32];
170 lv_snprintf(time, sizeof(time), "%"LV_PRIu32":%02"LV_PRIu32, t / 60, t % 60);
171 const char * title = _lv_demo_music_get_title(track_id);
172 const char * artist = _lv_demo_music_get_artist(track_id);
173
174 lv_obj_t * btn = lv_obj_create(parent);
175 lv_obj_remove_style_all(btn);
176#if LV_DEMO_MUSIC_LARGE
177 lv_obj_set_size(btn, lv_pct(100), 110);
178#else
179 lv_obj_set_size(btn, lv_pct(100), 60);
180#endif
181
182 lv_obj_add_style(btn, &style_btn, 0);
183 lv_obj_add_style(btn, &style_btn_pr, LV_STATE_PRESSED);
184 lv_obj_add_style(btn, &style_btn_chk, LV_STATE_CHECKED);
185 lv_obj_add_style(btn, &style_btn_dis, LV_STATE_DISABLED);
186 lv_obj_add_event_cb(btn, btn_click_event_cb, LV_EVENT_CLICKED, NULL);
187
188 if(track_id >= 3) {
189 lv_obj_add_state(btn, LV_STATE_DISABLED);
190 }
191
192 lv_obj_t * icon = lv_img_create(btn);
193 lv_img_set_src(icon, &img_lv_demo_music_btn_list_play);
194 lv_obj_set_grid_cell(icon, LV_GRID_ALIGN_START, 0, 1, LV_GRID_ALIGN_CENTER, 0, 2);
195
196 lv_obj_t * title_label = lv_label_create(btn);
197 lv_label_set_text(title_label, title);
198 lv_obj_set_grid_cell(title_label, LV_GRID_ALIGN_START, 1, 1, LV_GRID_ALIGN_CENTER, 0, 1);
199 lv_obj_add_style(title_label, &style_title, 0);
200
201 lv_obj_t * artist_label = lv_label_create(btn);
202 lv_label_set_text(artist_label, artist);
203 lv_obj_add_style(artist_label, &style_artist, 0);
204 lv_obj_set_grid_cell(artist_label, LV_GRID_ALIGN_START, 1, 1, LV_GRID_ALIGN_CENTER, 1, 1);
205
206 lv_obj_t * time_label = lv_label_create(btn);
207 lv_label_set_text(time_label, time);
208 lv_obj_add_style(time_label, &style_time, 0);
209 lv_obj_set_grid_cell(time_label, LV_GRID_ALIGN_END, 2, 1, LV_GRID_ALIGN_CENTER, 0, 2);
210
211 LV_IMG_DECLARE(img_lv_demo_music_list_border);
212 lv_obj_t * border = lv_img_create(btn);
213 lv_img_set_src(border, &img_lv_demo_music_list_border);
214 lv_obj_set_width(border, lv_pct(120));
215 lv_obj_align(border, LV_ALIGN_BOTTOM_MID, 0, 0);
216 lv_obj_add_flag(border, LV_OBJ_FLAG_IGNORE_LAYOUT);
217
218 return btn;
219}
