49static uint8_t tca_readReg(uint8_t reg) {
50 Wire.beginTransmission(TCA_ADDR);
51 Wire.write(reg);

57static uint8_t tca_readPort(uint8_t port) {

62static void tca_init() {
63 // Todos los pines como entradas
64 tca_writeReg(0x06, 0xFF); // Configuration Port 0
65 tca_writeReg(0x07, 0xFF); // Configuration Port 1
66
67 // Polaridad normal
68 tca_writeReg(0x04, 0x00); // Polarity Port 0
69 tca_writeReg(0x05, 0x00); // Polarity Port 1

78static uint8_t hctl_readByte(uint8_t port, bool highByte) {

89static uint16_t hctl_readVertical() {
90 hctl_setOE(0);
91
92 uint8_t hi = hctl_readByte(0, true);
93 uint8_t lo = hctl_readByte(0, false);
94
95 return ((uint16_t)hi << 8) | lo;

107static float countsToDegrees(uint16_t count, float countsPerRev) {
108 if (countsPerRev <= 0.0f) return 0.0f;
109 uint16_t modulo = (uint16_t)countsPerRev;
110 return 360.0f * (float(count % modulo) / countsPerRev);
111}

117void Encoders_begin(uint8_t pinSEL,
118 uint8_t pinRST,
119 uint8_t pinOE,
120 const EncoderConfig &config) {
121 s_pinSEL = pinSEL;
122 s_pinRST = pinRST;
123 s_pinOE = pinOE;
124 s_config = config;
125
126 pinMode(s_pinSEL, OUTPUT);
127 pinMode(s_pinRST, OUTPUT);
128 pinMode(s_pinOE, OUTPUT);
129
130 // Estado inicial
131 hctl_setOE(1);
132 hctl_setRST(1);
133 hctl_setSEL(0);
134
135 // Inicializar expansor TCA9539
136 tca_init();
137
138 // Reset de contadores
139 hctl_resetCounters_internal();
140
141 // Activar salidas
142 hctl_setOE(0);
143
144 Serial.println("Encoders: inicializados (HCTL + TCA9539).");
145}

151EncoderAngles Encoders_readAngles() {
