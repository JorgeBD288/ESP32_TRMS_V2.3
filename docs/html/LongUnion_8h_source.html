28#ifndef _WORD_UNION_H
29#define _WORD_UNION_H
30/**
31 * Union to specify parts / manifestations of a 16 bit Word without casts and shifts.
32 * It also supports the compiler generating small code.
33 * Usage: WordUnion tWord;
34 * tWord.UByte.HighByte = 0x12;
35 */
36union WordUnion {
37 struct {
38 uint8_t LowByte;
39 uint8_t HighByte;
40 } UByte;
41 struct {
42 int8_t LowByte;
43 int8_t HighByte;
44 } Byte;
45 uint8_t UBytes[2]; // UBytes[0] is LowByte
46 int8_t Bytes[2];
47 uint16_t UWord;
48 int16_t Word;
49 uint8_t *BytePointer;
50};
51#endif // _WORD_UNION_H

55/**
56 * Union to specify parts / manifestations of a 32 bit Long without casts and shifts.
57 * It also supports the compiler generating small code.
58 */
59union LongUnion {

65 } UByte;

80 uint16_t LowWord;

84 int16_t LowWord;
