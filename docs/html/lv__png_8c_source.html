59/**
60 * Get info about a PNG image
61 * @param src can be file name or pointer to a C array
62 * @param header store the info here
63 * @return LV_RES_OK: no error; LV_RES_INV: can't get the info
64 */
65static lv_res_t decoder_info(struct _lv_img_decoder_t * decoder, const void * src, lv_img_header_t * header)
66{
67 (void) decoder; /*Unused*/
68 lv_img_src_t src_type = lv_img_src_get_type(src); /*Get the source type*/
69
70 /*If it's a PNG file...*/
71 if(src_type == LV_IMG_SRC_FILE) {
72 const char * fn = src;
73 if(strcmp(lv_fs_get_ext(fn), "png") == 0) { /*Check the extension*/
74
75 /* Read the width and height from the file. They have a constant location:
76 * [16..23]: width
77 * [24..27]: height
78 */
79 uint32_t size[2];
80 lv_fs_file_t f;
81 lv_fs_res_t res = lv_fs_open(&f, fn, LV_FS_MODE_RD);
82 if(res != LV_FS_RES_OK) return LV_RES_INV;
83
84 lv_fs_seek(&f, 16, LV_FS_SEEK_SET);
85
86 uint32_t rn;
87 lv_fs_read(&f, &size, 8, &rn);
88 lv_fs_close(&f);
89
90 if(rn != 8) return LV_RES_INV;
91
92 /*Save the data in the header*/
93 header->always_zero = 0;
94 header->cf = LV_IMG_CF_TRUE_COLOR_ALPHA;
95 /*The width and height are stored in Big endian format so convert them to little endian*/
96 header->w = (lv_coord_t)((size[0] & 0xff000000) >> 24) + ((size[0] & 0x00ff0000) >> 8);
97 header->h = (lv_coord_t)((size[1] & 0xff000000) >> 24) + ((size[1] & 0x00ff0000) >> 8);
98
99 return LV_RES_OK;
100 }
101 }
102 /*If it's a PNG file in a C array...*/
103 else if(src_type == LV_IMG_SRC_VARIABLE) {
104 const lv_img_dsc_t * img_dsc = src;
105 const uint32_t data_size = img_dsc->data_size;
106 const uint32_t * size = ((uint32_t *)img_dsc->data) + 4;
107 const uint8_t magic[] = {0x89, 0x50, 0x4e, 0x47, 0x0d, 0x0a, 0x1a, 0x0a};
108 if(data_size < sizeof(magic)) return LV_RES_INV;
109 if(memcmp(magic, img_dsc->data, sizeof(magic))) return LV_RES_INV;
110 header->always_zero = 0;
111
112 if(img_dsc->header.cf) {
113 header->cf = img_dsc->header.cf; /*Save the color format*/
114 }
115 else {
116 header->cf = LV_IMG_CF_TRUE_COLOR_ALPHA;
117 }
118
119 if(img_dsc->header.w) {
120 header->w = img_dsc->header.w; /*Save the image width*/
121 }
122 else {
123 header->w = (lv_coord_t)((size[0] & 0xff000000) >> 24) + ((size[0] & 0x00ff0000) >> 8);
124 }
125
126 if(img_dsc->header.h) {
127 header->h = img_dsc->header.h; /*Save the color height*/
128 }
129 else {
130 header->h = (lv_coord_t)((size[1] & 0xff000000) >> 24) + ((size[1] & 0x00ff0000) >> 8);
131 }
132
133 return LV_RES_OK;
134 }
135
136 return LV_RES_INV; /*If didn't succeeded earlier then it's an error*/
137}

140/**
141 * Open a PNG image and return the decided image
142 * @param src can be file name or pointer to a C array
143 * @param style style of the image object (unused now but certain formats might use it)
144 * @return pointer to the decoded image or `LV_IMG_DECODER_OPEN_FAIL` if failed
145 */
146static lv_res_t decoder_open(lv_img_decoder_t * decoder, lv_img_decoder_dsc_t * dsc)
147{
148
149 (void) decoder; /*Unused*/
150 uint32_t error; /*For the return values of PNG decoder functions*/
151
152 uint8_t * img_data = NULL;
153
154 /*If it's a PNG file...*/
155 if(dsc->src_type == LV_IMG_SRC_FILE) {
156 const char * fn = dsc->src;
157 if(strcmp(lv_fs_get_ext(fn), "png") == 0) { /*Check the extension*/
158
159 /*Load the PNG file into buffer. It's still compressed (not decoded)*/
160 unsigned char * png_data; /*Pointer to the loaded data. Same as the original file just loaded into the RAM*/
161 size_t png_data_size; /*Size of `png_data` in bytes*/
162
163 error = lodepng_load_file(&png_data, &png_data_size, fn); /*Load the file*/
164 if(error) {
165 LV_LOG_WARN("error %" LV_PRIu32 ": %s\n", error, lodepng_error_text(error));
166 return LV_RES_INV;
167 }
168
169 /*Decode the PNG image*/
170 unsigned png_width; /*Will be the width of the decoded image*/
171 unsigned png_height; /*Will be the width of the decoded image*/
172
173 /*Decode the loaded image in ARGB8888 */
174 error = lodepng_decode32(&img_data, &png_width, &png_height, png_data, png_data_size);
175 lv_mem_free(png_data); /*Free the loaded file*/
176 if(error) {
177 if(img_data != NULL) {
178 lv_mem_free(img_data);
179 }
180 LV_LOG_WARN("error %" LV_PRIu32 ": %s\n", error, lodepng_error_text(error));
181 return LV_RES_INV;
182 }
183
184 /*Convert the image to the system's color depth*/
185 convert_color_depth(img_data, png_width * png_height);
186 dsc->img_data = img_data;
187 return LV_RES_OK; /*The image is fully decoded. Return with its pointer*/
188 }
189 }
190 /*If it's a PNG file in a C array...*/
191 else if(dsc->src_type == LV_IMG_SRC_VARIABLE) {
192 const lv_img_dsc_t * img_dsc = dsc->src;
193 unsigned png_width; /*No used, just required by he decoder*/
194 unsigned png_height; /*No used, just required by he decoder*/
195
196 /*Decode the image in ARGB8888 */
197 error = lodepng_decode32(&img_data, &png_width, &png_height, img_dsc->data, img_dsc->data_size);
198
199 if(error) {
200 if(img_data != NULL) {
201 lv_mem_free(img_data);
202 }
203 return LV_RES_INV;
204 }
205
206 /*Convert the image to the system's color depth*/
207 convert_color_depth(img_data, png_width * png_height);
208
209 dsc->img_data = img_data;
210 return LV_RES_OK; /*Return with its pointer*/
211 }
212
213 return LV_RES_INV; /*If not returned earlier then it failed*/
214}

228/**
229 * If the display is not in 32 bit format (ARGB888) then covert the image to the current color depth
230 * @param img the ARGB888 image
231 * @param px_cnt number of pixels in `img`
232 */
233static void convert_color_depth(uint8_t * img, uint32_t px_cnt)
234{
235#if LV_COLOR_DEPTH == 32
236 lv_color32_t * img_argb = (lv_color32_t *)img;
237 lv_color_t c;
238 lv_color_t * img_c = (lv_color_t *) img;
239 uint32_t i;
240 for(i = 0; i < px_cnt; i++) {
241 c = lv_color_make(img_argb[i].ch.red, img_argb[i].ch.green, img_argb[i].ch.blue);
242 img_c[i].ch.red = c.ch.blue;
243 img_c[i].ch.blue = c.ch.red;
244 }
245#elif LV_COLOR_DEPTH == 16
246 lv_color32_t * img_argb = (lv_color32_t *)img;
247 lv_color_t c;
248 uint32_t i;
249 for(i = 0; i < px_cnt; i++) {
250 c = lv_color_make(img_argb[i].ch.blue, img_argb[i].ch.green, img_argb[i].ch.red);
251 img[i * 3 + 2] = img_argb[i].ch.alpha;
252 img[i * 3 + 1] = c.full >> 8;
253 img[i * 3 + 0] = c.full & 0xFF;
254 }
255#elif LV_COLOR_DEPTH == 8
256 lv_color32_t * img_argb = (lv_color32_t *)img;
257 lv_color_t c;
258 uint32_t i;
259 for(i = 0; i < px_cnt; i++) {
260 c = lv_color_make(img_argb[i].ch.red, img_argb[i].ch.green, img_argb[i].ch.blue);
261 img[i * 2 + 1] = img_argb[i].ch.alpha;
262 img[i * 2 + 0] = c.full;
263 }
264#elif LV_COLOR_DEPTH == 1
265 lv_color32_t * img_argb = (lv_color32_t *)img;
266 uint8_t b;
267 uint32_t i;
268 for(i = 0; i < px_cnt; i++) {
269 b = img_argb[i].ch.red | img_argb[i].ch.green | img_argb[i].ch.blue;
270 img[i * 2 + 1] = img_argb[i].ch.alpha;
271 img[i * 2 + 0] = b > 128 ? 1 : 0;
272 }
273#endif
274}
