86/**
87 * Draw a letter in the Virtual Display Buffer
88 * @param pos_p left-top coordinate of the latter
89 * @param mask_p the letter will be drawn only on this area (truncated to draw_buf area)
90 * @param font_p pointer to font
91 * @param letter a letter to draw
92 * @param color color of letter
93 * @param opa opacity of letter (0..255)
94 */
95void lv_draw_sw_letter(lv_draw_ctx_t * draw_ctx, const lv_draw_label_dsc_t * dsc, const lv_point_t * pos_p,
96 uint32_t letter)
97{
98 lv_font_glyph_dsc_t g;
99 bool g_ret = lv_font_get_glyph_dsc(dsc->font, &g, letter, '\0');
100 if(g_ret == false) {
101 /*Add warning if the dsc is not found
102 *but do not print warning for non printable ASCII chars (e.g. '\n')*/
103 if(letter >= 0x20 &&
104 letter != 0xf8ff && /*LV_SYMBOL_DUMMY*/
105 letter != 0x200c) { /*ZERO WIDTH NON-JOINER*/
106 LV_LOG_WARN("lv_draw_letter: glyph dsc. not found for U+%" LV_PRIX32, letter);
107
108#if LV_USE_FONT_PLACEHOLDER
109 /* draw placeholder */
110 lv_area_t glyph_coords;
111 lv_draw_rect_dsc_t glyph_dsc;
112 lv_coord_t begin_x = pos_p->x + g.ofs_x;
113 lv_coord_t begin_y = pos_p->y + g.ofs_y;
114 lv_area_set(&glyph_coords, begin_x, begin_y, begin_x + g.box_w, begin_y + g.box_h);
115 lv_draw_rect_dsc_init(&glyph_dsc);
116 glyph_dsc.bg_opa = LV_OPA_MIN;
117 glyph_dsc.outline_opa = LV_OPA_MIN;
118 glyph_dsc.shadow_opa = LV_OPA_MIN;
119 glyph_dsc.bg_img_opa = LV_OPA_MIN;
120 glyph_dsc.border_color = dsc->color;
121 glyph_dsc.border_width = 1;
122 draw_ctx->draw_rect(draw_ctx, &glyph_dsc, &glyph_coords);
123#endif
124 }
125 return;
126 }
127
128 /*Don't draw anything if the character is empty. E.g. space*/
129 if((g.box_h == 0) || (g.box_w == 0)) return;
130
131 lv_point_t gpos;
132 gpos.x = pos_p->x + g.ofs_x;
133 gpos.y = pos_p->y + (dsc->font->line_height - dsc->font->base_line) - g.box_h - g.ofs_y;
134
135 /*If the letter is completely out of mask don't draw it*/
136 if(gpos.x + g.box_w < draw_ctx->clip_area->x1 ||
137 gpos.x > draw_ctx->clip_area->x2 ||
138 gpos.y + g.box_h < draw_ctx->clip_area->y1 ||
139 gpos.y > draw_ctx->clip_area->y2) {
140 return;
141 }
142
143 const uint8_t * map_p = lv_font_get_glyph_bitmap(g.resolved_font, letter);
144 if(map_p == NULL) {
145 LV_LOG_WARN("lv_draw_letter: character's bitmap not found");
146 return;
147 }
148
149 if(g.resolved_font->subpx) {
150#if LV_DRAW_COMPLEX && LV_USE_FONT_SUBPX
151 draw_letter_subpx(draw_ctx, dsc, &gpos, &g, map_p);
152#else
153 LV_LOG_WARN("Can't draw sub-pixel rendered letter because LV_USE_FONT_SUBPX == 0 in lv_conf.h");
154#endif
155 }
156 else {
157 draw_letter_normal(draw_ctx, dsc, &gpos, &g, map_p);
158 }
159}

165static void LV_ATTRIBUTE_FAST_MEM draw_letter_normal(lv_draw_ctx_t * draw_ctx, const lv_draw_label_dsc_t * dsc,
166 const lv_point_t * pos, lv_font_glyph_dsc_t * g, const uint8_t * map_p)
167{
168
169 const uint8_t * bpp_opa_table_p;
170 uint32_t bitmask_init;
171 uint32_t bitmask;
172 uint32_t bpp = g->bpp;
173 lv_opa_t opa = dsc->opa;
174 uint32_t shades;
175 if(bpp == 3) bpp = 4;
176
177#if LV_USE_IMGFONT
178 if(bpp == LV_IMGFONT_BPP) { //is imgfont
179 lv_area_t fill_area;
180 fill_area.x1 = pos->x;
181 fill_area.y1 = pos->y;
182 fill_area.x2 = pos->x + g->box_w - 1;
183 fill_area.y2 = pos->y + g->box_h - 1;
184 lv_draw_img_dsc_t img_dsc;
185 lv_draw_img_dsc_init(&img_dsc);
186 img_dsc.angle = 0;
187 img_dsc.zoom = LV_IMG_ZOOM_NONE;
188 img_dsc.opa = dsc->opa;
189 img_dsc.blend_mode = dsc->blend_mode;
190 lv_draw_img(draw_ctx, &img_dsc, &fill_area, map_p);
191 return;
192 }
193#endif
194
195 switch(bpp) {
196 case 1:
197 bpp_opa_table_p = _lv_bpp1_opa_table;
198 bitmask_init = 0x80;
199 shades = 2;
200 break;
201 case 2:
202 bpp_opa_table_p = _lv_bpp2_opa_table;
203 bitmask_init = 0xC0;
204 shades = 4;
205 break;
206 case 4:
207 bpp_opa_table_p = _lv_bpp4_opa_table;
208 bitmask_init = 0xF0;
209 shades = 16;
210 break;
211 case 8:
212 bpp_opa_table_p = _lv_bpp8_opa_table;
213 bitmask_init = 0xFF;
214 shades = 256;
215 break; /*No opa table, pixel value will be used directly*/
216 default:
217 LV_LOG_WARN("lv_draw_letter: invalid bpp");
218 return; /*Invalid bpp. Can't render the letter*/
219 }
220
221 static lv_opa_t opa_table[256];
222 static lv_opa_t prev_opa = LV_OPA_TRANSP;
223 static uint32_t prev_bpp = 0;
224 if(opa < LV_OPA_MAX) {
225 if(prev_opa != opa || prev_bpp != bpp) {
226 uint32_t i;
227 for(i = 0; i < shades; i++) {
228 opa_table[i] = bpp_opa_table_p[i] == LV_OPA_COVER ? opa : ((bpp_opa_table_p[i] * opa) >> 8);
229 }
230 }
231 bpp_opa_table_p = opa_table;
232 prev_opa = opa;
233 prev_bpp = bpp;
234 }
235
236 int32_t col, row;
237 int32_t box_w = g->box_w;
238 int32_t box_h = g->box_h;
239 int32_t width_bit = box_w * bpp; /*Letter width in bits*/
240
241 /*Calculate the col/row start/end on the map*/
242 int32_t col_start = pos->x >= draw_ctx->clip_area->x1 ? 0 : draw_ctx->clip_area->x1 - pos->x;
243 int32_t col_end = pos->x + box_w <= draw_ctx->clip_area->x2 ? box_w : draw_ctx->clip_area->x2 - pos->x + 1;
244 int32_t row_start = pos->y >= draw_ctx->clip_area->y1 ? 0 : draw_ctx->clip_area->y1 - pos->y;
245 int32_t row_end = pos->y + box_h <= draw_ctx->clip_area->y2 ? box_h : draw_ctx->clip_area->y2 - pos->y + 1;
246
247 /*Move on the map too*/
248 uint32_t bit_ofs = (row_start * width_bit) + (col_start * bpp);
249 map_p += bit_ofs >> 3;
250
251 uint8_t letter_px;
252 uint32_t col_bit;
253 col_bit = bit_ofs & 0x7; /*"& 0x7" equals to "% 8" just faster*/
254
255 lv_draw_sw_blend_dsc_t blend_dsc;
256 lv_memset_00(&blend_dsc, sizeof(blend_dsc));
257 blend_dsc.color = dsc->color;
258 blend_dsc.opa = dsc->opa;
259 blend_dsc.blend_mode = dsc->blend_mode;
260
261 lv_coord_t hor_res = lv_disp_get_hor_res(_lv_refr_get_disp_refreshing());
262 uint32_t mask_buf_size = box_w * box_h > hor_res ? hor_res : box_w * box_h;
263 lv_opa_t * mask_buf = lv_mem_buf_get(mask_buf_size);
264 blend_dsc.mask_buf = mask_buf;
265 int32_t mask_p = 0;
266
267 lv_area_t fill_area;
268 fill_area.x1 = col_start + pos->x;
269 fill_area.x2 = col_end + pos->x - 1;
270 fill_area.y1 = row_start + pos->y;
271 fill_area.y2 = fill_area.y1;
272#if LV_DRAW_COMPLEX
273 lv_coord_t fill_w = lv_area_get_width(&fill_area);
274 lv_area_t mask_area;
275 lv_area_copy(&mask_area, &fill_area);
276 mask_area.y2 = mask_area.y1 + row_end;
277 bool mask_any = lv_draw_mask_is_any(&mask_area);
278#endif
279 blend_dsc.blend_area = &fill_area;
280 blend_dsc.mask_area = &fill_area;
281
282 uint32_t col_bit_max = 8 - bpp;
283 uint32_t col_bit_row_ofs = (box_w + col_start - col_end) * bpp;
284
285 for(row = row_start ; row < row_end; row++) {
286#if LV_DRAW_COMPLEX
287 int32_t mask_p_start = mask_p;
288#endif
289 bitmask = bitmask_init >> col_bit;
290 for(col = col_start; col < col_end; col++) {
291 /*Load the pixel's opacity into the mask*/
292 letter_px = (*map_p & bitmask) >> (col_bit_max - col_bit);
293 if(letter_px) {
294 mask_buf[mask_p] = bpp_opa_table_p[letter_px];
295 }
296 else {
297 mask_buf[mask_p] = 0;
298 }
299
300 /*Go to the next column*/
301 if(col_bit < col_bit_max) {
302 col_bit += bpp;
303 bitmask = bitmask >> bpp;
304 }
305 else {
306 col_bit = 0;
307 bitmask = bitmask_init;
308 map_p++;
309 }
310
311 /*Next mask byte*/
312 mask_p++;
313 }
314
315#if LV_DRAW_COMPLEX
316 /*Apply masks if any*/
317 if(mask_any) {
318 blend_dsc.mask_res = lv_draw_mask_apply(mask_buf + mask_p_start, fill_area.x1, fill_area.y2,
319 fill_w);
320 if(blend_dsc.mask_res == LV_DRAW_MASK_RES_TRANSP) {
321 lv_memset_00(mask_buf + mask_p_start, fill_w);
322 }
323 }
324#endif
325
326 if((uint32_t) mask_p + (col_end - col_start) < mask_buf_size) {
327 fill_area.y2 ++;
328 }
329 else {
330 blend_dsc.mask_res = LV_DRAW_MASK_RES_CHANGED;
331 lv_draw_sw_blend(draw_ctx, &blend_dsc);
332
333 fill_area.y1 = fill_area.y2 + 1;
334 fill_area.y2 = fill_area.y1;
335 mask_p = 0;
336 }
337
338 col_bit += col_bit_row_ofs;
339 map_p += (col_bit >> 3);
340 col_bit = col_bit & 0x7;
341 }
342
343 /*Flush the last part*/
344 if(fill_area.y1 != fill_area.y2) {
345 fill_area.y2--;
346 blend_dsc.mask_res = LV_DRAW_MASK_RES_CHANGED;
347 lv_draw_sw_blend(draw_ctx, &blend_dsc);
348 mask_p = 0;
349 }
350
351 lv_mem_buf_release(mask_buf);
352}

354#if LV_DRAW_COMPLEX && LV_USE_FONT_SUBPX
355static void draw_letter_subpx(lv_draw_ctx_t * draw_ctx, const lv_draw_label_dsc_t * dsc, const lv_point_t * pos,
356 lv_font_glyph_dsc_t * g, const uint8_t * map_p)
357{
358 const uint8_t * bpp_opa_table;
359 uint32_t bitmask_init;
360 uint32_t bitmask;
361 uint32_t bpp = g->bpp;
362 lv_opa_t opa = dsc->opa;
363 if(bpp == 3) bpp = 4;
364
365 switch(bpp) {
366 case 1:
367 bpp_opa_table = _lv_bpp1_opa_table;
368 bitmask_init = 0x80;
369 break;
370 case 2:
371 bpp_opa_table = _lv_bpp2_opa_table;
372 bitmask_init = 0xC0;
373 break;
374 case 4:
375 bpp_opa_table = _lv_bpp4_opa_table;
376 bitmask_init = 0xF0;
377 break;
378 case 8:
379 bpp_opa_table = _lv_bpp8_opa_table;
380 bitmask_init = 0xFF;
381 break; /*No opa table, pixel value will be used directly*/
382 default:
383 LV_LOG_WARN("lv_draw_letter: invalid bpp not found");
384 return; /*Invalid bpp. Can't render the letter*/
385 }
386
387 int32_t col, row;
388
389 int32_t box_w = g->box_w;
390 int32_t box_h = g->box_h;
391 int32_t width_bit = box_w * bpp; /*Letter width in bits*/
392
393 /*Calculate the col/row start/end on the map*/
394 int32_t col_start = pos->x >= draw_ctx->clip_area->x1 ? 0 : (draw_ctx->clip_area->x1 - pos->x) * 3;
395 int32_t col_end = pos->x + box_w / 3 <= draw_ctx->clip_area->x2 ? box_w : (draw_ctx->clip_area->x2 - pos->x + 1) * 3;
396 int32_t row_start = pos->y >= draw_ctx->clip_area->y1 ? 0 : draw_ctx->clip_area->y1 - pos->y;
397 int32_t row_end = pos->y + box_h <= draw_ctx->clip_area->y2 ? box_h : draw_ctx->clip_area->y2 - pos->y + 1;
398
399 /*Move on the map too*/
400 int32_t bit_ofs = (row_start * width_bit) + (col_start * bpp);
401 map_p += bit_ofs >> 3;
402
403 uint8_t letter_px;
404 lv_opa_t px_opa;
405 int32_t col_bit;
406 col_bit = bit_ofs & 0x7; /*"& 0x7" equals to "% 8" just faster*/
407
408 lv_area_t map_area;
409 map_area.x1 = col_start / 3 + pos->x;
410 map_area.x2 = col_end / 3 + pos->x - 1;
411 map_area.y1 = row_start + pos->y;
412 map_area.y2 = map_area.y1;
413
414 if(map_area.x2 <= map_area.x1) return;
415
416 lv_coord_t hor_res = lv_disp_get_hor_res(_lv_refr_get_disp_refreshing());
417 int32_t mask_buf_size = box_w * box_h > hor_res ? hor_res : g->box_w * g->box_h;
418 lv_opa_t * mask_buf = lv_mem_buf_get(mask_buf_size);
419 int32_t mask_p = 0;
420
421 lv_color_t * color_buf = lv_mem_buf_get(mask_buf_size * sizeof(lv_color_t));
422
423 int32_t dest_buf_stride = lv_area_get_width(draw_ctx->buf_area);
424 lv_color_t * dest_buf_tmp = draw_ctx->buf;
425
426 /*Set a pointer on draw_buf to the first pixel of the letter*/
427 dest_buf_tmp += ((pos->y - draw_ctx->buf_area->y1) * dest_buf_stride) + pos->x - draw_ctx->buf_area->x1;
428
429 /*If the letter is partially out of mask the move there on draw_buf*/
430 dest_buf_tmp += (row_start * dest_buf_stride) + col_start / 3;
431
432 lv_area_t mask_area;
433 lv_area_copy(&mask_area, &map_area);
434 mask_area.y2 = mask_area.y1 + row_end;
435 bool mask_any = lv_draw_mask_is_any(&map_area);
436 uint8_t font_rgb[3];
437
438 lv_color_t color = dsc->color;
439#if LV_COLOR_16_SWAP == 0
440 uint8_t txt_rgb[3] = {color.ch.red, color.ch.green, color.ch.blue};
441#else
442 uint8_t txt_rgb[3] = {color.ch.red, (color.ch.green_h << 3) + color.ch.green_l, color.ch.blue};
443#endif
444
445 lv_draw_sw_blend_dsc_t blend_dsc;
446 lv_memset_00(&blend_dsc, sizeof(blend_dsc));
447 blend_dsc.blend_area = &map_area;
448 blend_dsc.mask_area = &map_area;
449 blend_dsc.src_buf = color_buf;
450 blend_dsc.mask_buf = mask_buf;
451 blend_dsc.opa = opa;
452 blend_dsc.blend_mode = dsc->blend_mode;
453
454 for(row = row_start ; row < row_end; row++) {
455 uint32_t subpx_cnt = 0;
456 bitmask = bitmask_init >> col_bit;
457 int32_t mask_p_start = mask_p;
458
459 for(col = col_start; col < col_end; col++) {
460 /*Load the pixel's opacity into the mask*/
461 letter_px = (*map_p & bitmask) >> (8 - col_bit - bpp);
462 if(letter_px != 0) {
463 if(opa >= LV_OPA_MAX) {
464 px_opa = bpp == 8 ? letter_px : bpp_opa_table[letter_px];
465 }
466 else {
467 px_opa = bpp == 8 ? (uint32_t)((uint32_t)letter_px * opa) >> 8
468 : (uint32_t)((uint32_t)bpp_opa_table[letter_px] * opa) >> 8;
469 }
470 }
471 else {
472 px_opa = 0;
473 }
474
475 font_rgb[subpx_cnt] = px_opa;
476
477 subpx_cnt ++;
478 if(subpx_cnt == 3) {
479 subpx_cnt = 0;
480
481 lv_color_t res_color;
482#if LV_COLOR_16_SWAP == 0
483 uint8_t bg_rgb[3] = {dest_buf_tmp->ch.red, dest_buf_tmp->ch.green, dest_buf_tmp->ch.blue};
484#else
485 uint8_t bg_rgb[3] = {dest_buf_tmp->ch.red,
486 (dest_buf_tmp->ch.green_h << 3) + dest_buf_tmp->ch.green_l,
487 dest_buf_tmp->ch.blue
488 };
489#endif
490
491#if LV_FONT_SUBPX_BGR
492 res_color.ch.red = (uint32_t)((uint16_t)txt_rgb[0] * font_rgb[2] + (bg_rgb[0] * (255 - font_rgb[2]))) >> 8;
493 res_color.ch.blue = (uint32_t)((uint16_t)txt_rgb[2] * font_rgb[0] + (bg_rgb[2] * (255 - font_rgb[0]))) >> 8;
494#else
495 res_color.ch.red = (uint32_t)((uint16_t)txt_rgb[0] * font_rgb[0] + (bg_rgb[0] * (255 - font_rgb[0]))) >> 8;
496 res_color.ch.blue = (uint32_t)((uint16_t)txt_rgb[2] * font_rgb[2] + (bg_rgb[2] * (255 - font_rgb[2]))) >> 8;
497#endif
498
499#if LV_COLOR_16_SWAP == 0
500 res_color.ch.green = (uint32_t)((uint32_t)txt_rgb[1] * font_rgb[1] + (bg_rgb[1] * (255 - font_rgb[1]))) >> 8;
501#else
502 uint8_t green = (uint32_t)((uint32_t)txt_rgb[1] * font_rgb[1] + (bg_rgb[1] * (255 - font_rgb[1]))) >> 8;
503 res_color.ch.green_h = green >> 3;
504 res_color.ch.green_l = green & 0x7;
505#endif
506
507#if LV_COLOR_DEPTH == 32
508 res_color.ch.alpha = 0xff;
509#endif
510
511 if(font_rgb[0] == 0 && font_rgb[1] == 0 && font_rgb[2] == 0) mask_buf[mask_p] = LV_OPA_TRANSP;
512 else mask_buf[mask_p] = LV_OPA_COVER;
513 color_buf[mask_p] = res_color;
514
515 /*Next mask byte*/
516 mask_p++;
517 dest_buf_tmp++;
518 }
519
520 /*Go to the next column*/
521 if(col_bit < (int32_t)(8 - bpp)) {
522 col_bit += bpp;
523 bitmask = bitmask >> bpp;
524 }
525 else {
526 col_bit = 0;
527 bitmask = bitmask_init;
528 map_p++;
529 }
530 }
531
532 /*Apply masks if any*/
533 if(mask_any) {
534 blend_dsc.mask_res = lv_draw_mask_apply(mask_buf + mask_p_start, map_area.x1, map_area.y2,
535 lv_area_get_width(&map_area));
536 if(blend_dsc.mask_res == LV_DRAW_MASK_RES_TRANSP) {
537 lv_memset_00(mask_buf + mask_p_start, lv_area_get_width(&map_area));
538 }
539 }
540
541 if((int32_t) mask_p + (col_end - col_start) < mask_buf_size) {
542 map_area.y2 ++;
543 }
544 else {
545 blend_dsc.mask_res = LV_DRAW_MASK_RES_CHANGED;
546 lv_draw_sw_blend(draw_ctx, &blend_dsc);
547
548 map_area.y1 = map_area.y2 + 1;
549 map_area.y2 = map_area.y1;
550 mask_p = 0;
551 }
552
553 col_bit += ((box_w - col_end) + col_start) * bpp;
554
555 map_p += (col_bit >> 3);
556 col_bit = col_bit & 0x7;
557
558 /*Next row in draw_buf*/
559 dest_buf_tmp += dest_buf_stride - (col_end - col_start) / 3;
560 }
561
562 /*Flush the last part*/
563 if(map_area.y1 != map_area.y2) {
564 map_area.y2--;
565 blend_dsc.mask_res = LV_DRAW_MASK_RES_CHANGED;
566 lv_draw_sw_blend(draw_ctx, &blend_dsc);
567 }
568
569 lv_mem_buf_release(mask_buf);
570 lv_mem_buf_release(color_buf);
571}
