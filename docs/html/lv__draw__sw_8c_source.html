81void lv_draw_sw_buffer_copy(lv_draw_ctx_t * draw_ctx,
82 void * dest_buf, lv_coord_t dest_stride, const lv_area_t * dest_area,
83 void * src_buf, lv_coord_t src_stride, const lv_area_t * src_area)
84{
85 LV_UNUSED(draw_ctx);
86
87 lv_color_t * dest_bufc = dest_buf;
88 lv_color_t * src_bufc = src_buf;
89
90 /*Got the first pixel of each buffer*/
91 dest_bufc += dest_stride * dest_area->y1;
92 dest_bufc += dest_area->x1;
93
94 src_bufc += src_stride * src_area->y1;
95 src_bufc += src_area->x1;
96
97 uint32_t line_length = lv_area_get_width(dest_area) * sizeof(lv_color_t);
98 lv_coord_t y;
99 for(y = dest_area->y1; y <= dest_area->y2; y++) {
100 lv_memcpy(dest_bufc, src_bufc, line_length);
101 dest_bufc += dest_stride;
102 src_bufc += src_stride;
103 }
104}
