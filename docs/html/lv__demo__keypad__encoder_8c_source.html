47void lv_demo_keypad_encoder(void)
48{
49 g = lv_group_get_default();
50 if(g == NULL) {
51 g = lv_group_create();
52 lv_group_set_default(g);
53 }
54
55 lv_indev_t * cur_drv = NULL;
56 for(;;) {
57 cur_drv = lv_indev_get_next(cur_drv);
58 if(!cur_drv) {
59 break;
60 }
61
62 if(cur_drv->driver->type == LV_INDEV_TYPE_KEYPAD) {
63 lv_indev_set_group(cur_drv, g);
64 }
65
66 if(cur_drv->driver->type == LV_INDEV_TYPE_ENCODER) {
67 lv_indev_set_group(cur_drv, g);
68 }
69 }
70
71 tv = lv_tabview_create(lv_scr_act(), LV_DIR_TOP, LV_DPI_DEF / 3);
72
73 t1 = lv_tabview_add_tab(tv, "Selectors");
74 t2 = lv_tabview_add_tab(tv, "Text input");
75
76 selectors_create(t1);
77 text_input_create(t2);
78
79 msgbox_create();
80}

187static void msgbox_event_cb(lv_event_t * e)
188{
189 lv_event_code_t code = lv_event_get_code(e);
190 lv_obj_t * msgbox = lv_event_get_current_target(e);
191
192 if(code == LV_EVENT_VALUE_CHANGED) {
193 const char * txt = lv_msgbox_get_active_btn_text(msgbox);
194 if(txt) {
195 lv_msgbox_close(msgbox);
196 lv_group_focus_freeze(g, false);
197 lv_group_focus_obj(lv_obj_get_child(t1, 0));
198 lv_obj_scroll_to(t1, 0, 0, LV_ANIM_OFF);
199
200 }
201 }

204static void ta_event_cb(lv_event_t * e)
205{
206 lv_indev_t * indev = lv_indev_get_act();
207 if(indev == NULL) return;
208 lv_indev_type_t indev_type = lv_indev_get_type(indev);
209
210 lv_event_code_t code = lv_event_get_code(e);
211 lv_obj_t * ta = lv_event_get_target(e);
212 lv_obj_t * kb = lv_event_get_user_data(e);
213
214 if(code == LV_EVENT_CLICKED && indev_type == LV_INDEV_TYPE_ENCODER) {
215 lv_keyboard_set_textarea(kb, ta);
216 lv_obj_clear_flag(kb, LV_OBJ_FLAG_HIDDEN);
217 lv_group_focus_obj(kb);
218 lv_group_set_editing(lv_obj_get_group(kb), kb);
219 lv_obj_set_height(tv, LV_VER_RES / 2);
220 lv_obj_align(kb, LV_ALIGN_BOTTOM_MID, 0, 0);
221 }
222
223 if(code == LV_EVENT_READY || code == LV_EVENT_CANCEL) {
224 lv_obj_add_flag(kb, LV_OBJ_FLAG_HIDDEN);
225 lv_obj_set_height(tv, LV_VER_RES);
226 }
227}
