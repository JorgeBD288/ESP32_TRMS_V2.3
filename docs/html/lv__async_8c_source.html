46{
47 /*Allocate an info structure*/
48 lv_async_info_t * info = lv_mem_alloc(sizeof(lv_async_info_t));
49
50 if(info == NULL)
51 return LV_RES_INV;
52
53 /*Create a new timer*/

55
56 if(timer == NULL) {
57 lv_mem_free(info);
58 return LV_RES_INV;
59 }
60

68lv_res_t lv_async_call_cancel(lv_async_cb_t async_xcb, void * user_data)
69{
70 lv_timer_t * timer = lv_timer_get_next(NULL);
71 lv_res_t res = LV_RES_INV;
72
73 while(timer != NULL) {
74 /*Find the next timer node*/
75 lv_timer_t * timer_next = lv_timer_get_next(timer);
76
77 /*Find async timer callback*/
78 if(timer->timer_cb == lv_async_timer_cb) {
79 lv_async_info_t * info = (lv_async_info_t *)timer->user_data;
80
81 /*Match user function callback and user data*/
82 if(info->cb == async_xcb && info->user_data == user_data) {
83 lv_timer_del(timer);
84 lv_mem_free(info);
85 res = LV_RES_OK;
86 }
87 }
88
89 timer = timer_next;
90 }
