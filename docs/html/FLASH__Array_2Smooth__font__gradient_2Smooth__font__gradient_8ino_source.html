1/*
2 This sketch uses Smooth fonts stored in FLASH program memory. It uses a method
3 for rendering anti-aliased fonts on a graded background. This is achieved by
4 telling the TFT_eSPI library the pixel color at each point on the screen. In
5 this sketch a background colour gradient is drawn, the color of each pixel can
6 therefore be determined by a function. The TFT does not need to support reading
7 of the graphics memory. The sketch could be adapted so any part of the screen
8 has a color gradient.
9
10 The TFT_eSPI library must be given the name of the function in the sketch that
11 will return the pixel color at a position x,y on the TFT. In this sketch that
12 function is called gradientColor, so this line is included in setup():
13
14 tft.setCallback(gradientColor);
15
16 TFT_eSPI will call this function during the rendering of the anti-aliased
17 font to blend the edges of each character with the returned color.
18
19 If the TFT supports reading the screen RAM then the returned value can be
20 tft.readPixel(x,y) and anti-aliased text can the be drawn over any screen
21 image. See "Smooth_font_over_image" example.
22*/
23// The fonts are stored in arrays within the sketch tabs.
24
25// A processing sketch to create new fonts can be found in the Tools folder of TFT_eSPI
26// https://github.com/Bodmer/TFT_eSPI/tree/master/Tools/Create_Smooth_Font/Create_font
27
28// This sketch uses fonts created from the Noto family of fonts:
29// https://www.google.com/get/noto/
30
31#include "NotoSansBold15.h"
32#include "NotoSansBold36.h"
33
34// Do not include "" around the array name!
35#define AA_FONT_SMALL NotoSansBold15
36#define AA_FONT_LARGE NotoSansBold36
37
38#include <SPI.h>
39#include <TFT_eSPI.h> // Hardware-specific library
40
41TFT_eSPI tft = TFT_eSPI();
42
43#define TOP_COLOR TFT_RED
44#define BOTTOM_COLOR TFT_BLACK
45
46#define GRADIENT_HEIGHT (9 + tft.fontHeight() * 5) // Gradient over 5 lines
47#define OUTSIDE_GRADIENT TFT_BLUE
48
49uint16_t gradientColor(uint16_t x, uint16_t y)
50{
51 if (y > GRADIENT_HEIGHT) return OUTSIDE_GRADIENT; // Outside gradient area
52 uint8_t alpha = (255 * y) / GRADIENT_HEIGHT; // alpha is a value in the range 0-255
53 return tft.alphaBlend(alpha, BOTTOM_COLOR, TOP_COLOR);
54}
55
56void fillGradient() {
57 uint16_t w = tft.width();
58 for (uint16_t y = 0; y < tft.height(); ++y) {
59 uint16_t color = gradientColor(0, y); // x not used here
60 tft.drawFastHLine(0, y, w, color);
61 }
62}
63
64void setup(void) {
65 Serial.begin(115200);
66
67 tft.begin();
68
69 tft.setCallback(gradientColor); // Switch on color callback for anti-aliased fonts
70 //tft.setCallback(nullptr); // Switch off callback (off by default)
71
72 tft.setRotation(1);
73}
74
75void loop() {
76
77 // Select a font size commensurate with screen size
78 if (tft.width()>= 320)
79 tft.loadFont(AA_FONT_LARGE);
80 else
81 tft.loadFont(AA_FONT_SMALL);
82
83 fillGradient(); // Put here after selecting the font so fontHeight() is already set
84
85 tft.setTextColor(TFT_WHITE); // Background color is ignored in gradient area
86 tft.setCursor(0, 10); // Set cursor at top left of screen
87
88 uint32_t t = millis();
89 tft.println(" Ode to a small\n lump of green\n putty I found\n in my armpit\n one midsummer\n morning ");
90 Serial.println(t = millis()-t);
91
92 tft.unloadFont(); // Remove the font to recover memory used
93
94 delay(2000);
95}
