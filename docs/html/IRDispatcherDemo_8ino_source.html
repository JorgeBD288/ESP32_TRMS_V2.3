109void setup() {
110 pinMode(LED_BUILTIN, OUTPUT);
111 Serial.begin(115200);
112 while (!Serial)
113 ; // Wait for Serial to become available. Is optimized away for some cores.
114
115#if defined(__AVR_ATmega32U4__) || defined(SERIAL_PORT_USBVIRTUAL) || defined(SERIAL_USB) /*stm32duino*/|| defined(USBCON) /*STM32_stm32*/ \
116 || defined(SERIALUSB_PID) || defined(ARDUINO_ARCH_RP2040) || defined(ARDUINO_attiny3217)
117 delay(4000); // To be able to connect Serial monitor after reset or power up and before first print out. Do not wait for an attached Serial Monitor!
118#endif
119#if defined(ESP8266)
120 Serial.println(); // to separate it from the internal boot output
121#endif
122
123 // Just to know which program is running on my Arduino
124#if defined(USE_TINY_IR_RECEIVER)
125 Serial.println(F("START " __FILE__ " from " __DATE__ "\r\nUsing TinyIRReceiver"));
126#elif defined(USE_IRREMOTE_LIBRARY)
127 Serial.println(F("START " __FILE__ " from " __DATE__ "\r\nUsing IRremote library version " VERSION_IRREMOTE));
128#elif defined(USE_IRMP_LIBRARY)
129 Serial.println(F("START " __FILE__ " from " __DATE__ "\r\nUsing IRMP library version " VERSION_IRMP));
130#endif
131
132#if !defined(ESP8266) && !defined(NRF5)
133 // play feedback tone before setup, since it kills the IR timer settings
134 tone(TONE_PIN, 1000, 50);
135 delay(50);
136#endif
137
138 IRDispatcher.init(); // This just calls irmp_init()
139#if defined(USE_TINY_IR_RECEIVER)
140 Serial.println(F("Ready to receive NEC IR signals at pin " STR(IR_RECEIVE_PIN)));
141#else
142 irmp_register_complete_callback_function(&handleReceivedIRData); // fixed function in IRCommandDispatcher.hpp
143
144 Serial.print(F("Ready to receive IR signals of protocols: "));
145 irmp_print_active_protocols(&Serial);
146 Serial.println(F("at pin " STR(IRMP_INPUT_PIN)));
147
148# if defined(ALTERNATIVE_IR_FEEDBACK_LED_PIN)
149 irmp_irsnd_LEDFeedback(true); // Enable receive signal feedback at ALTERNATIVE_IR_FEEDBACK_LED_PIN
150 Serial.println(F("IR feedback pin is " STR(ALTERNATIVE_IR_FEEDBACK_LED_PIN)));
151# endif
152#endif
153
154 Serial.print(F("Listening to commands of IR remote of type "));
155 Serial.println(IR_REMOTE_NAME);
156 doPrintMenu();
157}

159void loop() {
160
161 IRDispatcher.checkAndRunSuspendedBlockingCommands();
162
163 if (doBlink) {
164 digitalWrite(LED_BUILTIN, HIGH);
165 DELAY_AND_RETURN_IF_STOP(sBlinkDelay);
166 digitalWrite(LED_BUILTIN, LOW);
167 DELAY_AND_RETURN_IF_STOP(sBlinkDelay);
168 }
169
170 if (millis() - IRDispatcher.IRReceivedData.MillisOfLastCode > 120000) {
171 //Short beep as remainder, if we did not receive any command in the last 2 minutes
172 IRDispatcher.IRReceivedData.MillisOfLastCode += 120000;
173 doTone1800();
174 }
175
176// delay(10);
177}

205void doIncreaseBlinkFrequency() {
206 doBlink = true;
207 if (sBlinkDelay > 5) {
208 sBlinkDelay -= sBlinkDelay / 4;
209 }
210}
211void doDecreaseBlinkFrequency() {
212 doBlink = true;
213 sBlinkDelay += sBlinkDelay / 4;
214}
215void doStop() {
216 doBlink = false;
217}
218void doResetBlinkFrequency() {
219 sBlinkDelay = 200;
220 digitalWrite(LED_BUILTIN, LOW);
221}
222void doLedBlinkStart() {
223 doBlink = true;
224}
225/*
226 * This is a blocking function and checks periodically for stop
227 */
228void doLedBlink20times() {
229 for (int i = 0; i < 20; ++i) {
230 digitalWrite(LED_BUILTIN, HIGH);
231 DELAY_AND_RETURN_IF_STOP(200);
232 digitalWrite(LED_BUILTIN, LOW);
233 DELAY_AND_RETURN_IF_STOP(200);
234 }
235}

259/*
260 * Convenience IRMP compatible wrapper function for Arduino tone() if IRMP_ENABLE_PIN_CHANGE_INTERRUPT is NOT activated
261 * It currently disables the receiving of repeats
262 */
263void irmp_tone(uint8_t _pin, unsigned int frequency, unsigned long duration) {
264# if defined(__AVR__) && !defined(IRMP_ENABLE_PIN_CHANGE_INTERRUPT)
265 storeIRTimer();
266 tone(_pin, frequency, 0);
267 if (duration == 0) {
268 duration = 100;
269 }
270 delay(duration);
271 noTone(_pin);
272 restoreIRTimer();
273#elif defined(ESP8266)
274 // tone() stops timer 1
275 (void) _pin;
276 (void) frequency;
277 (void) duration;
278#else
279 tone(_pin, frequency, duration);
280#endif
281}
