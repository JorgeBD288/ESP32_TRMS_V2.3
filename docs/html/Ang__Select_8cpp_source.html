138static void AngSelect_Event(lv_event_t * e)
139{
140 lv_event_code_t code = lv_event_get_code(e);
141 lv_obj_t * img = lv_event_get_target(e);
142
143 // Para depuración básica: ver si entra
144 if (code == LV_EVENT_PRESSED) {
145 Serial.println("[AngSelect] LV_EVENT_PRESSED");
146 } else if (code == LV_EVENT_PRESSING) {
147 Serial.println("[AngSelect] LV_EVENT_PRESSING");
148 } else if (code == LV_EVENT_RELEASED) {
149 Serial.println("[AngSelect] LV_EVENT_RELEASED");
150 } else {
151 return;
152 }
153
154 // Solo hacemos el cálculo cuando está presionado o arrastrando
155 if (code != LV_EVENT_PRESSED && code != LV_EVENT_PRESSING) {
156 return;
157 }
158
159 // 1) Posición del toque en coordenadas de pantalla
160 lv_point_t p;
161 lv_indev_t * indev = lv_indev_get_act();
162 if (!indev) {
163 Serial.println("[AngSelect] indev nullptr");
164 return;
165 }
166
167 lv_indev_get_point(indev, &p);
168 Serial.print("[AngSelect] Touch screen coords: x=");
169 Serial.print(p.x);
170 Serial.print(" y=");
171 Serial.println(p.y);
172
173 // 2) Coordenadas reales de la imagen en pantalla
174 lv_area_t img_area;
175 lv_obj_get_coords(img, &img_area); // x1,y1,x2,y2 en coordenadas de pantalla
176
177 // 3) Coordenadas locales dentro de la imagen
178 int local_x = p.x - img_area.x1;
179 int local_y = p.y - img_area.y1;
180
181 Serial.print("[AngSelect] Local in image: x=");
182 Serial.print(local_x);
183 Serial.print(" y=");
184 Serial.println(local_y);
185
186 // 4) Coordenadas relativas a la cuadrícula
187 int grid_x = local_x - GRID_X0;
188 int grid_y = local_y - GRID_Y0;
189
190 Serial.print("[AngSelect] In grid: x=");
191 Serial.print(grid_x);
192 Serial.print(" y=");
193 Serial.println(grid_y);
194
195 // 5) Si tocamos fuera de la cuadrícula, no hay nada que hacer
196 if (grid_x < 0 || grid_x >= GRID_W || grid_y < 0 || grid_y >= GRID_H) {
197 Serial.println("[AngSelect] Fuera de la cuadrícula");
198 return;
199 }
200
201 // 6) ACTUALIZAR GRAFICOS: punto rojo + líneas
202 AngSelect_UpdateGraphics(img, local_x, local_y);
203
204 // 7) Centro de la cuadrícula en píxeles (origen angular 0,0)
205 float cx = GRID_W / 2.0f;
206 float cy = GRID_H / 2.0f;
207
208 // 8) Desplazamiento relativo al centro (en píxeles)
209 float dx = (float)grid_x - cx; // + derecha
210 float dy = cy - (float)grid_y; // + arriba (invertimos Y)
211
212 // 9) Conversión a grados
213 float angle_h = dx * DEG_PER_PX_X; // eje horizontal
214 float angle_v = dy * DEG_PER_PX_Y; // eje vertical
215
216 Serial.print("[AngSelect] Angles: H=");
217 Serial.print(angle_h);
218 Serial.print(" V=");
219 Serial.println(angle_v);
220
221 // 10) Mostrar en labels
222 if (g_labelH) {
223 char buf[48];
224 snprintf(buf, sizeof(buf), "A. horizontal = %.1f°", angle_h);
225 lv_label_set_text(g_labelH, buf);
226 }
227
228 if (g_labelV) {
229 char buf[48];
230 snprintf(buf, sizeof(buf), "A. vertical = %.1f°", angle_v);
231 lv_label_set_text(g_labelV, buf);
232 }
233
234 // 11) Guardar como últimas referencias (en grados)
235 g_refH = angle_h;
236 g_refV = angle_v;

295void AngSelect_ApplyRefFromAngles(float angle_h, float angle_v)
296{
297 // Limitar a [-180, 180]
298 angle_h = constrain(angle_h, -180.0f, 180.0f);
299 angle_v = constrain(angle_v, -180.0f, 180.0f);
300
301 // Guardar referencias
302 g_refH = angle_h;
303 g_refV = angle_v;
304
305 // Actualizar labels
306 if (g_labelH) {
307 char buf[48];
308 snprintf(buf, sizeof(buf), "A. horizontal = %.1f°", g_refH);
309 lv_label_set_text(g_labelH, buf);
310 }
311 if (g_labelV) {
312 char buf[48];
313 snprintf(buf, sizeof(buf), "A. vertical = %.1f°", g_refV);
314 lv_label_set_text(g_labelV, buf);
315 }
316
317 // Actualizar gráficos (punto rojo + líneas)
318 if (!g_gridImage) return;
319
320 // Centro de la cuadrícula
321 float cx = GRID_W / 2.0f;
322 float cy = GRID_H / 2.0f;
323
324 // Inversa de lo que haces al tocar:
325 // angle_h = dx * DEG_PER_PX_X → dx = angle_h / DEG_PER_PX_X
326 // angle_v = dy * DEG_PER_PX_Y → dy = angle_v / DEG_PER_PX_Y
327 float dx = angle_h / DEG_PER_PX_X;
328 float dy = angle_v / DEG_PER_PX_Y;
329
330 float grid_x_f = cx + dx;
331 float grid_y_f = cy - dy;
332
333 // Asegurar dentro de la cuadrícula
334 if (grid_x_f < 0) grid_x_f = 0;
335 if (grid_x_f > GRID_W) grid_x_f = GRID_W;
336 if (grid_y_f < 0) grid_y_f = 0;
337 if (grid_y_f > GRID_H) grid_y_f = GRID_H;
338
339 int grid_x = (int)(grid_x_f + 0.5f);
340 int grid_y = (int)(grid_y_f + 0.5f);
341
342 int local_x = GRID_X0 + grid_x;
343 int local_y = GRID_Y0 + grid_y;
344
345 // Esto dibuja el punto y las líneas como cuando tocas
346 AngSelect_UpdateGraphics(g_gridImage, local_x, local_y);
347}
